18  package org.apache.hadoop.hbase.types;
19  
20  import org.apache.hadoop.hbase.classification.InterfaceAudience;
21  import org.apache.hadoop.hbase.classification.InterfaceStability;
22  import org.apache.hadoop.hbase.util.Bytes;
23  import org.apache.hadoop.hbase.util.Order;
24  import org.apache.hadoop.hbase.util.PositionedByteRange;
25  
26  /**
27   * An {@code DataType} for interacting with values encoded using
28   * {@link Bytes#putLong(byte[], int, long)}. Intended to make it easier to
29   * transition away from direct use of {@link Bytes}.
30   * @see Bytes#putLong(byte[], int, long)
31   * @see Bytes#toLong(byte[])
32   */
33  @InterfaceAudience.Public
34  @InterfaceStability.Evolving
35  public class RawLong implements DataType<Long> {
36  
37    @Override
38    public boolean isOrderPreserving() { return false; }
39  
40    @Override
41    public Order getOrder() { return null; }
42  
43    @Override
44    public boolean isNullable() { return false; }
45  
46    @Override
47    public boolean isSkippable() { return true; }
48  
49    @Override
50    public int encodedLength(Long val) { return Bytes.SIZEOF_LONG; }
51  
52    @Override
53    public Class<Long> encodedClass() { return Long.class; }
54  
55    @Override
56    public int skip(PositionedByteRange src) {
57      src.setPosition(src.getPosition() + Bytes.SIZEOF_LONG);
58      return Bytes.SIZEOF_LONG;
59    }
60  
61    @Override
62    public Long decode(PositionedByteRange src) {
63      long val = Bytes.toLong(src.getBytes(), src.getOffset() + src.getPosition());
64      skip(src);
65      return val;
66    }
67  
68    @Override
69    public int encode(PositionedByteRange dst, Long val) {
70      Bytes.putLong(dst.getBytes(), dst.getOffset() + dst.getPosition(), val);
71      return skip(dst);
72    }
73  
74    /**
75     * Read a {@code long} value from the buffer {@code buff}.
76     */
77    public long decodeLong(byte[] buff, int offset) {
78      return Bytes.toLong(buff, offset);
79    }
80  
81    /**
82     * Write instance {@code val} into buffer {@code buff}.
83     */
84    public int encodeLong(byte[] buff, int offset, long val) {
85      return Bytes.putLong(buff, offset, val);
86    }
87  }
