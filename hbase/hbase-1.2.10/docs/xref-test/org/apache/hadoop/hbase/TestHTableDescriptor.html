129     HTableDescriptor desc = new HTableDescriptor(TableName.valueOf("testGetSetRemoveCP"));
130     // simple CP
131     String className1 = "org.apache.hadoop.hbase.coprocessor.BaseRegionObserver";
132     String className2 = "org.apache.hadoop.hbase.coprocessor.SampleRegionWALObserver";
133     // Check that any coprocessor is present.
134     assertTrue(desc.getCoprocessors().size() == 0);
135 
136     // Add the 1 coprocessor and check if present.
137     desc.addCoprocessor(className1);
138     assertTrue(desc.getCoprocessors().size() == 1);
139     assertTrue(desc.getCoprocessors().contains(className1));
140 
141     // Add the 2nd coprocessor and check if present.
142     // remove it and check that it is gone
143     desc.addCoprocessor(className2);
144     assertTrue(desc.getCoprocessors().size() == 2);
145     assertTrue(desc.getCoprocessors().contains(className2));
146 
147     // Remove one and check
148     desc.removeCoprocessor(className1);
149     assertTrue(desc.getCoprocessors().size() == 1);
150     assertFalse(desc.getCoprocessors().contains(className1));
151     assertTrue(desc.getCoprocessors().contains(className2));
152 
153     // Remove the last and check
154     desc.removeCoprocessor(className2);
155     assertTrue(desc.getCoprocessors().size() == 0);
156     assertFalse(desc.getCoprocessors().contains(className1));
157     assertFalse(desc.getCoprocessors().contains(className2));
158   }
