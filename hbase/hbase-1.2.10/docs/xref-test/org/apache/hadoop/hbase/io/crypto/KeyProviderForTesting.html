17  package org.apache.hadoop.hbase.io.crypto;
18  
19  import java.security.Key;
20  import javax.crypto.spec.SecretKeySpec;
21  
22  /**
23   * Return a fixed secret key for AES for testing.
24   */
25  public class KeyProviderForTesting implements KeyProvider {
26  
27    @Override
28    public void init(String parameters) { }
29  
30    @Override
31    public Key getKey(String name) {
32      return new SecretKeySpec(Encryption.hash128(name), "AES");
33    }
34  
35    @Override
36    public Key[] getKeys(String[] aliases) {
37      Key[] result = new Key[aliases.length];
38      for (int i = 0; i < aliases.length; i++) {
39        result[i] = new SecretKeySpec(Encryption.hash128(aliases[i]), "AES");;
40      }
41      return result;
42    }
43  
44  }
