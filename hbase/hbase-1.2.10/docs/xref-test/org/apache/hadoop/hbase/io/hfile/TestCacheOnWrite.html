186   private void clearBlockCache(BlockCache blockCache) throws InterruptedException {
187     if (blockCache instanceof LruBlockCache) {
188       ((LruBlockCache) blockCache).clearCache();
189     } else {
190       // BucketCache may not return all cached blocks(blocks in write queue), so check it here.
191       for (int clearCount = 0; blockCache.getBlockCount() > 0; clearCount++) {
192         if (clearCount > 0) {
193           LOG.warn("clear block cache " + blockCache + " " + clearCount + " times, "
194               + blockCache.getBlockCount() + " blocks remaining");
195           Thread.sleep(10);
196         }
197         for (CachedBlock block : Lists.newArrayList(blockCache)) {
198           BlockCacheKey key = new BlockCacheKey(block.getFilename(), block.getOffset());
199           // CombinedBucketCache may need evict two times.
200           for (int evictCount = 0; blockCache.evictBlock(key); evictCount++) {
201             if (evictCount > 1) {
202               LOG.warn("evict block " + block + " in " + blockCache + " " + evictCount
203                   + " times, maybe a bug here");
204             }
205           }
206         }
207       }
208     }
209   }

245   private void readStoreFile(boolean useTags) throws IOException {
246     AbstractHFileReader reader;
247     if (useTags) {
248         reader = (HFileReaderV3) HFile.createReader(fs, storeFilePath, cacheConf, conf);
249     } else {
250         reader = (HFileReaderV2) HFile.createReader(fs, storeFilePath, cacheConf, conf);
251     }
252     LOG.info("HFile information: " + reader);
253     HFileContext meta = new HFileContextBuilder().withCompression(compress)
254       .withBytesPerCheckSum(CKBYTES).withChecksumType(ChecksumType.NULL)
255       .withBlockSize(DATA_BLOCK_SIZE)
256       .withDataBlockEncoding(NoOpDataBlockEncoder.INSTANCE.getDataBlockEncoding())
257       .withIncludesTags(useTags).build();
258     final boolean cacheBlocks = false;
259     final boolean pread = false;
260     HFileScanner scanner = reader.getScanner(cacheBlocks, pread);
261     assertTrue(testDescription, scanner.seekTo());
262 
263     long offset = 0;
264     HFileBlock prevBlock = null;
265     EnumMap<BlockType, Integer> blockCountByType =
266         new EnumMap<BlockType, Integer>(BlockType.class);
267 
268     DataBlockEncoding encodingInCache = NoOpDataBlockEncoder.INSTANCE.getDataBlockEncoding();
269     while (offset < reader.getTrailer().getLoadOnOpenDataOffset()) {
270       long onDiskSize = -1;
271       if (prevBlock != null) {
272          onDiskSize = prevBlock.getNextBlockOnDiskSizeWithHeader();
273       }
274       // Flags: don't cache the block, use pread, this is not a compaction.
275       // Also, pass null for expected block type to avoid checking it.
276       HFileBlock block = reader.readBlock(offset, onDiskSize, false, true,
277         false, true, null, encodingInCache);
278       BlockCacheKey blockCacheKey = new BlockCacheKey(reader.getName(),
279           offset);
280       HFileBlock fromCache = (HFileBlock) blockCache.getBlock(blockCacheKey, true, false, true);
281       boolean isCached = fromCache != null;
282       boolean shouldBeCached = cowType.shouldBeCached(block.getBlockType());
283       assertTrue("shouldBeCached: " + shouldBeCached+ "\n" +
284           "isCached: " + isCached + "\n" +
285           "Test description: " + testDescription + "\n" +
286           "block: " + block + "\n" +
287           "encodingInCache: " + encodingInCache + "\n" +
288           "blockCacheKey: " + blockCacheKey,
289         shouldBeCached == isCached);
290       if (isCached) {
291         if (cacheConf.shouldCacheCompressed(fromCache.getBlockType().getCategory())) {
292           if (compress != Compression.Algorithm.NONE) {
293             assertFalse(fromCache.isUnpacked());
294           }
295           fromCache = fromCache.unpack(meta, reader.getUncachedBlockReader());
296         } else {
297           assertTrue(fromCache.isUnpacked());
298         }
299         // block we cached at write-time and block read from file should be identical
300         assertEquals(block.getChecksumType(), fromCache.getChecksumType());
301         assertEquals(block.getBlockType(), fromCache.getBlockType());
302         assertNotEquals(block.getBlockType(), BlockType.ENCODED_DATA);
303         assertEquals(block.getOnDiskSizeWithHeader(), fromCache.getOnDiskSizeWithHeader());
304         assertEquals(block.getOnDiskSizeWithoutHeader(), fromCache.getOnDiskSizeWithoutHeader());
305         assertEquals(
306           block.getUncompressedSizeWithoutHeader(), fromCache.getUncompressedSizeWithoutHeader());
307       }
308       prevBlock = block;
309       offset += block.getOnDiskSizeWithHeader();
310       BlockType bt = block.getBlockType();
311       Integer count = blockCountByType.get(bt);
312       blockCountByType.put(bt, (count == null ? 0 : count) + 1);
313     }
314 
315     LOG.info("Block count by type: " + blockCountByType);
316     String countByType = blockCountByType.toString();
317     if (useTags) {
318       assertEquals("{" + BlockType.DATA
319           + "=2663, LEAF_INDEX=297, BLOOM_CHUNK=9, INTERMEDIATE_INDEX=32}", countByType);
320     } else {
321       assertEquals("{" + BlockType.DATA
322           + "=2498, LEAF_INDEX=278, BLOOM_CHUNK=9, INTERMEDIATE_INDEX=31}", countByType);
323     }
324 
325     // iterate all the keyvalue from hfile
326     while (scanner.next()) {
327       scanner.getKeyValue();
328     }
329     reader.close();
330   }

332   public static KeyValue.Type generateKeyType(Random rand) {
333     if (rand.nextBoolean()) {
334       // Let's make half of KVs puts.
335       return KeyValue.Type.Put;
336     } else {
337       KeyValue.Type keyType = KeyValue.Type.values()[1 + rand.nextInt(NUM_VALID_KEY_TYPES)];
338       if (keyType == KeyValue.Type.Minimum || keyType == KeyValue.Type.Maximum) {
339         throw new RuntimeException("Generated an invalid key type: " + keyType + ". "
340             + "Probably the layout of KeyValue.Type has changed.");
341       }
342       return keyType;
343     }
344   }

346   private void writeStoreFile(boolean useTags) throws IOException {
347     if(useTags) {
348       TEST_UTIL.getConfiguration().setInt("hfile.format.version", 3);
349     } else {
350       TEST_UTIL.getConfiguration().setInt("hfile.format.version", 2);
351     }
352     Path storeFileParentDir = new Path(TEST_UTIL.getDataTestDir(),
353         "test_cache_on_write");
354     HFileContext meta = new HFileContextBuilder().withCompression(compress)
355         .withBytesPerCheckSum(CKBYTES).withChecksumType(ChecksumType.NULL)
356         .withBlockSize(DATA_BLOCK_SIZE)
357         .withDataBlockEncoding(NoOpDataBlockEncoder.INSTANCE.getDataBlockEncoding())
358         .withIncludesTags(useTags).build();
359     StoreFile.Writer sfw = new StoreFile.WriterBuilder(conf, cacheConf, fs)
360         .withOutputDir(storeFileParentDir).withComparator(KeyValue.COMPARATOR)
361         .withFileContext(meta)
362         .withBloomType(BLOOM_TYPE).withMaxKeyCount(NUM_KV).build();
363     byte[] cf = Bytes.toBytes("fam");
364     for (int i = 0; i < NUM_KV; ++i) {
365       byte[] row = TestHFileWriterV2.randomOrderedKey(rand, i);
366       byte[] qualifier = TestHFileWriterV2.randomRowOrQualifier(rand);
367       byte[] value = TestHFileWriterV2.randomValue(rand);
368       KeyValue kv;
369       if(useTags) {
370         Tag t = new Tag((byte) 1, "visibility");
371         List<Tag> tagList = new ArrayList<Tag>();
372         tagList.add(t);
373         Tag[] tags = new Tag[1];
374         tags[0] = t;
375         kv =
376             new KeyValue(row, 0, row.length, cf, 0, cf.length, qualifier, 0, qualifier.length,
377                 rand.nextLong(), generateKeyType(rand), value, 0, value.length, tagList);
378       } else {
379         kv =
380             new KeyValue(row, 0, row.length, cf, 0, cf.length, qualifier, 0, qualifier.length,
381                 rand.nextLong(), generateKeyType(rand), value, 0, value.length);
382       }
383       sfw.append(kv);
384     }
385 
386     sfw.close();
387     storeFilePath = sfw.getPath();
388   }

390   private void testNotCachingDataBlocksDuringCompactionInternals(boolean useTags)
391       throws IOException, InterruptedException {
392     if (useTags) {
393       TEST_UTIL.getConfiguration().setInt("hfile.format.version", 3);
394     } else {
395       TEST_UTIL.getConfiguration().setInt("hfile.format.version", 2);
396     }
397     // TODO: need to change this test if we add a cache size threshold for
398     // compactions, or if we implement some other kind of intelligent logic for
399     // deciding what blocks to cache-on-write on compaction.
400     final String table = "CompactionCacheOnWrite";
401     final String cf = "myCF";
402     final byte[] cfBytes = Bytes.toBytes(cf);
403     final int maxVersions = 3;
404     Region region = TEST_UTIL.createTestRegion(table, 
405         new HColumnDescriptor(cf)
406             .setCompressionType(compress)
407             .setBloomFilterType(BLOOM_TYPE)
408             .setMaxVersions(maxVersions)
409             .setDataBlockEncoding(NoOpDataBlockEncoder.INSTANCE.getDataBlockEncoding())
410     );
411     int rowIdx = 0;
412     long ts = EnvironmentEdgeManager.currentTime();
413     for (int iFile = 0; iFile < 5; ++iFile) {
414       for (int iRow = 0; iRow < 500; ++iRow) {
415         String rowStr = "" + (rowIdx * rowIdx * rowIdx) + "row" + iFile + "_" + 
416             iRow;
417         Put p = new Put(Bytes.toBytes(rowStr));
418         ++rowIdx;
419         for (int iCol = 0; iCol < 10; ++iCol) {
420           String qualStr = "col" + iCol;
421           String valueStr = "value_" + rowStr + "_" + qualStr;
422           for (int iTS = 0; iTS < 5; ++iTS) {
423             if (useTags) {
424               Tag t = new Tag((byte) 1, "visibility");
425               Tag[] tags = new Tag[1];
426               tags[0] = t;
427               KeyValue kv = new KeyValue(Bytes.toBytes(rowStr), cfBytes, Bytes.toBytes(qualStr),
428                   HConstants.LATEST_TIMESTAMP, Bytes.toBytes(valueStr), tags);
429               p.add(kv);
430             } else {
431               p.addColumn(cfBytes, Bytes.toBytes(qualStr), ts++, Bytes.toBytes(valueStr));
432             }
433           }
434         }
435         p.setDurability(Durability.ASYNC_WAL);
436         region.put(p);
437       }
438       region.flush(true);
439     }
440     clearBlockCache(blockCache);
441     assertEquals(0, blockCache.getBlockCount());
442     region.compact(false);
443     LOG.debug("compactStores() returned");
444 
445     for (CachedBlock block: blockCache) {
446       assertNotEquals(BlockType.ENCODED_DATA, block.getBlockType());
447       assertNotEquals(BlockType.DATA, block.getBlockType());
448     }
449     ((HRegion)region).close();
450   }
