58    /**
59     * Basic sanity test that spins up multiple HDFS and HBase clusters that share
60     * the same ZK ensemble. We then create the same table in both and make sure
61     * that what we insert in one place doesn't end up in the other.
62     * @throws Exception
63     */
64    @Test (timeout=180000)
65    public void testMultiClusters() throws Exception {
66      // Create three clusters
67  
68      // Cluster 1.
69      HBaseTestingUtility htu1 = new HBaseTestingUtility();
70      // Set a different zk path for each cluster
71      htu1.getConfiguration().set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/1");
72      htu1.startMiniZKCluster();
73  
74      // Cluster 2
75      HBaseTestingUtility htu2 = new HBaseTestingUtility();
76      htu2.getConfiguration().set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/2");
77      htu2.getConfiguration().set(HConstants.ZOOKEEPER_CLIENT_PORT,
78        htu1.getConfiguration().get(HConstants.ZOOKEEPER_CLIENT_PORT, "-1"));
79      htu2.setZkCluster(htu1.getZkCluster());
80  
81      // Cluster 3; seed it with the conf from htu1 so we pickup the 'right'
82      // zk cluster config; it is set back into the config. as part of the
83      // start of minizkcluster.
84      HBaseTestingUtility htu3 = new HBaseTestingUtility();
85      htu3.getConfiguration().set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/3");
86      htu3.getConfiguration().set(HConstants.ZOOKEEPER_CLIENT_PORT,
87        htu1.getConfiguration().get(HConstants.ZOOKEEPER_CLIENT_PORT, "-1"));
88      htu3.setZkCluster(htu1.getZkCluster());
89  
90      try {
91        htu1.startMiniCluster();
92        htu2.startMiniCluster();
93        htu3.startMiniCluster();
94  
95        final TableName TABLE_NAME = TableName.valueOf("test");
96        final byte[] FAM_NAME = Bytes.toBytes("fam");
97        final byte[] ROW = Bytes.toBytes("row");
98        final byte[] QUAL_NAME = Bytes.toBytes("qual");
99        final byte[] VALUE = Bytes.toBytes("value");
100 
101       Table table1 = htu1.createTable(TABLE_NAME, FAM_NAME);
102       Table table2 = htu2.createTable(TABLE_NAME, FAM_NAME);
103 
104       Put put = new Put(ROW);
105       put.add(FAM_NAME, QUAL_NAME, VALUE);
106       table1.put(put);
107 
108       Get get = new Get(ROW);
109       get.addColumn(FAM_NAME, QUAL_NAME);
110       Result res = table1.get(get);
111       assertEquals(1, res.size());
112 
113       res = table2.get(get);
114       assertEquals(0, res.size());
115 
116       table1.close();
117       table2.close();
118 
119     } finally {
120       htu3.shutdownMiniCluster();
121       htu2.shutdownMiniCluster();
122       htu1.shutdownMiniCluster();
123     }
124   }

207   @Test
208   public void testMiniZooKeeperWithMultipleServers() throws Exception {
209     HBaseTestingUtility hbt = new HBaseTestingUtility();
210     // set up zookeeper cluster with 5 zk servers
211     MiniZooKeeperCluster cluster2 = hbt.startMiniZKCluster(5);
212     int defaultClientPort = 21818;
213     cluster2.setDefaultClientPort(defaultClientPort);
214     try {
215       assertEquals(4, cluster2.getBackupZooKeeperServerNum());
216 
217       // killing the current active zk server
218       int currentActivePort = cluster2.killCurrentActiveZooKeeperServer();
219       assertTrue(currentActivePort >= defaultClientPort);
220       // Check if the client port is returning a proper value
221       assertTrue(cluster2.getClientPort() == currentActivePort);
222 
223       // kill another active zk server
224       currentActivePort = cluster2.killCurrentActiveZooKeeperServer();
225       assertTrue(currentActivePort >= defaultClientPort);
226       assertTrue(cluster2.getClientPort() == currentActivePort);      
227       assertEquals(2, cluster2.getBackupZooKeeperServerNum());
228       assertEquals(3, cluster2.getZooKeeperServerNum());
229 
230       // killing the backup zk servers
231       cluster2.killOneBackupZooKeeperServer();
232       cluster2.killOneBackupZooKeeperServer();
233       assertEquals(0, cluster2.getBackupZooKeeperServerNum());
234       assertEquals(1, cluster2.getZooKeeperServerNum());
235 
236       // killing the last zk server
237       currentActivePort = cluster2.killCurrentActiveZooKeeperServer();
238       assertTrue(currentActivePort == -1);
239       assertTrue(cluster2.getClientPort() == currentActivePort);
240       // this should do nothing.
241       cluster2.killOneBackupZooKeeperServer();
242       assertEquals(-1, cluster2.getBackupZooKeeperServerNum());
243       assertEquals(0, cluster2.getZooKeeperServerNum());
244     } finally {
245       hbt.shutdownMiniZKCluster();
246     }
247   }

249   @Test
250   public void testMiniZooKeeperWithMultipleClientPorts() throws Exception {
251     int defaultClientPort = 8888;
252     int i, j;
253     HBaseTestingUtility hbt = new HBaseTestingUtility();
254 
255     // Test 1 - set up zookeeper cluster with same number of ZK servers and specified client ports
256     int [] clientPortList1 = {1111, 1112, 1113};
257     MiniZooKeeperCluster cluster1 = hbt.startMiniZKCluster(clientPortList1.length, clientPortList1);
258     try {
259       List<Integer> clientPortListInCluster = cluster1.getClientPortList();
260 
261       for (i = 0; i < clientPortListInCluster.size(); i++) {
262         assertEquals(clientPortListInCluster.get(i).intValue(), clientPortList1[i]);
263       }
264     } finally {
265       hbt.shutdownMiniZKCluster();
266     }
267 
268     // Test 2 - set up zookeeper cluster with more ZK servers than specified client ports
269     hbt.getConfiguration().setInt("test.hbase.zookeeper.property.clientPort", defaultClientPort);
270     int [] clientPortList2 = {2222, 2223};
271     MiniZooKeeperCluster cluster2 =
272         hbt.startMiniZKCluster(clientPortList2.length + 2, clientPortList2);
273 
274     try {
275       List<Integer> clientPortListInCluster = cluster2.getClientPortList();
276 
277       for (i = 0, j = 0; i < clientPortListInCluster.size(); i++) {
278         if (i < clientPortList2.length) {
279           assertEquals(clientPortListInCluster.get(i).intValue(), clientPortList2[i]);
280         } else {
281           // servers with no specified client port will use defaultClientPort or some other ports
282           // based on defaultClientPort
283           assertEquals(clientPortListInCluster.get(i).intValue(), defaultClientPort + j);
284           j++;
285         }
286       }
287     } finally {
288       hbt.shutdownMiniZKCluster();
289     }
290 
291     // Test 3 - set up zookeeper cluster with invalid client ports
292     hbt.getConfiguration().setInt("test.hbase.zookeeper.property.clientPort", defaultClientPort);
293     int [] clientPortList3 = {3333, -3334, 3335, 0};
294     MiniZooKeeperCluster cluster3 =
295         hbt.startMiniZKCluster(clientPortList3.length + 1, clientPortList3);
296 
297     try {
298       List<Integer> clientPortListInCluster = cluster3.getClientPortList();
299 
300       for (i = 0, j = 0; i < clientPortListInCluster.size(); i++) {
301         // Servers will only use valid client ports; if ports are not specified or invalid,
302         // the default port or a port based on default port will be used.
303         if (i < clientPortList3.length && clientPortList3[i] > 0) {
304           assertEquals(clientPortListInCluster.get(i).intValue(), clientPortList3[i]);
305         } else {
306           assertEquals(clientPortListInCluster.get(i).intValue(), defaultClientPort + j);
307           j++;
308         }
309       }
310     } finally {
311       hbt.shutdownMiniZKCluster();
312     }
313 
314     // Test 4 - set up zookeeper cluster with default port and some other ports used
315     // This test tests that the defaultClientPort and defaultClientPort+2 are used, so
316     // the algorithm should choice defaultClientPort+1 and defaultClientPort+3 to fill
317     // out the ports for servers without ports specified.
318     hbt.getConfiguration().setInt("test.hbase.zookeeper.property.clientPort", defaultClientPort);
319     int [] clientPortList4 = {-4444, defaultClientPort+2, 4446, defaultClientPort};
320     MiniZooKeeperCluster cluster4 =
321         hbt.startMiniZKCluster(clientPortList4.length + 1, clientPortList4);
322 
323     try {
324       List<Integer> clientPortListInCluster = cluster4.getClientPortList();
325 
326       for (i = 0, j = 1; i < clientPortListInCluster.size(); i++) {
327         // Servers will only use valid client ports; if ports are not specified or invalid,
328         // the default port or a port based on default port will be used.
329         if (i < clientPortList4.length && clientPortList4[i] > 0) {
330           assertEquals(clientPortListInCluster.get(i).intValue(), clientPortList4[i]);
331         } else {
332           assertEquals(clientPortListInCluster.get(i).intValue(), defaultClientPort + j);
333           j +=2;
334         }
335       }
336     } finally {
337       hbt.shutdownMiniZKCluster();
338     }
339 
340     // Test 5 - set up zookeeper cluster with same ports specified - fail is expected.
341     int [] clientPortList5 = {5555, 5556, 5556};
342 
343     try {
344       MiniZooKeeperCluster cluster5 =
345           hbt.startMiniZKCluster(clientPortList5.length, clientPortList5);
346       assertTrue(cluster5.getClientPort() == -1); // expected failure
347     } catch (Exception e) {
348       // exception is acceptable
349     } finally {
350       hbt.shutdownMiniZKCluster();
351     }
352   }

392   @Test public void testMRYarnConfigsPopulation() throws IOException {
393     Map<String, String> dummyProps = new HashMap<>();
394     dummyProps.put("mapreduce.jobtracker.address", "dummyhost:11234");
395     dummyProps.put("yarn.resourcemanager.address", "dummyhost:11235");
396     dummyProps.put("mapreduce.jobhistory.address", "dummyhost:11236");
397     dummyProps.put("yarn.resourcemanager.scheduler.address", "dummyhost:11237");
398     dummyProps.put("mapreduce.jobhistory.webapp.address", "dummyhost:11238");
399     dummyProps.put("yarn.resourcemanager.webapp.address", "dummyhost:11239");
400   
401     HBaseTestingUtility hbt = new HBaseTestingUtility();
402     
403     // populate the mr props to the Configuration instance
404     for (Entry<String, String> entry : dummyProps.entrySet()) {
405       hbt.getConfiguration().set(entry.getKey(), entry.getValue());
406     }
407     
408     for (Entry<String,String> entry : dummyProps.entrySet()) {
409       assertTrue("The Configuration for key " + entry.getKey() +" and value: " + entry.getValue() +
410                  " is not populated correctly", hbt.getConfiguration().get(entry.getKey()).equals(entry.getValue()));
411     }
412 
413     hbt.startMiniMapReduceCluster();
414     
415     // Confirm that MiniMapReduceCluster overwrites the mr properties and updates the Configuration 
416     for (Entry<String,String> entry : dummyProps.entrySet()) {
417       assertFalse("The MR prop: " + entry.getValue() + " is not overwritten when map reduce mini"+
418                   "cluster is started", hbt.getConfiguration().get(entry.getKey()).equals(entry.getValue()));
419     }
420     
421     hbt.shutdownMiniMapReduceCluster();
422   }
