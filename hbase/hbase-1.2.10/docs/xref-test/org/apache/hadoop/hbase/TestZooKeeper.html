348   /**
349    * A test for HBASE-3238
350    * @throws IOException A connection attempt to zk failed
351    * @throws InterruptedException One of the non ZKUtil actions was interrupted
352    * @throws KeeperException Any of the zookeeper connections had a
353    * KeeperException
354    */
355   @Test
356   public void testCreateSilentIsReallySilent() throws InterruptedException,
357       KeeperException, IOException {
358     Configuration c = TEST_UTIL.getConfiguration();
359 
360     String aclZnode = "/aclRoot";
361     String quorumServers = ZKConfig.getZKQuorumServersString(c);
362     int sessionTimeout = 5 * 1000; // 5 seconds
363     ZooKeeper zk = new ZooKeeper(quorumServers, sessionTimeout, EmptyWatcher.instance);
364     zk.addAuthInfo("digest", "hbase:rox".getBytes());
365 
366     // Assumes the  root of the ZooKeeper space is writable as it creates a node
367     // wherever the cluster home is defined.
368     ZooKeeperWatcher zk2 = new ZooKeeperWatcher(TEST_UTIL.getConfiguration(),
369       "testCreateSilentIsReallySilent", null);
370 
371     // Save the previous ACL
372     Stat s =  null;
373     List<ACL> oldACL = null;
374     while (true) {
375       try {
376         s = new Stat();
377         oldACL = zk.getACL("/", s);
378         break;
379       } catch (KeeperException e) {
380         switch (e.code()) {
381           case CONNECTIONLOSS:
382           case SESSIONEXPIRED:
383           case OPERATIONTIMEOUT:
384             LOG.warn("Possibly transient ZooKeeper exception", e);
385             Threads.sleep(100);
386             break;
387          default:
388             throw e;
389         }
390       }
391     }
392 
393     // I set this acl after the attempted creation of the cluster home node.
394     // Add retries in case of retryable zk exceptions.
395     while (true) {
396       try {
397         zk.setACL("/", ZooDefs.Ids.CREATOR_ALL_ACL, -1);
398         break;
399       } catch (KeeperException e) {
400         switch (e.code()) {
401           case CONNECTIONLOSS:
402           case SESSIONEXPIRED:
403           case OPERATIONTIMEOUT:
404             LOG.warn("Possibly transient ZooKeeper exception: " + e);
405             Threads.sleep(100);
406             break;
407          default:
408             throw e;
409         }
410       }
411     }
412 
413     while (true) {
414       try {
415         zk.create(aclZnode, null, ZooDefs.Ids.CREATOR_ALL_ACL, CreateMode.PERSISTENT);
416         break;
417       } catch (KeeperException e) {
418         switch (e.code()) {
419           case CONNECTIONLOSS:
420           case SESSIONEXPIRED:
421           case OPERATIONTIMEOUT:
422             LOG.warn("Possibly transient ZooKeeper exception: " + e);
423             Threads.sleep(100);
424             break;
425          default:
426             throw e;
427         }
428       }
429     }
430     zk.close();
431     ZKUtil.createAndFailSilent(zk2, aclZnode);
432 
433     // Restore the ACL
434     ZooKeeper zk3 = new ZooKeeper(quorumServers, sessionTimeout, EmptyWatcher.instance);
435     zk3.addAuthInfo("digest", "hbase:rox".getBytes());
436     try {
437       zk3.setACL("/", oldACL, -1);
438     } finally {
439       zk3.close();
440     }

456   /**
457    * Tests that the master does not call retainAssignment after recovery from expired zookeeper
458    * session. Without the HBASE-6046 fix master always tries to assign all the user regions by
459    * calling retainAssignment.
460    */
461   @Test(timeout = 300000)
462   public void testRegionAssignmentAfterMasterRecoveryDueToZKExpiry() throws Exception {
463     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
464     cluster.startRegionServer();
465     cluster.waitForActiveAndReadyMaster(10000);
466     HMaster m = cluster.getMaster();
467     final ZooKeeperWatcher zkw = m.getZooKeeper();
468     // now the cluster is up. So assign some regions.
469     try (Admin admin = TEST_UTIL.getHBaseAdmin()) {
470       byte[][] SPLIT_KEYS = new byte[][] { Bytes.toBytes("a"), Bytes.toBytes("b"),
471           Bytes.toBytes("c"), Bytes.toBytes("d"), Bytes.toBytes("e"), Bytes.toBytes("f"),
472           Bytes.toBytes("g"), Bytes.toBytes("h"), Bytes.toBytes("i"), Bytes.toBytes("j") };
473       String tableName = "testRegionAssignmentAfterMasterRecoveryDueToZKExpiry";
474       HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(tableName));
475       htd.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
476       admin.createTable(htd, SPLIT_KEYS);
477       TEST_UTIL.waitUntilNoRegionsInTransition(60000);
478       m.getZooKeeper().close();
479       MockLoadBalancer.retainAssignCalled = false;
480       final int expectedNumOfListeners = countPermanentListeners(zkw);
481       m.abort("Test recovery from zk session expired",
482           new KeeperException.SessionExpiredException());
483       assertTrue(m.isStopped()); // Master doesn't recover any more
484       // The recovered master should not call retainAssignment, as it is not a
485       // clean startup.
486       assertFalse("Retain assignment should not be called", MockLoadBalancer.retainAssignCalled);
487       // number of listeners should be same as the value before master aborted
488       // wait for new master is initialized
489       cluster.waitForActiveAndReadyMaster(120000);
490       final HMaster newMaster = cluster.getMasterThread().getMaster();
491       assertEquals(expectedNumOfListeners, countPermanentListeners(newMaster.getZooKeeper()));
492     }
493   }

495   /**
496    * Count listeners in zkw excluding listeners, that belongs to workers or other
497    * temporary processes.
498    */
499   private int countPermanentListeners(ZooKeeperWatcher watcher) {
500     return countListeners(watcher, ZkSplitLogWorkerCoordination.class);
501   }
502 
503   /**
504    * Count listeners in zkw excluding provided classes
505    */
506   private int countListeners(ZooKeeperWatcher watcher, Class<?>... exclude) {
507     int cnt = 0;
508     for (Object o : watcher.getListeners()) {
509       boolean skip = false;
510       for (Class<?> aClass : exclude) {
511         if (aClass.isAssignableFrom(o.getClass())) {
512           skip = true;
513           break;
514         }
515       }
516       if (!skip) {
517         cnt += 1;
518       }
519     }
520     return cnt;
521   }

541       String tableName = "testLogSplittingAfterMasterRecoveryDueToZKExpiry";

547       ZKAssign.blockUntilNoRIT(zooKeeperWatcher);
548       table = new HTable(TEST_UTIL.getConfiguration(), htd.getTableName());
549       Put p;
550       int numberOfPuts;
551       for (numberOfPuts = 0; numberOfPuts < 6; numberOfPuts++) {

554         table.put(p);
555       }
556       m.getZooKeeper().close();
557       m.abort("Test recovery from zk session expired",
558         new KeeperException.SessionExpiredException());
559       assertTrue(m.isStopped()); // Master doesn't recover any more
560       cluster.getRegionServer(0).abort("Aborting");
561       // Without patch for HBASE-6046 this test case will always timeout
562       // with patch the test case should pass.
563       Scan scan = new Scan();
564       int numberOfRows = 0;
565       ResultScanner scanner = table.getScanner(scan);
566       Result[] result = scanner.next(1);
567       while (result != null && result.length > 0) {
568         numberOfRows++;
569         result = scanner.next(1);
570       }
571       assertEquals("Number of rows should be equal to number of puts.", numberOfPuts,
572         numberOfRows);
573     } finally {
574       if (table != null) table.close();
575       admin.close();
576     }
