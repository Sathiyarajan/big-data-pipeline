214    * Simulates splitting a WAL out from under a regionserver that is still trying to write it.
215    * Ensures we do not lose edits.
216    * @throws IOException
217    * @throws InterruptedException
218    */
219   @Test (timeout=300000)
220   public void testLogCannotBeWrittenOnceParsed() throws IOException, InterruptedException {
221     final AtomicLong counter = new AtomicLong(0);
222     AtomicBoolean stop = new AtomicBoolean(false);
223     // Region we'll write edits too and then later examine to make sure they all made it in.
224     final String region = REGIONS.get(0);
225     final int numWriters = 3;
226     Thread zombie = new ZombieLastLogWriterRegionServer(counter, stop, region, numWriters);
227     try {
228       long startCount = counter.get();
229       zombie.start();
230       // Wait till writer starts going.
231       while (startCount == counter.get()) Threads.sleep(1);
232       // Give it a second to write a few appends.
233       Threads.sleep(1000);
234       final Configuration conf2 = HBaseConfiguration.create(this.conf);
235       final User robber = User.createUserForTesting(conf2, ROBBER, GROUP);
236       int count = robber.runAs(new PrivilegedExceptionAction<Integer>() {
237         @Override
238         public Integer run() throws Exception {
239           StringBuilder ls = new StringBuilder("Contents of WALDIR (").append(WALDIR)
240               .append("):\n");
241           for (FileStatus status : fs.listStatus(WALDIR)) {
242             ls.append("\t").append(status.toString()).append("\n");
243           }
244           LOG.debug(ls);
245           LOG.info("Splitting WALs out from under zombie. Expecting " + numWriters + " files.");
246           WALSplitter.split(HBASEDIR, WALDIR, OLDLOGDIR, fs, conf2, wals);
247           LOG.info("Finished splitting out from under zombie.");
248           Path[] logfiles = getLogForRegion(HBASEDIR, TABLE_NAME, region);
249           assertEquals("wrong number of split files for region", numWriters, logfiles.length);
250           int count = 0;
251           for (Path logfile: logfiles) {
252             count += countWAL(logfile);
253           }
254           return count;
255         }
256       });
257       LOG.info("zombie=" + counter.get() + ", robber=" + count);
258       assertTrue("The log file could have at most 1 extra log entry, but can't have less. " +
259               "Zombie could write " + counter.get() + " and logfile had only " + count,
260           counter.get() == count || counter.get() + 1 == count);
261     } finally {
262       stop.set(true);
263       zombie.interrupt();
264       Threads.threadDumpingIsAlive(zombie);
265     }
266   }

268   /**
269    * This thread will keep writing to a 'wal' file even after the split process has started.
270    * It simulates a region server that was considered dead but woke up and wrote some more to the
271    * last log entry. Does its writing as an alternate user in another filesystem instance to
272    * simulate better it being a regionserver.
273    */
274   class ZombieLastLogWriterRegionServer extends Thread {
275     final AtomicLong editsCount;
276     final AtomicBoolean stop;
277     final int numOfWriters;
278     /**
279      * Region to write edits for.
280      */
281     final String region;
282     final User user;
283 
284     public ZombieLastLogWriterRegionServer(AtomicLong counter, AtomicBoolean stop,
285         final String region, final int writers)
286         throws IOException, InterruptedException {
287       super("ZombieLastLogWriterRegionServer");
288       setDaemon(true);
289       this.stop = stop;
290       this.editsCount = counter;
291       this.region = region;
292       this.user = User.createUserForTesting(conf, ZOMBIE, GROUP);
293       numOfWriters = writers;
294     }
295 
296     @Override
297     public void run() {
298       try {
299         doWriting();
300       } catch (IOException e) {
301         LOG.warn(getName() + " Writer exiting " + e);
302       } catch (InterruptedException e) {
303         LOG.warn(getName() + " Writer exiting " + e);
304       }
305     }
306 
307     private void doWriting() throws IOException, InterruptedException {
308       this.user.runAs(new PrivilegedExceptionAction<Object>() {
309         @Override
310         public Object run() throws Exception {
311           // Index of the WAL we want to keep open.  generateWALs will leave open the WAL whose
312           // index we supply here.
313           int walToKeepOpen = numOfWriters - 1;
314           // The below method writes numOfWriters files each with ENTRIES entries for a total of
315           // numOfWriters * ENTRIES added per column family in the region.
316           Writer writer = null;
317           try {
318             writer = generateWALs(numOfWriters, ENTRIES, walToKeepOpen);
319           } catch (IOException e1) {
320             throw new RuntimeException("Failed", e1);
321           }
322           // Update counter so has all edits written so far.
323           editsCount.addAndGet(numOfWriters * ENTRIES);
324           loop(writer);
325           // If we've been interruped, then things should have shifted out from under us.
326           // closing should error
327           try {
328             writer.close();
329             fail("Writing closing after parsing should give an error.");
330           } catch (IOException exception) {
331             LOG.debug("ignoring error when closing final writer.", exception);
332           }
333           return null;
334         }
335       });
336     }
337 
338     private void loop(final Writer writer) {
339       byte [] regionBytes = Bytes.toBytes(this.region);
340       while (!stop.get()) {
341         try {
342           long seq = appendEntry(writer, TABLE_NAME, regionBytes,
343               ("r" + editsCount.get()).getBytes(), regionBytes, QUALIFIER, VALUE, 0);
344           long count = editsCount.incrementAndGet();
345           LOG.info(getName() + " sync count=" + count + ", seq=" + seq);
346           try {
347             Thread.sleep(1);
348           } catch (InterruptedException e) {
349             //
350           }
351         } catch (IOException ex) {
352           LOG.error(getName() + " ex " + ex.toString());
353           if (ex instanceof RemoteException) {
354             LOG.error("Juliet: got RemoteException " + ex.getMessage() +
355                 " while writing " + (editsCount.get() + 1));
356           } else {
357             LOG.error(getName() + " failed to write....at " + editsCount.get());
358             fail("Failed to write " + editsCount.get());
359           }
360           break;
361         } catch (Throwable t) {
362           LOG.error(getName() + " HOW? " + t);
363           LOG.debug("exception details", t);
364           break;
365         }
366       }
367       LOG.info(getName() + " Writer exiting");
368     }
369   }

490   /**
491    * @param expectedEntries -1 to not assert
492    * @return the count across all regions
493    */
494   private int splitAndCount(final int expectedFiles, final int expectedEntries)
495       throws IOException {
496     useDifferentDFSClient();
497     WALSplitter.split(HBASEDIR, WALDIR, OLDLOGDIR, fs, conf, wals);
498     int result = 0;
499     for (String region : REGIONS) {
500       Path[] logfiles = getLogForRegion(HBASEDIR, TABLE_NAME, region);
501       assertEquals(expectedFiles, logfiles.length);
502       int count = 0;
503       for (Path logfile: logfiles) {
504         count += countWAL(logfile);
505       }
506       if (-1 != expectedEntries) {
507         assertEquals(expectedEntries, count);
508       }
509       result += count;
510     }
511     return result;
512   }

593   /**
594    * @return set of wal names present prior to split attempt.
595    * @throws IOException if the split process fails
596    */
597   private Set<String> splitCorruptWALs(final FaultySequenceFileLogReader.FailureType failureType)
598       throws IOException {
599     Class<?> backupClass = conf.getClass("hbase.regionserver.hlog.reader.impl",
600         Reader.class);
601     InstrumentedLogWriter.activateFailure = false;
602 
603     try {
604       conf.setClass("hbase.regionserver.hlog.reader.impl",
605           FaultySequenceFileLogReader.class, Reader.class);
606       conf.set("faultysequencefilelogreader.failuretype", failureType.name());
607       // Clean up from previous tests or previous loop
608       try {
609         wals.shutdown();
610       } catch (IOException exception) {
611         // since we're splitting out from under the factory, we should expect some closing failures.
612         LOG.debug("Ignoring problem closing WALFactory.", exception);
613       }
614       wals.close();
615       try {
616         for (FileStatus log : fs.listStatus(CORRUPTDIR)) {
617           fs.delete(log.getPath(), true);
618         }
619       } catch (FileNotFoundException exception) {
620         LOG.debug("no previous CORRUPTDIR to clean.");
621       }
622       // change to the faulty reader
623       wals = new WALFactory(conf, null, name.getMethodName());
624       generateWALs(-1);
625       // Our reader will render all of these files corrupt.
626       final Set<String> walDirContents = new HashSet<String>();
627       for (FileStatus status : fs.listStatus(WALDIR)) {
628         walDirContents.add(status.getPath().getName());
629       }
630       useDifferentDFSClient();
631       WALSplitter.split(HBASEDIR, WALDIR, OLDLOGDIR, fs, conf, wals);
632       return walDirContents;
633     } finally {
634       conf.setClass("hbase.regionserver.hlog.reader.impl", backupClass,
635           Reader.class);
636     }
637   }

967    * @param bufferSize size of in-memory buffer
968    * @param writerSlowness writer threads will sleep this many ms per edit

970   private void doTestThreading(final int numFakeEdits,
971       final int bufferSize,
972       final int writerSlowness) throws Exception {
973 
974     Configuration localConf = new Configuration(conf);
975     localConf.setInt("hbase.regionserver.hlog.splitlog.buffersize", bufferSize);
976 
977     // Create a fake log file (we'll override the reader to produce a stream of edits)

1260   private void corruptWAL(Path path, Corruptions corruption, boolean close) throws IOException {
1261     FSDataOutputStream out;
1262     int fileSize = (int) fs.listStatus(path)[0].getLen();
1263 
1264     FSDataInputStream in = fs.open(path);
1265     byte[] corrupted_bytes = new byte[fileSize];
1266     in.readFully(0, corrupted_bytes, 0, fileSize);
1267     in.close();
1268

1270     case APPEND_GARBAGE:
1271       fs.delete(path, false);
1272       out = fs.create(path);
1273       out.write(corrupted_bytes);
1274       out.write("-----".getBytes());
1275       closeOrFlush(close, out);
1276       break;
1277 
1278     case INSERT_GARBAGE_ON_FIRST_LINE:
1279       fs.delete(path, false);
1280       out = fs.create(path);
1281       out.write(0);
1282       out.write(corrupted_bytes);
1283       closeOrFlush(close, out);
1284       break;
1285 
1286     case INSERT_GARBAGE_IN_THE_MIDDLE:
1287       fs.delete(path, false);
1288       out = fs.create(path);
1289       int middle = (int) Math.floor(corrupted_bytes.length / 2);
1290       out.write(corrupted_bytes, 0, middle);
1291       out.write(0);
1292       out.write(corrupted_bytes, middle, corrupted_bytes.length - middle);
1293       closeOrFlush(close, out);
1294       break;
1295 
1296     case TRUNCATE:
1297       fs.delete(path, false);
1298       out = fs.create(path);
1299       out.write(corrupted_bytes, 0, fileSize
1300           - (32 + ProtobufLogReader.PB_WAL_COMPLETE_MAGIC.length + Bytes.SIZEOF_INT));
1301       closeOrFlush(close, out);
1302       break;
1303 
1304     case TRUNCATE_TRAILER:
1305       fs.delete(path, false);
1306       out = fs.create(path);
1307       out.write(corrupted_bytes, 0, fileSize - Bytes.SIZEOF_INT);// trailer is truncated.

1313   private void closeOrFlush(boolean close, FSDataOutputStream out)
1314       throws IOException {
1315     if (close) {
1316       out.close();
1317     } else {
1318       Method syncMethod = null;
1319       try {
1320         syncMethod = out.getClass().getMethod("hflush", new Class<?> []{});
1321       } catch (NoSuchMethodException e) {
1322         try {
1323           syncMethod = out.getClass().getMethod("sync", new Class<?> []{});
1324         } catch (NoSuchMethodException ex) {
1325           throw new IOException("This version of Hadoop supports " +
1326               "neither Syncable.sync() nor Syncable.hflush().");
1327         }
1328       }
1329       try {
1330         syncMethod.invoke(out, new Object[]{});
1331       } catch (Exception e) {
1332         throw new IOException(e);
1333       }
1334       // Not in 0out.hflush();
1335     }
