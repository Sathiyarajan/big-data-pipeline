154   /**
155    * Test that the contract from the master for checking on a snapshot are valid.
156    * <p>
157    * <ol>
158    * <li>If a snapshot fails with an error, we expect to get the source error.</li>
159    * <li>If there is no snapshot name supplied, we should get an error.</li>
160    * <li>If asking about a snapshot has hasn't occurred, you should get an error.</li>
161    * </ol>
162    */
163   @Test(timeout = 300000)
164   public void testIsDoneContract() throws Exception {
165 
166     IsSnapshotDoneRequest.Builder builder = IsSnapshotDoneRequest.newBuilder();
167 
168     String snapshotName = "asyncExpectedFailureTest";
169 
170     // check that we get an exception when looking up snapshot where one hasn't happened
171     SnapshotTestingUtils.expectSnapshotDoneException(master, builder.build(),
172       UnknownSnapshotException.class);
173 
174     // and that we get the same issue, even if we specify a name
175     SnapshotDescription desc = SnapshotDescription.newBuilder()
176       .setName(snapshotName).setTable(TABLE_NAME.getNameAsString()).build();
177     builder.setSnapshot(desc);
178     SnapshotTestingUtils.expectSnapshotDoneException(master, builder.build(),
179       UnknownSnapshotException.class);
180 
181     // set a mock handler to simulate a snapshot
182     DisabledTableSnapshotHandler mockHandler = Mockito.mock(DisabledTableSnapshotHandler.class);
183     Mockito.when(mockHandler.getException()).thenReturn(null);
184     Mockito.when(mockHandler.getSnapshot()).thenReturn(desc);
185     Mockito.when(mockHandler.isFinished()).thenReturn(new Boolean(true));
186     Mockito.when(mockHandler.getCompletionTimestamp())
187       .thenReturn(EnvironmentEdgeManager.currentTime());
188 
189     master.getSnapshotManager()
190         .setSnapshotHandlerForTesting(TABLE_NAME, mockHandler);
191 
192     // if we do a lookup without a snapshot name, we should fail - you should always know your name
193     builder = IsSnapshotDoneRequest.newBuilder();
194     SnapshotTestingUtils.expectSnapshotDoneException(master, builder.build(),
195       UnknownSnapshotException.class);
196 
197     // then do the lookup for the snapshot that it is done
198     builder.setSnapshot(desc);
199     IsSnapshotDoneResponse response =
200       master.getMasterRpcServices().isSnapshotDone(null, builder.build());
201     assertTrue("Snapshot didn't complete when it should have.", response.getDone());
202 
203     // now try the case where we are looking for a snapshot we didn't take
204     builder.setSnapshot(SnapshotDescription.newBuilder().setName("Not A Snapshot").build());
205     SnapshotTestingUtils.expectSnapshotDoneException(master, builder.build(),
206       UnknownSnapshotException.class);
207 
208     // then create a snapshot to the fs and make sure that we can find it when checking done
209     snapshotName = "completed";
210     Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
211     desc = desc.toBuilder().setName(snapshotName).build();
212     SnapshotDescriptionUtils.writeSnapshotInfo(desc, snapshotDir, fs);
213 
214     builder.setSnapshot(desc);
215     response = master.getMasterRpcServices().isSnapshotDone(null, builder.build());
216     assertTrue("Completed, on-disk snapshot not found", response.getDone());
217   }

219   @Test(timeout = 300000)
220   public void testGetCompletedSnapshots() throws Exception {
221     // first check when there are no snapshots
222     GetCompletedSnapshotsRequest request = GetCompletedSnapshotsRequest.newBuilder().build();
223     GetCompletedSnapshotsResponse response =
224       master.getMasterRpcServices().getCompletedSnapshots(null, request);
225     assertEquals("Found unexpected number of snapshots", 0, response.getSnapshotsCount());
226 
227     // write one snapshot to the fs
228     String snapshotName = "completed";
229     Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
230     SnapshotDescription snapshot = SnapshotDescription.newBuilder().setName(snapshotName).build();
231     SnapshotDescriptionUtils.writeSnapshotInfo(snapshot, snapshotDir, fs);
232 
233     // check that we get one snapshot
234     response = master.getMasterRpcServices().getCompletedSnapshots(null, request);
235     assertEquals("Found unexpected number of snapshots", 1, response.getSnapshotsCount());
236     List<SnapshotDescription> snapshots = response.getSnapshotsList();
237     List<SnapshotDescription> expected = Lists.newArrayList(snapshot);
238     assertEquals("Returned snapshots don't match created snapshots", expected, snapshots);
239 
240     // write a second snapshot
241     snapshotName = "completed_two";
242     snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
243     snapshot = SnapshotDescription.newBuilder().setName(snapshotName).build();
244     SnapshotDescriptionUtils.writeSnapshotInfo(snapshot, snapshotDir, fs);
245     expected.add(snapshot);
246 
247     // check that we get one snapshot
248     response = master.getMasterRpcServices().getCompletedSnapshots(null, request);
249     assertEquals("Found unexpected number of snapshots", 2, response.getSnapshotsCount());
250     snapshots = response.getSnapshotsList();
251     assertEquals("Returned snapshots don't match created snapshots", expected, snapshots);
252   }

277   /**
278    * Test that the snapshot hfile archive cleaner works correctly. HFiles that are in snapshots
279    * should be retained, while those that are not in a snapshot should be deleted.
280    * @throws Exception on failure
281    */
282   @Test(timeout = 300000)
283   public void testSnapshotHFileArchiving() throws Exception {
284     Admin admin = UTIL.getHBaseAdmin();
285     // make sure we don't fail on listing snapshots
286     SnapshotTestingUtils.assertNoSnapshots(admin);
287 
288     // recreate test table with disabled compactions; otherwise compaction may happen before
289     // snapshot, the call after snapshot will be a no-op and checks will fail
290     UTIL.deleteTable(TABLE_NAME);
291     HTableDescriptor htd = new HTableDescriptor(TABLE_NAME);
292     htd.setCompactionEnabled(false);
293 
294     UTIL.createTable(htd, new byte[][] { TEST_FAM }, null);
295 
296     // load the table (creates at least 4 hfiles)
297     for ( int i = 0; i < 5; i++) {
298       UTIL.loadTable(UTIL.getConnection().getTable(TABLE_NAME), TEST_FAM);
299       UTIL.flush(TABLE_NAME);
300     }
301 
302     // disable the table so we can take a snapshot
303     admin.disableTable(TABLE_NAME);
304     htd.setCompactionEnabled(true);
305 
306     // take a snapshot of the table
307     String snapshotName = "snapshot";
308     byte[] snapshotNameBytes = Bytes.toBytes(snapshotName);
309     admin.snapshot(snapshotNameBytes, TABLE_NAME);
310 
311     LOG.info("After snapshot File-System state");
312     FSUtils.logFileSystemState(fs, rootDir, LOG);
313 
314     // ensure we only have one snapshot
315     SnapshotTestingUtils.assertOneSnapshotThatMatches(admin, snapshotNameBytes, TABLE_NAME);
316 
317     // enable compactions now
318     admin.modifyTable(TABLE_NAME, htd);
319 
320     // renable the table so we can compact the regions
321     admin.enableTable(TABLE_NAME);
322 
323     // compact the files so we get some archived files for the table we just snapshotted
324     List<HRegion> regions = UTIL.getHBaseCluster().getRegions(TABLE_NAME);
325     for (HRegion region : regions) {
326       region.waitForFlushesAndCompactions(); // enable can trigger a compaction, wait for it.
327       region.compactStores(); // min is 2 so will compact and archive
328     }
329     LOG.info("After compaction File-System state");
330     FSUtils.logFileSystemState(fs, rootDir, LOG);
331 
332     // make sure the cleaner has run
333     LOG.debug("Running hfile cleaners");
334     ensureHFileCleanersRun();
335     LOG.info("After cleaners File-System state: " + rootDir);
336     FSUtils.logFileSystemState(fs, rootDir, LOG);
337 
338     // get the snapshot files for the table
339     Path snapshotTable = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
340     Set<String> snapshotHFiles = SnapshotReferenceUtil.getHFileNames(
341         UTIL.getConfiguration(), fs, snapshotTable);
342     // check that the files in the archive contain the ones that we need for the snapshot
343     LOG.debug("Have snapshot hfiles:");
344     for (String fileName : snapshotHFiles) {
345       LOG.debug(fileName);
346     }
347     // get the archived files for the table
348     Collection<String> files = getArchivedHFiles(archiveDir, rootDir, fs, TABLE_NAME);
349 
350     // and make sure that there is a proper subset
351     for (String fileName : snapshotHFiles) {
352       assertTrue("Archived hfiles " + files + " is missing snapshot file:" + fileName,
353         files.contains(fileName));
354     }
355 
356     // delete the existing snapshot
357     admin.deleteSnapshot(snapshotNameBytes);
358     SnapshotTestingUtils.assertNoSnapshots(admin);
359 
360     // make sure that we don't keep around the hfiles that aren't in a snapshot
361     // make sure we wait long enough to refresh the snapshot hfile
362     List<BaseHFileCleanerDelegate> delegates = UTIL.getMiniHBaseCluster().getMaster()
363         .getHFileCleaner().cleanersChain;
364     for (BaseHFileCleanerDelegate delegate: delegates) {
365       if (delegate instanceof SnapshotHFileCleaner) {
366         ((SnapshotHFileCleaner)delegate).getFileCacheForTesting().triggerCacheRefreshForTesting();
367       }
368     }
369     // run the cleaner again
370     LOG.debug("Running hfile cleaners");
371     ensureHFileCleanersRun();
372     LOG.info("After delete snapshot cleaners run File-System state");
373     FSUtils.logFileSystemState(fs, rootDir, LOG);
374 
375     files = getArchivedHFiles(archiveDir, rootDir, fs, TABLE_NAME);
376     assertEquals("Still have some hfiles in the archive, when their snapshot has been deleted.", 0,
377       files.size());
378   }

397   @Test
398   public void testAsyncSnapshotWillNotBlockSnapshotHFileCleaner() throws Exception {
399     // Write some data
400     Table table = UTIL.getConnection().getTable(TABLE_NAME);
401     for (int i = 0; i < 10; i++) {
402       Put put = new Put(Bytes.toBytes(i)).addColumn(TEST_FAM, Bytes.toBytes("q"), Bytes.toBytes(i));
403       table.put(put);
404     }
405     final String snapshotName = "testAsyncSnapshotWillNotBlockSnapshotHFileCleaner01";
406     SnapshotDescription snapshot = SnapshotDescription.newBuilder().setName(snapshotName)
407         .setTable(TABLE_NAME.getNameAsString()).setType(SnapshotDescription.Type.FLUSH).build();
408     UTIL.getHBaseAdmin().takeSnapshotAsync(snapshot);
409     UTIL.waitFor(10 * 1000L, 200L, new Predicate<Exception>() {
410       @Override
411       public boolean evaluate() throws Exception {
412         return UTIL.getHBaseAdmin().listSnapshots(Pattern.compile(snapshotName)).size() == 1;
413       }
414     });
415     assertTrue(master.getSnapshotManager().isTakingAnySnapshot());
416     Thread.sleep(11 * 1000L);
417     assertFalse(master.getSnapshotManager().isTakingAnySnapshot());
418   }
