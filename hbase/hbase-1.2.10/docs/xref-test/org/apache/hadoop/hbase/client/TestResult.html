48    static KeyValue[] genKVs(final byte[] row, final byte[] family,
49                             final byte[] value,
50                      final long timestamp,
51                      final int cols) {
52      KeyValue [] kvs = new KeyValue[cols];
53  
54      for (int i = 0; i < cols ; i++) {
55        kvs[i] = new KeyValue(
56            row, family, Bytes.toBytes(i),
57            timestamp,
58            Bytes.add(value, Bytes.toBytes(i)));
59      }
60      return kvs;
61    }

67    /**
68     * Run some tests to ensure Result acts like a proper CellScanner.
69     * @throws IOException
70     */
71    public void testResultAsCellScanner() throws IOException {
72      Cell [] cells = genKVs(row, family, value, 1, 10);
73      Arrays.sort(cells, KeyValue.COMPARATOR);
74      Result r = Result.create(cells);
75      assertSame(r, cells);
76      // Assert I run over same result multiple times.
77      assertSame(r.cellScanner(), cells);
78      assertSame(r.cellScanner(), cells);
79      // Assert we are not creating new object when doing cellscanner
80      assertTrue(r == r.cellScanner());
81    }
82  
83    private void assertSame(final CellScanner cellScanner, final Cell [] cells) throws IOException {
84      int count = 0;
85      while (cellScanner.advance()) {
86        assertTrue(cells[count].equals(cellScanner.current()));
87        count++;
88      }
89      assertEquals(cells.length, count);
90    }

277   /**
278    * Microbenchmark that compares {@link Result#getValue} and {@link Result#loadValue} performance.
279    *
280    * @throws Exception
281    */
282   public void doReadBenchmark() throws Exception {
283 
284     final int n = 5;
285     final int m = 100000000;
286 
287     StringBuilder valueSB = new StringBuilder();
288     for (int i = 0; i < 100; i++) {
289       valueSB.append((byte)(Math.random() * 10));
290     }
291 
292     StringBuilder rowSB = new StringBuilder();
293     for (int i = 0; i < 50; i++) {
294       rowSB.append((byte)(Math.random() * 10));
295     }
296 
297     KeyValue [] kvs = genKVs(Bytes.toBytes(rowSB.toString()), family,
298         Bytes.toBytes(valueSB.toString()), 1, n);
299     Arrays.sort(kvs, KeyValue.COMPARATOR);
300     ByteBuffer loadValueBuffer = ByteBuffer.allocate(1024);
301     Result r = Result.create(kvs);
302 
303     byte[][] qfs = new byte[n][Bytes.SIZEOF_INT];
304     for (int i = 0; i < n; ++i) {
305       System.arraycopy(qfs[i], 0, Bytes.toBytes(i), 0, Bytes.SIZEOF_INT);
306     }
307 
308     // warm up
309     for (int k = 0; k < 100000; k++) {
310       for (int i = 0; i < n; ++i) {
311         r.getValue(family, qfs[i]);
312         loadValueBuffer.clear();
313         r.loadValue(family, qfs[i], loadValueBuffer);
314         loadValueBuffer.flip();
315       }
316     }
317 
318     System.gc();
319     long start = System.nanoTime();
320     for (int k = 0; k < m; k++) {
321       for (int i = 0; i < n; ++i) {
322         loadValueBuffer.clear();
323         r.loadValue(family, qfs[i], loadValueBuffer);
324         loadValueBuffer.flip();
325       }
326     }
327     long stop = System.nanoTime();
328     System.out.println("loadValue(): " + (stop - start));
329 
330     System.gc();
331     start = System.nanoTime();
332     for (int k = 0; k < m; k++) {
333       for (int i = 0; i < n; i++) {
334         r.getValue(family, qfs[i]);
335       }
336     }
337     stop = System.nanoTime();
338     System.out.println("getValue():  " + (stop - start));
339   }
