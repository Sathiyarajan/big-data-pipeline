441   /**
442    * Test we do not lose data if we fail a flush and then close.
443    * Part of HBase-10466.  Tests the following from the issue description:
444    * "Bug 1: Wrong calculation of HRegion.memstoreSize: When a flush fails, data to be flushed is
445    * kept in each MemStore's snapshot and wait for next flush attempt to continue on it. But when
446    * the next flush succeeds, the counter of total memstore size in HRegion is always deduced by
447    * the sum of current memstore sizes instead of snapshots left from previous failed flush. This
448    * calculation is problematic that almost every time there is failed flush, HRegion.memstoreSize
449    * gets reduced by a wrong value. If region flush could not proceed for a couple cycles, the size
450    * in current memstore could be much larger than the snapshot. It's likely to drift memstoreSize
451    * much smaller than expected. In extreme case, if the error accumulates to even bigger than
452    * HRegion's memstore size limit, any further flush is skipped because flush does not do anything
453    * if memstoreSize is not larger than 0."
454    * @throws Exception
455    */
456   @Test (timeout=60000)
457   public void testFlushSizeAccounting() throws Exception {
458     final Configuration conf = HBaseConfiguration.create(CONF);
459     // Only retry once.
460     conf.setInt("hbase.hstore.flush.retries.number", 1);
461     final User user =
462       User.createUserForTesting(conf, this.name.getMethodName(), new String[]{"foo"});
463     // Inject our faulty LocalFileSystem
464     conf.setClass("fs.file.impl", FaultyFileSystem.class, FileSystem.class);
465     user.runAs(new PrivilegedExceptionAction<Object>() {
466       @Override
467       public Object run() throws Exception {
468         // Make sure it worked (above is sensitive to caching details in hadoop core)
469         FileSystem fs = FileSystem.get(conf);
470         Assert.assertEquals(FaultyFileSystem.class, fs.getClass());
471         FaultyFileSystem ffs = (FaultyFileSystem)fs;
472         HRegion region = null;
473         try {
474           // Initialize region
475           region = initHRegion(tableName, name.getMethodName(), conf, COLUMN_FAMILY_BYTES);
476           long size = region.getMemstoreSize();
477           Assert.assertEquals(0, size);
478           // Put one item into memstore.  Measure the size of one item in memstore.
479           Put p1 = new Put(row);
480           p1.add(new KeyValue(row, COLUMN_FAMILY_BYTES, qual1, 1, (byte[])null));
481           region.put(p1);
482           final long sizeOfOnePut = region.getMemstoreSize();
483           // Fail a flush which means the current memstore will hang out as memstore 'snapshot'.
484           try {
485             LOG.info("Flushing");
486             region.flush(true);
487             Assert.fail("Didn't bubble up IOE!");
488           } catch (DroppedSnapshotException dse) {
489             // What we are expecting
490             region.closing.set(false); // this is needed for the rest of the test to work
491           } catch (Exception e) {
492             // What we are expecting
493             region.closing.set(false); // this is needed for the rest of the test to work
494           }
495           // Make it so all writes succeed from here on out
496           ffs.fault.set(false);
497           // WAL is bad because of above faulty fs. Roll WAL.
498           try {
499             region.getWAL().rollWriter(true);
500           } catch (Exception e) {
501             int x = 0;
502           }
503           // Check sizes.  Should still be the one entry.
504           Assert.assertEquals(sizeOfOnePut, region.getMemstoreSize());
505           // Now add two entries so that on this next flush that fails, we can see if we
506           // subtract the right amount, the snapshot size only.
507           Put p2 = new Put(row);
508           p2.add(new KeyValue(row, COLUMN_FAMILY_BYTES, qual2, 2, (byte[])null));
509           p2.add(new KeyValue(row, COLUMN_FAMILY_BYTES, qual3, 3, (byte[])null));
510           region.put(p2);

2651    * @param midkey
2652    *          Key to split around.
2653    * @return The Regions we created.
2654    * @throws IOException
2655    */
2656   HRegion[] splitRegion(final HRegion parent, final byte[] midkey) throws IOException {
2657     PairOfSameType<Region> result = null;
2658     SplitTransactionImpl st = new SplitTransactionImpl(parent, midkey);
2659     // If prepare does not return true, for some reason -- logged inside in
2660     // the prepare call -- we are not ready to split just now. Just return.
2661     if (!st.prepare()) {
2662       parent.clearSplit();
2663       return null;
2664     }
2665     try {
2666       result = st.execute(null, null);
2667     } catch (IOException ioe) {
2668       try {
2669         LOG.info("Running rollback of failed split of " +
2670           parent.getRegionInfo().getRegionNameAsString() + "; " + ioe.getMessage());
2671         st.rollback(null, null);
2672         LOG.info("Successful rollback of failed split of " +
2673           parent.getRegionInfo().getRegionNameAsString());
2674         return null;
2675       } catch (RuntimeException e) {
2676         // If failed rollback, kill this server to avoid having a hole in table.
2677         LOG.info("Failed rollback of failed split of " +
2678           parent.getRegionInfo().getRegionNameAsString() + " -- aborting server", e);
2679       }
2680     }
2681     finally {
2682       parent.clearSplit();
2683     }
2684     return new HRegion[] { (HRegion)result.getFirst(), (HRegion)result.getSecond() };
