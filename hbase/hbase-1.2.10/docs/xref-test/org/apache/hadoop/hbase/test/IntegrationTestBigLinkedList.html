124 /**
125  * This is an integration test borrowed from goraci, written by Keith Turner,
126  * which is in turn inspired by the Accumulo test called continous ingest (ci).
127  * The original source code can be found here:
128  * https://github.com/keith-turner/goraci
129  * https://github.com/enis/goraci/
130  *
131  * Apache Accumulo [0] has a simple test suite that verifies that data is not
132  * lost at scale. This test suite is called continuous ingest. This test runs
133  * many ingest clients that continually create linked lists containing 25
134  * million nodes. At some point the clients are stopped and a map reduce job is
135  * run to ensure no linked list has a hole. A hole indicates data was lost.··
136  *
137  * The nodes in the linked list are random. This causes each linked list to
138  * spread across the table. Therefore if one part of a table loses data, then it
139  * will be detected by references in another part of the table.
140  *
141  * THE ANATOMY OF THE TEST
142  *
143  * Below is rough sketch of how data is written. For specific details look at
144  * the Generator code.
145  *
146  * 1 Write out 1 million nodes· 2 Flush the client· 3 Write out 1 million that
147  * reference previous million· 4 If this is the 25th set of 1 million nodes,
148  * then update 1st set of million to point to last· 5 goto 1
149  *
150  * The key is that nodes only reference flushed nodes. Therefore a node should
151  * never reference a missing node, even if the ingest client is killed at any
152  * point in time.
153  *
154  * When running this test suite w/ Accumulo there is a script running in
155  * parallel called the Aggitator that randomly and continuously kills server
156  * processes.·· The outcome was that many data loss bugs were found in Accumulo
157  * by doing this.· This test suite can also help find bugs that impact uptime
158  * and stability when· run for days or weeks.··
159  *
160  * This test suite consists the following· - a few Java programs· - a little
161  * helper script to run the java programs - a maven script to build it.··
162  *
163  * When generating data, its best to have each map task generate a multiple of
164  * 25 million. The reason for this is that circular linked list are generated
165  * every 25M. Not generating a multiple in 25M will result in some nodes in the
166  * linked list not having references. The loss of an unreferenced node can not
167  * be detected.
168  *
169  *
170  * Below is a description of the Java programs
171  *
172  * Generator - A map only job that generates data. As stated previously,·
173  * its best to generate data in multiples of 25M.
174  *
175  * Verify - A map reduce job that looks for holes. Look at the counts after running. REFERENCED and
176  * UNREFERENCED are· ok, any UNDEFINED counts are bad. Do not run at the· same
177  * time as the Generator.
178  *
179  * Walker - A standalone program that start following a linked list· and emits timing info.··
180  *
181  * Print - A standalone program that prints nodes in the linked list
182  *
183  * Delete - A standalone program that deletes a single node
184  *
185  * This class can be run as a unit test, as an integration test, or from the command line
186  */

374      */
375     static class GeneratorMapper
376       extends Mapper<BytesWritable, NullWritable, NullWritable, NullWritable> {
377 
378       byte[][] first = null;
379       byte[][] prev = null;
380       byte[][] current = null;
381       byte[] id;
382       long count = 0;
383       int i;
384       BufferedMutator mutator;
385       Connection connection;
386       long numNodes;
387       long wrap;
388       int width;
389       boolean multipleUnevenColumnFamilies;
390       byte[] tinyValue = new byte[] { 't' };
391       byte[] bigValue = null;
392 
393       @Override
394       protected void setup(Context context) throws IOException, InterruptedException {
395         id = Bytes.toBytes("Job: "+context.getJobID() + " Task: " + context.getTaskAttemptID());
396         Configuration conf = context.getConfiguration();
397         connection = ConnectionFactory.createConnection(conf);
398         instantiateHTable();
399         this.width = context.getConfiguration().getInt(GENERATOR_WIDTH_KEY, WIDTH_DEFAULT);
400         current = new byte[this.width][];
401         int wrapMultiplier = context.getConfiguration().getInt(GENERATOR_WRAP_KEY, WRAP_DEFAULT);
402         this.wrap = (long)wrapMultiplier * width;
403         this.numNodes = context.getConfiguration().getLong(
404             GENERATOR_NUM_ROWS_PER_MAP_KEY, (long)WIDTH_DEFAULT * WRAP_DEFAULT);
405         if (this.numNodes < this.wrap) {
406           this.wrap = this.numNodes;
407         }
408         this.multipleUnevenColumnFamilies = isMultiUnevenColumnFamilies(context.getConfiguration());
409       }
410 
411       protected void instantiateHTable() throws IOException {
412         mutator = connection.getBufferedMutator(
413             new BufferedMutatorParams(getTableName(connection.getConfiguration()))
414                 .writeBufferSize(4 * 1024 * 1024));
415       }
416 
417       @Override
418       protected void cleanup(Context context) throws IOException ,InterruptedException {
419         mutator.close();
420         connection.close();
421       }
422 
423       @Override
424       protected void map(BytesWritable key, NullWritable value, Context output) throws IOException {
425         current[i] = new byte[key.getLength()];
426         System.arraycopy(key.getBytes(), 0, current[i], 0, key.getLength());
427         if (++i == current.length) {
428           LOG.info("Persisting current.length=" + current.length + ", count=" + count + ", id=" +
429             Bytes.toStringBinary(id) + ", current=" + Bytes.toStringBinary(current[0]) +
430             ", i=" + i);
431           persist(output, count, prev, current, id);
432           i = 0;
433 
434           if (first == null) {
435             first = current;
436           }
437           prev = current;
438           current = new byte[this.width][];
439 
440           count += current.length;
441           output.setStatus("Count " + count);
442 
443           if (count % wrap == 0) {
444             // this block of code turns the 1 million linked list of length 25 into one giant
445             //circular linked list of 25 million
446             circularLeftShift(first);
447 
448             persist(output, -1, prev, first, null);
449 
450             first = null;
451             prev = null;
452           }
453         }
454       }
455 
456       private static <T> void circularLeftShift(T[] first) {
457         T ez = first[0];
458         System.arraycopy(first, 1, first, 0, first.length - 1);
459         first[first.length - 1] = ez;
460       }
461 
462       protected void persist(Context output, long count, byte[][] prev, byte[][] current, byte[] id)
463           throws IOException {
464         for (int i = 0; i < current.length; i++) {
465 
466           if (i % 100 == 0) {
467             // Tickle progress every so often else maprunner will think us hung
468             output.progress();
469           }
470 
471           Put put = new Put(current[i]);
472           put.addColumn(FAMILY_NAME, COLUMN_PREV, prev == null ? NO_KEY : prev[i]);
473 
474           if (count >= 0) {
475             put.addColumn(FAMILY_NAME, COLUMN_COUNT, Bytes.toBytes(count + i));
476           }
477           if (id != null) {
478             put.addColumn(FAMILY_NAME, COLUMN_CLIENT, id);
479           }
480           // See if we are to write multiple columns.
481           if (this.multipleUnevenColumnFamilies) {
482             // Use any column name.
483             put.addColumn(TINY_FAMILY_NAME, TINY_FAMILY_NAME, this.tinyValue);
484             // If we've not allocated bigValue, do it now. Reuse same value each time.
485             if (this.bigValue == null) {
486               this.bigValue = new byte[current[i].length * 10];
487               ThreadLocalRandom.current().nextBytes(this.bigValue);
488             }
489             // Use any column name.
490             put.addColumn(BIG_FAMILY_NAME, BIG_FAMILY_NAME, this.bigValue);
491           }
492           mutator.mutate(put);
493         }
494 
495         mutator.flush();
496       }

517     protected void createSchema() throws IOException {
518       Configuration conf = getConf();
519       TableName tableName = getTableName(conf);
520       try (Connection conn = ConnectionFactory.createConnection(conf);
521           Admin admin = conn.getAdmin()) {
522         if (!admin.tableExists(tableName)) {
523           HTableDescriptor htd = new HTableDescriptor(getTableName(getConf()));
524           htd.addFamily(new HColumnDescriptor(FAMILY_NAME));
525           // Always add these families. Just skip writing to them when we do not test per CF flush.
526           htd.addFamily(new HColumnDescriptor(BIG_FAMILY_NAME));
527           htd.addFamily(new HColumnDescriptor(TINY_FAMILY_NAME));
528 
529           // If we want to pre-split compute how many splits.
530           if (conf.getBoolean(HBaseTestingUtility.PRESPLIT_TEST_TABLE_KEY,
531               HBaseTestingUtility.PRESPLIT_TEST_TABLE)) {
532             int numberOfServers = admin.getClusterStatus().getServers().size();
533             if (numberOfServers == 0) {
534               throw new IllegalStateException("No live regionservers");
535             }
536             int regionsPerServer = conf.getInt(HBaseTestingUtility.REGIONS_PER_SERVER_KEY,
537                 HBaseTestingUtility.DEFAULT_REGIONS_PER_SERVER);
538             int totalNumberOfRegions = numberOfServers * regionsPerServer;
539             LOG.info("Number of live regionservers: " + numberOfServers + ", " +
540                 "pre-splitting table into " + totalNumberOfRegions + " regions " +
541                 "(default regions per server: " + regionsPerServer + ")");
542 
543 
544             byte[][] splits = new RegionSplitter.UniformSplit().split(totalNumberOfRegions);
545 
546             admin.createTable(htd, splits);
547           } else {
548             // Looks like we're just letting things play out.
549             // Create a table with on region by default.
550             // This will make the splitting work hard.
551             admin.createTable(htd);
552           }
553         }
554       } catch (MasterNotRunningException e) {
555         LOG.error("Master not running", e);
556         throw new IOException(e);
557       }
558     }

747     static SortedSet<byte []> readKeysToSearch(final Configuration conf)
748     throws IOException, InterruptedException {
749       Path keysInputDir = new Path(conf.get(SEARCHER_INPUTDIR_KEY));
750       FileSystem fs = FileSystem.get(conf);
751       SortedSet<byte []> result = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
752       if (!fs.exists(keysInputDir)) {
753         throw new FileNotFoundException(keysInputDir.toString());
754       }
755       if (!fs.isDirectory(keysInputDir)) {
756         throw new UnsupportedOperationException("TODO");
757       } else {
758         RemoteIterator<LocatedFileStatus> iterator = fs.listFiles(keysInputDir, false);
759         while(iterator.hasNext()) {
760           LocatedFileStatus keyFileStatus = iterator.next();
761           // Skip "_SUCCESS" file.
762           if (keyFileStatus.getPath().getName().startsWith("_")) continue;

764         }

769     private static SortedSet<byte[]> readFileToSearch(final Configuration conf,
770         final FileSystem fs, final LocatedFileStatus keyFileStatus) throws IOException,
771         InterruptedException {
772       SortedSet<byte []> result = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
773       // Return entries that are flagged Counts.UNDEFINED in the value. Return the row. This is
774       // what is missing.
775       TaskAttemptContext context = new TaskAttemptContextImpl(conf, new TaskAttemptID());
776       try (SequenceFileAsBinaryInputFormat.SequenceFileAsBinaryRecordReader rr =
777           new SequenceFileAsBinaryInputFormat.SequenceFileAsBinaryRecordReader()) {
778         InputSplit is =
779           new FileSplit(keyFileStatus.getPath(), 0, keyFileStatus.getLen(), new String [] {});
780         rr.initialize(is, context);
781         while (rr.nextKeyValue()) {
782           rr.getCurrentKey();
783           BytesWritable bw = rr.getCurrentValue();
784           if (Verify.VerifyReducer.whichType(bw.getBytes()) == Verify.Counts.UNDEFINED) {
785             byte[] key = new byte[rr.getCurrentKey().getLength()];
786             System.arraycopy(rr.getCurrentKey().getBytes(), 0, key, 0, rr.getCurrentKey()
787                 .getLength());
788             result.add(key);
789           }
790         }
791       }
792       return result;
793     }

886        * @param ordinal
887        * @param r
888        * @return Return new byte array that has <code>ordinal</code> as prefix on front taking up
889        * Bytes.SIZEOF_SHORT bytes followed by <code>r</code>

891       public static byte[] addPrefixFlag(final int ordinal, final byte [] r) {
892         byte[] prefix = Bytes.toBytes((short)ordinal);
893         if (prefix.length != Bytes.SIZEOF_SHORT) {
894           throw new RuntimeException("Unexpected size: " + prefix.length);
895         }
896         byte[] result = new byte[prefix.length + r.length];
897         System.arraycopy(prefix, 0, result, 0, prefix.length);
898         System.arraycopy(r, 0, result, prefix.length, r.length);
899         return result;

922       @Override
923       public void reduce(BytesWritable key, Iterable<BytesWritable> values, Context context)
924           throws IOException, InterruptedException {
925         int defCount = 0;
926         boolean lostFamilies = false;
927         refs.clear();
928         for (BytesWritable type : values) {
929           if (type.getLength() == DEF.getLength()) {
930             defCount++;
931             if (type.getBytes()[0] == 1) {
932               lostFamilies = true;
933             }
934           } else {
935             byte[] bytes = new byte[type.getLength()];
936             System.arraycopy(type.getBytes(), 0, bytes, 0, type.getLength());
937             refs.add(bytes);
938           }
939         }
940 
941         // TODO check for more than one def, should not happen
942         StringBuilder refsSb = null;
943         String keyString = Bytes.toStringBinary(key.getBytes(), 0, key.getLength());
944         if (defCount == 0 || refs.size() != 1) {
945           refsSb = dumpExtraInfoOnRefs(key, context, refs);
946           LOG.error("LinkedListError: key=" + keyString + ", reference(s)=" +
947             (refsSb != null? refsSb.toString(): ""));
948         }
949         if (lostFamilies) {
950           LOG.error("LinkedListError: key=" + keyString + ", lost big or tiny families");
951           context.getCounter(Counts.LOST_FAMILIES).increment(1);
952           context.write(key, LOSTFAM);
953         }
954 
955         if (defCount == 0 && refs.size() > 0) {
956           // This is bad, found a node that is referenced but not defined. It must have been
957           // lost, emit some info about this node for debugging purposes.
958           // Write out a line per reference. If more than one, flag it.;
959           for (int i = 0; i < refs.size(); i++) {
960             byte[] bs = refs.get(i);
961             int ordinal;
962             if (i <= 0) {
963               ordinal = Counts.UNDEFINED.ordinal();
964               context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
965               context.getCounter(Counts.UNDEFINED).increment(1);
966             } else {
967               ordinal = Counts.EXTRA_UNDEF_REFERENCES.ordinal();
968               context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
969             }
970           }
971           if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
972             // Print out missing row; doing get on reference gives info on when the referencer
973             // was added which can help a little debugging. This info is only available in mapper
974             // output -- the 'Linked List error Key...' log message above. What we emit here is
975             // useless for debugging.
976             context.getCounter("undef", keyString).increment(1);
977           }
978         } else if (defCount > 0 && refs.size() == 0) {
979           // node is defined but not referenced
980           context.write(key, UNREF);
981           context.getCounter(Counts.UNREFERENCED).increment(1);
982           if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
983             context.getCounter("unref", keyString).increment(1);
984           }
985         } else {
986           if (refs.size() > 1) {
987             // Skip first reference.
988             for (int i = 1; i < refs.size(); i++) {
989               context.write(key,
990                 new BytesWritable(addPrefixFlag(Counts.EXTRAREFERENCES.ordinal(), refs.get(i))));
991             }
992             context.getCounter(Counts.EXTRAREFERENCES).increment(refs.size() - 1);
993           }
994           // node is defined and referenced
995           context.getCounter(Counts.REFERENCED).increment(1);
996         }
997       }

1001        * @return StringBuilder filled with references if any.
1002        * @throws IOException
1003        */
1004       private StringBuilder dumpExtraInfoOnRefs(final BytesWritable key, final Context context,
1005           final List<byte []> refs)
1006       throws IOException {
1007         StringBuilder refsSb = null;
1008         if (refs.isEmpty()) return refsSb;
1009         refsSb = new StringBuilder();
1010         String comma = "";
1011         // If a row is a reference but has no define, print the content of the row that has
1012         // this row as a 'prev'; it will help debug.  The missing row was written just before
1013         // the row we are dumping out here.
1014         TableName tn = getTableName(context.getConfiguration());
1015         try (Table t = this.connection.getTable(tn)) {
1016           for (byte [] ref : refs) {
1017             Result r = t.get(new Get(ref));

1019             String ts = (cells != null && !cells.isEmpty())?
1020                 new java.util.Date(cells.get(0).getTimestamp()).toString(): "";
1021             byte [] b = r.getValue(FAMILY_NAME, COLUMN_CLIENT);
1022             String jobStr = (b != null && b.length > 0)? Bytes.toString(b): "";
1023             b = r.getValue(FAMILY_NAME, COLUMN_COUNT);
1024             long count = (b != null && b.length > 0)? Bytes.toLong(b): -1;
1025             b = r.getValue(FAMILY_NAME, COLUMN_PREV);
1026             String refRegionLocation = "";
1027             String keyRegionLocation = "";
1028             if (b != null && b.length > 0) {
1029               try (RegionLocator rl = this.connection.getRegionLocator(tn)) {
1030                 HRegionLocation hrl = rl.getRegionLocation(b);
1031                 if (hrl != null) refRegionLocation = hrl.toString();

1033                 hrl = rl.getRegionLocation(key.getBytes());
1034                 if (hrl != null) keyRegionLocation = hrl.toString();
1035               }
1036             }
1037             LOG.error("Extras on ref without a def, ref=" + Bytes.toStringBinary(ref) +
1038               ", refPrevEqualsKey=" +
1039                 (Bytes.compareTo(key.getBytes(), 0, key.getLength(), b, 0, b.length) == 0) +

1042                 ", ref row count=" + count +
1043                 ", ref row regionLocation=" + refRegionLocation +
1044                 ", key row regionLocation=" + keyRegionLocation);
1045             refsSb.append(comma);
1046             comma = ",";
1047             refsSb.append(Bytes.toStringBinary(ref));
1048           }
1049         }
1050         return refsSb;
1051       }
