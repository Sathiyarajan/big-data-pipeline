123   public static class SleepAndFailFirstTime extends BaseRegionObserver {
124     static final AtomicLong ct = new AtomicLong(0);
125     static final String SLEEP_TIME_CONF_KEY =
126         "hbase.coprocessor.SleepAndFailFirstTime.sleepTime";
127     static final long DEFAULT_SLEEP_TIME = 20000;
128     static final AtomicLong sleepTime = new AtomicLong(DEFAULT_SLEEP_TIME);
129 
130     public SleepAndFailFirstTime() {
131     }
132 
133     @Override
134     public void postOpen(ObserverContext<RegionCoprocessorEnvironment> c) {
135       RegionCoprocessorEnvironment env = c.getEnvironment();
136       Configuration conf = env.getConfiguration();
137       sleepTime.set(conf.getLong(SLEEP_TIME_CONF_KEY, DEFAULT_SLEEP_TIME));
138     }
139 
140     @Override
141     public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
142               final Get get, final List<Cell> results) throws IOException {
143       Threads.sleep(sleepTime.get());
144       if (ct.incrementAndGet() == 1){
145         throw new IOException("first call I fail");
146       }
147     }
148   }

177   @Test
178   public void testClusterConnection() throws IOException {
179     ThreadPoolExecutor otherPool = new ThreadPoolExecutor(1, 1,
180         5, TimeUnit.SECONDS,
181         new SynchronousQueue<Runnable>(),
182         Threads.newDaemonThreadFactory("test-hcm"));
183 
184     HConnection con1 = HConnectionManager.createConnection(TEST_UTIL.getConfiguration());
185     HConnection con2 = HConnectionManager.createConnection(TEST_UTIL.getConfiguration(), otherPool);
186     // make sure the internally created ExecutorService is the one passed
187     assertTrue(otherPool == ((HConnectionImplementation)con2).getCurrentBatchPool());
188 
189     String tableName = "testClusterConnection";
190     TEST_UTIL.createTable(tableName.getBytes(), FAM_NAM).close();
191     HTable t = (HTable)con1.getTable(tableName, otherPool);
192     // make sure passing a pool to the getTable does not trigger creation of an internal pool
193     assertNull("Internal Thread pool should be null", ((HConnectionImplementation)con1).getCurrentBatchPool());
194     // table should use the pool passed
195     assertTrue(otherPool == t.getPool());
196     t.close();
197 
198     t = (HTable)con2.getTable(tableName);
199     // table should use the connectin's internal pool
200     assertTrue(otherPool == t.getPool());
201     t.close();
202 
203     t = (HTable)con2.getTable(Bytes.toBytes(tableName));
204     // try other API too
205     assertTrue(otherPool == t.getPool());
206     t.close();
207 
208     t = (HTable)con2.getTable(TableName.valueOf(tableName));
209     // try other API too
210     assertTrue(otherPool == t.getPool());
211     t.close();
212 
213     t = (HTable)con1.getTable(tableName);
214     ExecutorService pool = ((HConnectionImplementation)con1).getCurrentBatchPool();
215     // make sure an internal pool was created
216     assertNotNull("An internal Thread pool should have been created", pool);
217     // and that the table is using it
218     assertTrue(t.getPool() == pool);
219     t.close();
220 
221     t = (HTable)con1.getTable(tableName);
222     // still using the *same* internal pool
223     assertTrue(t.getPool() == pool);
224     t.close();
225 
226     con1.close();
227     // if the pool was created on demand it should be closed upon connection close
228     assertTrue(pool.isShutdown());
229 
230     con2.close();
231     // if the pool is passed, it is not closed
232     assertFalse(otherPool.isShutdown());
233     otherPool.shutdownNow();
234   }

322   /**
323    * Test that an operation can fail if we read the global operation timeout, even if the
324    * individual timeout is fine. We do that with:
325    * - client side: an operation timeout of 30 seconds
326    * - server side: we sleep 20 second at each attempt. The first work fails, the second one
327    * succeeds. But the client won't wait that much, because 20 + 20 > 30, so the client
328    * timeouted when the server answers.
329    */
330   @Test
331   public void testOperationTimeout() throws Exception {
332     HTableDescriptor hdt = TEST_UTIL.createTableDescriptor("HCM-testOperationTimeout");
333     hdt.addCoprocessor(SleepAndFailFirstTime.class.getName());
334     HTable table = TEST_UTIL.createTable(hdt, new byte[][]{FAM_NAM}, TEST_UTIL.getConfiguration());
335     table.setRpcTimeout(Integer.MAX_VALUE);
336     // Check that it works if the timeout is big enough
337     table.setOperationTimeout(120 * 1000);
338     table.get(new Get(FAM_NAM));
339 
340     // Resetting and retrying. Will fail this time, not enough time for the second try
341     SleepAndFailFirstTime.ct.set(0);
342     try {
343       table.setOperationTimeout(30 * 1000);
344       table.get(new Get(FAM_NAM));
345       Assert.fail("We expect an exception here");
346     } catch (SocketTimeoutException e) {
347       // The client has a CallTimeout class, but it's not shared.We're not very clean today,
348       //  in the general case you can expect the call to stop, but the exception may vary.
349       // In this test however, we're sure that it will be a socket timeout.
350       LOG.info("We received an exception, as expected ", e);
351     } catch (IOException e) {
352       Assert.fail("Wrong exception:" + e.getMessage());
353     } finally {
354       table.close();
355     }
356   }

463   private void testConnectionClose(boolean allowsInterrupt) throws Exception {
464     TableName tableName = TableName.valueOf("HCM-testConnectionClose" + allowsInterrupt);
465     TEST_UTIL.createTable(tableName, FAM_NAM).close();
466 
467     boolean previousBalance = TEST_UTIL.getHBaseAdmin().setBalancerRunning(false, true);
468 
469     Configuration c2 = new Configuration(TEST_UTIL.getConfiguration());
470     // We want to work on a separate connection.
471     c2.set(HConstants.HBASE_CLIENT_INSTANCE_ID, String.valueOf(-1));
472     c2.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 100); // retry a lot
473     c2.setInt(HConstants.HBASE_CLIENT_PAUSE, 1); // don't wait between retries.
474     c2.setInt(RpcClient.FAILED_SERVER_EXPIRY_KEY, 0); // Server do not really expire
475     c2.setBoolean(RpcClient.SPECIFIC_WRITE_THREAD, allowsInterrupt);
476 
477     final HTable table = new HTable(c2, tableName);
478 
479     Put put = new Put(ROW);
480     put.add(FAM_NAM, ROW, ROW);
481     table.put(put);
482 
483     // 4 steps: ready=0; doGets=1; mustStop=2; stopped=3
484     final AtomicInteger step = new AtomicInteger(0);
485 
486     final AtomicReference<Throwable> failed = new AtomicReference<Throwable>(null);
487     Thread t = new Thread("testConnectionCloseThread") {
488       @Override
489       public void run() {
490         int done = 0;
491         try {
492           step.set(1);
493           while (step.get() == 1) {
494             Get get = new Get(ROW);
495             table.get(get);
496             done++;
497             if (done % 100 == 0)
498               LOG.info("done=" + done);
499           }
500         } catch (Throwable t) {
501           failed.set(t);
502           LOG.error(t);
503         }
504         step.set(3);
505       }
506     };
507     t.start();
508     TEST_UTIL.waitFor(20000, new Waiter.Predicate<Exception>() {
509       @Override
510       public boolean evaluate() throws Exception {
511         return step.get() == 1;
512       }
513     });
514 
515     ServerName sn = table.getRegionLocation(ROW).getServerName();
516     ConnectionManager.HConnectionImplementation conn =
517         (ConnectionManager.HConnectionImplementation) table.getConnection();
518     RpcClient rpcClient = conn.getRpcClient();
519 
520     LOG.info("Going to cancel connections. connection=" + conn.toString() + ", sn=" + sn);
521     for (int i = 0; i < 5000; i++) {
522       rpcClient.cancelConnections(sn);
523       Thread.sleep(5);
524     }
525 
526     step.compareAndSet(1, 2);
527     // The test may fail here if the thread doing the gets is stuck. The way to find
528     //  out what's happening is to look for the thread named 'testConnectionCloseThread'
529     TEST_UTIL.waitFor(40000, new Waiter.Predicate<Exception>() {
530       @Override
531       public boolean evaluate() throws Exception {
532         return step.get() == 3;
533       }
534     });
535     table.close();
536     Assert.assertTrue("Unexpected exception is " + failed.get(), failed.get() == null);
537     TEST_UTIL.getHBaseAdmin().setBalancerRunning(previousBalance, true);
538   }

540   /**
541    * Test that connection can become idle without breaking everything.
542    */
543   @Test
544   public void testConnectionIdle() throws Exception {
545     TableName tableName = TableName.valueOf("HCM-testConnectionIdle");
546     TEST_UTIL.createTable(tableName, FAM_NAM).close();
547     int idleTime =  20000;
548     boolean previousBalance = TEST_UTIL.getHBaseAdmin().setBalancerRunning(false, true);
549 
550     Configuration c2 = new Configuration(TEST_UTIL.getConfiguration());
551     // We want to work on a separate connection.
552     c2.set(HConstants.HBASE_CLIENT_INSTANCE_ID, String.valueOf(-1));
553     c2.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 1); // Don't retry: retry = test failed
554     c2.setInt(RpcClient.IDLE_TIME, idleTime);
555 
556     final Table table = new HTable(c2, tableName);
557 
558     Put put = new Put(ROW);
559     put.add(FAM_NAM, ROW, ROW);
560     table.put(put);
561 
562     ManualEnvironmentEdge mee = new ManualEnvironmentEdge();
563     mee.setValue(System.currentTimeMillis());
564     EnvironmentEdgeManager.injectEdge(mee);
565     LOG.info("first get");
566     table.get(new Get(ROW));
567 
568     LOG.info("first get - changing the time & sleeping");
569     mee.incValue(idleTime + 1000);
570     Thread.sleep(1500); // we need to wait a little for the connection to be seen as idle.
571                         // 1500 = sleep time in RpcClient#waitForWork + a margin
572 
573     LOG.info("second get - connection has been marked idle in the middle");
574     // To check that the connection actually became idle would need to read some private
575     //  fields of RpcClient.
576     table.get(new Get(ROW));
577     mee.incValue(idleTime + 1000);
578 
579     LOG.info("third get - connection is idle, but the reader doesn't know yet");
580     // We're testing here a special case:
581     //  time limit reached BUT connection not yet reclaimed AND a new call.
582     //  in this situation, we don't close the connection, instead we use it immediately.
583     // If we're very unlucky we can have a race condition in the test: the connection is already
584     //  under closing when we do the get, so we have an exception, and we don't retry as the
