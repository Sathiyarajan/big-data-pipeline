41  @Category(SmallTests.class)
42  public class TestCompressionTest {
43    private static final Log LOG = LogFactory.getLog(TestCompressionTest.class);
44  
45    @Test
46    public void testExceptionCaching() {
47      // This test will fail if you run the tests with LZO compression available.
48      try {
49        CompressionTest.testCompression(Compression.Algorithm.LZO);
50        fail(); // always throws
51      } catch (IOException e) {
52        // there should be a 'cause'.
53        assertNotNull(e.getCause());
54      }
55  
56      // this is testing the caching of the test results.
57      try {
58        CompressionTest.testCompression(Compression.Algorithm.LZO);
59        fail(); // always throws
60      } catch (IOException e) {
61        // there should be NO cause because it's a direct exception not wrapped
62        assertNull(e.getCause());
63      }
64  
65      assertFalse(CompressionTest.testCompression("LZO"));
66    }
67  
68    @Test
69    public void testTestCompression() {
70      assertTrue(CompressionTest.testCompression("NONE"));
71      assertTrue(CompressionTest.testCompression("GZ"));
72  
73      if (NativeCodeLoader.isNativeCodeLoaded()) {
74        nativeCodecTest("LZO", "lzo2", "com.hadoop.compression.lzo.LzoCodec");
75        nativeCodecTest("LZ4", null, "org.apache.hadoop.io.compress.Lz4Codec");
76        nativeCodecTest("SNAPPY", "snappy", "org.apache.hadoop.io.compress.SnappyCodec");
77        nativeCodecTest("ZSTD", "zstd", "org.apache.hadoop.io.compress.ZStandardCodec");
78      } else {
79        // Hadoop nativelib is not available
80        LOG.debug("Native code not loaded");
81        assertFalse(CompressionTest.testCompression("LZO"));
82        assertFalse(CompressionTest.testCompression("LZ4"));
83        assertFalse(CompressionTest.testCompression("SNAPPY"));
84        assertFalse(CompressionTest.testCompression("ZSTD"));
85      }
86    }
87  
88    private boolean isCompressionAvailable(String codecClassName) {
89      try {
90        Thread.currentThread().getContextClassLoader().loadClass(codecClassName);
91        return true;
92      } catch (Exception ex) {
93        return false;
94      }
95    }
96  
97    /**
98     * Verify CompressionTest.testCompression() on a native codec.
99     */
100   private void nativeCodecTest(String codecName, String libName, String codecClassName) {
101     if (isCompressionAvailable(codecClassName)) {
102       try {
103         if (libName != null) {
104           System.loadLibrary(libName);
105         }
106 
107         try {
108             Configuration conf = new Configuration();
109             CompressionCodec codec = (CompressionCodec)
110               ReflectionUtils.newInstance(conf.getClassByName(codecClassName), conf);
111 
112             DataOutputBuffer compressedDataBuffer = new DataOutputBuffer();
113             CompressionOutputStream deflateFilter = codec.createOutputStream(compressedDataBuffer);
114 
115             byte[] data = new byte[1024];
116             DataOutputStream deflateOut = new DataOutputStream(new BufferedOutputStream(deflateFilter));
117             deflateOut.write(data, 0, data.length);
118             deflateOut.flush();
119             deflateFilter.finish();
120 
121             // Codec class, codec nativelib and Hadoop nativelib with codec JNIs are present
122             assertTrue(CompressionTest.testCompression(codecName));
123         } catch (UnsatisfiedLinkError e) {
124           // Hadoop nativelib does not have codec JNIs.
125           // cannot assert the codec here because the current logic of
126           // CompressionTest checks only classloading, not the codec
127           // usage.
128           LOG.debug("No JNI for codec '" + codecName + "' " + e.getMessage());
129         } catch (Exception e) {
130           LOG.error(codecName, e);
131         }
132       } catch (UnsatisfiedLinkError e) {
133         // nativelib is not available
134         LOG.debug("Native lib not available: " + codecName);
135         assertFalse(CompressionTest.testCompression(codecName));
136       }
137     } else {
138       // Compression Codec class is not available
139       LOG.debug("Codec class not available: " + codecName);
140       assertFalse(CompressionTest.testCompression(codecName));
141     }
142   }
