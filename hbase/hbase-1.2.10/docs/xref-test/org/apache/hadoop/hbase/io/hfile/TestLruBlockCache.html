47    @Test
48    public void testBackgroundEvictionThread() throws Exception {
49      long maxSize = 100000;
50      int numBlocks = 9;
51      long blockSize = calculateBlockSizeDefault(maxSize, numBlocks);
52      assertTrue("calculateBlockSize appears broken.", blockSize * numBlocks <= maxSize);
53  
54      LruBlockCache cache = new LruBlockCache(maxSize,blockSize);
55      EvictionThread evictionThread = cache.getEvictionThread();
56      assertTrue(evictionThread != null);
57  
58      CachedItem[] blocks = generateFixedBlocks(numBlocks + 1, blockSize, "block");
59  
60      // Make sure eviction thread has entered run method
61      while (!evictionThread.isEnteringRun()) {
62        Thread.sleep(1);
63      }
64  
65      // Add all the blocks
66      for (CachedItem block : blocks) {
67        cache.cacheBlock(block.cacheKey, block);
68      }
69  
70      // wait until at least one eviction has run
71      int n = 0;
72      while(cache.getStats().getEvictionCount() == 0) {
73        Thread.sleep(200);
74        assertTrue("Eviction never happened.", n++ < 20);
75      }
76  
77      // let cache stabilize
78      // On some systems, the cache will run multiple evictions before it attains
79      // steady-state. For instance, after populating the cache with 10 blocks,
80      // the first eviction evicts a single block and then a second eviction
81      // evicts another. I think this is due to the delta between minSize and
82      // acceptableSize, combined with variance between object overhead on
83      // different environments.
84      n = 0;
85      for (long prevCnt = 0 /* < number of blocks added */,
86                curCnt = cache.getBlockCount();
87          prevCnt != curCnt; prevCnt = curCnt, curCnt = cache.getBlockCount()) {
88        Thread.sleep(200);
89        assertTrue("Cache never stabilized.", n++ < 20);
90      }
91  
92      long evictionCount = cache.getStats().getEvictionCount();
93      assertTrue(evictionCount >= 1);
94      System.out.println("Background Evictions run: " + evictionCount);
95    }

112     }

128     }
129

131     long expectedBlockCount = cache.getBlockCount();
132     for (CachedItem block : blocks) {
133       cache.cacheBlock(block.cacheKey, block);
134     }
135     assertEquals(
136             "Cache should ignore cache requests for blocks already in cache",
137             expectedBlockCount, cache.getBlockCount());
138     
139     // Verify correctly calculated cache heap size
140     assertEquals(expectedCacheSize, cache.heapSize());
141

198 
199     long maxSize = 100000;
200     long blockSize = calculateBlockSizeDefault(maxSize, 10);
201 
202     LruBlockCache cache = new LruBlockCache(maxSize,blockSize,false);
203 
204     CachedItem [] singleBlocks = generateFixedBlocks(5, 10000, "single");
205     CachedItem [] multiBlocks = generateFixedBlocks(5, 10000, "multi");
206 
207     long expectedCacheSize = cache.heapSize();
208 
209     // Add and get the multi blocks
210     for (CachedItem block : multiBlocks) {
211       cache.cacheBlock(block.cacheKey, block);
212       expectedCacheSize += block.cacheBlockHeapSize();
213       assertEquals(cache.getBlock(block.cacheKey, true, false, true), block);
214     }
215 
216     // Add the single blocks (no get)
217     for (CachedItem block : singleBlocks) {
218       cache.cacheBlock(block.cacheKey, block);
219       expectedCacheSize += block.heapSize();
220     }
221 
222     // A single eviction run should have occurred
223     assertEquals(cache.getStats().getEvictionCount(), 1);
224 
225     // We expect two entries evicted
226     assertEquals(cache.getStats().getEvictedCount(), 2);
227 
228     // Our expected size overruns acceptable limit
229     assertTrue(expectedCacheSize >
230       (maxSize * LruBlockCache.DEFAULT_ACCEPTABLE_FACTOR));
231 
232     // But the cache did not grow beyond max
233     assertTrue(cache.heapSize() <= maxSize);
234 
235     // And is now below the acceptable limit
236     assertTrue(cache.heapSize() <=
237         (maxSize * LruBlockCache.DEFAULT_ACCEPTABLE_FACTOR));
238 
239     // We expect fairness across the two priorities.
240     // This test makes multi go barely over its limit, in-memory
241     // empty, and the rest in single.  Two single evictions and
242     // one multi eviction expected.
243     assertTrue(cache.getBlock(singleBlocks[0].cacheKey, true, false, true) == null);
244     assertTrue(cache.getBlock(multiBlocks[0].cacheKey, true, false, true) == null);
245 
246     // And all others to be cached
247     for(int i=1;i<4;i++) {
248       assertEquals(cache.getBlock(singleBlocks[i].cacheKey, true, false, true),
249           singleBlocks[i]);
250       assertEquals(cache.getBlock(multiBlocks[i].cacheKey, true, false, true),
251           multiBlocks[i]);
252     }

255   @Test
256   public void testCacheEvictionThreePriorities() throws Exception {
257 
258     long maxSize = 100000;
259     long blockSize = calculateBlockSize(maxSize, 10);
260 
261     LruBlockCache cache = new LruBlockCache(maxSize, blockSize, false,
262         (int)Math.ceil(1.2*maxSize/blockSize),
263         LruBlockCache.DEFAULT_LOAD_FACTOR,
264         LruBlockCache.DEFAULT_CONCURRENCY_LEVEL,
265         0.98f, // min
266         0.99f, // acceptable
267         0.33f, // single
268         0.33f, // multi
269         0.34f, // memory
270         1.2f,  // limit
271         false,
272         16 * 1024 * 1024);
273 
274     CachedItem [] singleBlocks = generateFixedBlocks(5, blockSize, "single");
275     CachedItem [] multiBlocks = generateFixedBlocks(5, blockSize, "multi");
276     CachedItem [] memoryBlocks = generateFixedBlocks(5, blockSize, "memory");
277 
278     long expectedCacheSize = cache.heapSize();
279 
280     // Add 3 blocks from each priority
281     for(int i=0;i<3;i++) {
282 
283       // Just add single blocks
284       cache.cacheBlock(singleBlocks[i].cacheKey, singleBlocks[i]);
285       expectedCacheSize += singleBlocks[i].cacheBlockHeapSize();
286 
287       // Add and get multi blocks
288       cache.cacheBlock(multiBlocks[i].cacheKey, multiBlocks[i]);
289       expectedCacheSize += multiBlocks[i].cacheBlockHeapSize();
290       cache.getBlock(multiBlocks[i].cacheKey, true, false, true);
291 
292       // Add memory blocks as such
293       cache.cacheBlock(memoryBlocks[i].cacheKey, memoryBlocks[i], true, false);
294       expectedCacheSize += memoryBlocks[i].cacheBlockHeapSize();
295 
296     }
297 
298     // Do not expect any evictions yet
299     assertEquals(0, cache.getStats().getEvictionCount());
300 
301     // Verify cache size
302     assertEquals(expectedCacheSize, cache.heapSize());
303 
304     // Insert a single block, oldest single should be evicted
305     cache.cacheBlock(singleBlocks[3].cacheKey, singleBlocks[3]);
306 
307     // Single eviction, one thing evicted
308     assertEquals(1, cache.getStats().getEvictionCount());
309     assertEquals(1, cache.getStats().getEvictedCount());
310 
311     // Verify oldest single block is the one evicted
312     assertEquals(null, cache.getBlock(singleBlocks[0].cacheKey, true, false, true));
313 
314     // Change the oldest remaining single block to a multi
315     cache.getBlock(singleBlocks[1].cacheKey, true, false, true);
316 
317     // Insert another single block
318     cache.cacheBlock(singleBlocks[4].cacheKey, singleBlocks[4]);
319 
320     // Two evictions, two evicted.
321     assertEquals(2, cache.getStats().getEvictionCount());
322     assertEquals(2, cache.getStats().getEvictedCount());
323 
324     // Oldest multi block should be evicted now
325     assertEquals(null, cache.getBlock(multiBlocks[0].cacheKey, true, false, true));
326 
327     // Insert another memory block
328     cache.cacheBlock(memoryBlocks[3].cacheKey, memoryBlocks[3], true, false);
329 
330     // Three evictions, three evicted.
331     assertEquals(3, cache.getStats().getEvictionCount());
332     assertEquals(3, cache.getStats().getEvictedCount());
333 
334     // Oldest memory block should be evicted now
335     assertEquals(null, cache.getBlock(memoryBlocks[0].cacheKey, true, false, true));
336 
337     // Add a block that is twice as big (should force two evictions)
338     CachedItem [] bigBlocks = generateFixedBlocks(3, blockSize*3, "big");
339     cache.cacheBlock(bigBlocks[0].cacheKey, bigBlocks[0]);
340 
341     // Four evictions, six evicted (inserted block 3X size, expect +3 evicted)
342     assertEquals(4, cache.getStats().getEvictionCount());
343     assertEquals(6, cache.getStats().getEvictedCount());
344 
345     // Expect three remaining singles to be evicted
346     assertEquals(null, cache.getBlock(singleBlocks[2].cacheKey, true, false, true));
347     assertEquals(null, cache.getBlock(singleBlocks[3].cacheKey, true, false, true));
348     assertEquals(null, cache.getBlock(singleBlocks[4].cacheKey, true, false, true));
349 
350     // Make the big block a multi block
351     cache.getBlock(bigBlocks[0].cacheKey, true, false, true);
352 
353     // Cache another single big block
354     cache.cacheBlock(bigBlocks[1].cacheKey, bigBlocks[1]);
355 
356     // Five evictions, nine evicted (3 new)
357     assertEquals(5, cache.getStats().getEvictionCount());
358     assertEquals(9, cache.getStats().getEvictedCount());
359 
360     // Expect three remaining multis to be evicted
361     assertEquals(null, cache.getBlock(singleBlocks[1].cacheKey, true, false, true));
362     assertEquals(null, cache.getBlock(multiBlocks[1].cacheKey, true, false, true));
363     assertEquals(null, cache.getBlock(multiBlocks[2].cacheKey, true, false, true));
364 
365     // Cache a big memory block
366     cache.cacheBlock(bigBlocks[2].cacheKey, bigBlocks[2], true, false);
367 
368     // Six evictions, twelve evicted (3 new)
369     assertEquals(6, cache.getStats().getEvictionCount());
370     assertEquals(12, cache.getStats().getEvictedCount());
371 
372     // Expect three remaining in-memory to be evicted
373     assertEquals(null, cache.getBlock(memoryBlocks[1].cacheKey, true, false, true));
374     assertEquals(null, cache.getBlock(memoryBlocks[2].cacheKey, true, false, true));
375     assertEquals(null, cache.getBlock(memoryBlocks[3].cacheKey, true, false, true));
376   }

378   @Test
379   public void testCacheEvictionInMemoryForceMode() throws Exception {
380     long maxSize = 100000;
381     long blockSize = calculateBlockSize(maxSize, 10);
382 
383     LruBlockCache cache = new LruBlockCache(maxSize, blockSize, false,
384         (int)Math.ceil(1.2*maxSize/blockSize),
385         LruBlockCache.DEFAULT_LOAD_FACTOR,
386         LruBlockCache.DEFAULT_CONCURRENCY_LEVEL,
387         0.98f, // min
388         0.99f, // acceptable
389         0.2f, // single
390         0.3f, // multi
391         0.5f, // memory
392         1.2f, // limit
393         true,
394         16 * 1024 * 1024);
395 
396     CachedItem [] singleBlocks = generateFixedBlocks(10, blockSize, "single");
397     CachedItem [] multiBlocks = generateFixedBlocks(10, blockSize, "multi");
398     CachedItem [] memoryBlocks = generateFixedBlocks(10, blockSize, "memory");
399 
400     long expectedCacheSize = cache.heapSize();
401 
402     // 0. Add 5 single blocks and 4 multi blocks to make cache full, si:mu:me = 5:4:0
403     for(int i = 0; i < 4; i++) {
404       // Just add single blocks
405       cache.cacheBlock(singleBlocks[i].cacheKey, singleBlocks[i]);
406       expectedCacheSize += singleBlocks[i].cacheBlockHeapSize();
407       // Add and get multi blocks
408       cache.cacheBlock(multiBlocks[i].cacheKey, multiBlocks[i]);
409       expectedCacheSize += multiBlocks[i].cacheBlockHeapSize();
410       cache.getBlock(multiBlocks[i].cacheKey, true, false, true);
411     }
412     // 5th single block
413     cache.cacheBlock(singleBlocks[4].cacheKey, singleBlocks[4]);
414     expectedCacheSize += singleBlocks[4].cacheBlockHeapSize();
415     // Do not expect any evictions yet
416     assertEquals(0, cache.getStats().getEvictionCount());
417     // Verify cache size
418     assertEquals(expectedCacheSize, cache.heapSize());
419 
420     // 1. Insert a memory block, oldest single should be evicted, si:mu:me = 4:4:1
421     cache.cacheBlock(memoryBlocks[0].cacheKey, memoryBlocks[0], true, false);
422     // Single eviction, one block evicted
423     assertEquals(1, cache.getStats().getEvictionCount());
424     assertEquals(1, cache.getStats().getEvictedCount());
425     // Verify oldest single block (index = 0) is the one evicted
426     assertEquals(null, cache.getBlock(singleBlocks[0].cacheKey, true, false, true));
427 
428     // 2. Insert another memory block, another single evicted, si:mu:me = 3:4:2
429     cache.cacheBlock(memoryBlocks[1].cacheKey, memoryBlocks[1], true, false);
430     // Two evictions, two evicted.
431     assertEquals(2, cache.getStats().getEvictionCount());
432     assertEquals(2, cache.getStats().getEvictedCount());
433     // Current oldest single block (index = 1) should be evicted now
434     assertEquals(null, cache.getBlock(singleBlocks[1].cacheKey, true, false, true));
435 
436     // 3. Insert 4 memory blocks, 2 single and 2 multi evicted, si:mu:me = 1:2:6
437     cache.cacheBlock(memoryBlocks[2].cacheKey, memoryBlocks[2], true, false);
438     cache.cacheBlock(memoryBlocks[3].cacheKey, memoryBlocks[3], true, false);
439     cache.cacheBlock(memoryBlocks[4].cacheKey, memoryBlocks[4], true, false);
440     cache.cacheBlock(memoryBlocks[5].cacheKey, memoryBlocks[5], true, false);
441     // Three evictions, three evicted.
442     assertEquals(6, cache.getStats().getEvictionCount());
443     assertEquals(6, cache.getStats().getEvictedCount());
444     // two oldest single blocks and two oldest multi blocks evicted
445     assertEquals(null, cache.getBlock(singleBlocks[2].cacheKey, true, false, true));
446     assertEquals(null, cache.getBlock(singleBlocks[3].cacheKey, true, false, true));
447     assertEquals(null, cache.getBlock(multiBlocks[0].cacheKey, true, false, true));
448     assertEquals(null, cache.getBlock(multiBlocks[1].cacheKey, true, false, true));
449 
450     // 4. Insert 3 memory blocks, the remaining 1 single and 2 multi evicted
451     // si:mu:me = 0:0:9
452     cache.cacheBlock(memoryBlocks[6].cacheKey, memoryBlocks[6], true, false);
453     cache.cacheBlock(memoryBlocks[7].cacheKey, memoryBlocks[7], true, false);
454     cache.cacheBlock(memoryBlocks[8].cacheKey, memoryBlocks[8], true, false);
455     // Three evictions, three evicted.
456     assertEquals(9, cache.getStats().getEvictionCount());
457     assertEquals(9, cache.getStats().getEvictedCount());
458     // one oldest single block and two oldest multi blocks evicted
459     assertEquals(null, cache.getBlock(singleBlocks[4].cacheKey, true, false, true));
460     assertEquals(null, cache.getBlock(multiBlocks[2].cacheKey, true, false, true));
461     assertEquals(null, cache.getBlock(multiBlocks[3].cacheKey, true, false, true));
462 
463     // 5. Insert one memory block, the oldest memory evicted
464     // si:mu:me = 0:0:9
465     cache.cacheBlock(memoryBlocks[9].cacheKey, memoryBlocks[9], true, false);
466     // one eviction, one evicted.
467     assertEquals(10, cache.getStats().getEvictionCount());
468     assertEquals(10, cache.getStats().getEvictedCount());
469     // oldest memory block evicted
470     assertEquals(null, cache.getBlock(memoryBlocks[0].cacheKey, true, false, true));
471 
472     // 6. Insert one new single block, itself evicted immediately since
473     //    all blocks in cache are memory-type which have higher priority
474     // si:mu:me = 0:0:9 (no change)
475     cache.cacheBlock(singleBlocks[9].cacheKey, singleBlocks[9]);
476     // one eviction, one evicted.
477     assertEquals(11, cache.getStats().getEvictionCount());
478     assertEquals(11, cache.getStats().getEvictedCount());
479     // the single block just cached now evicted (can't evict memory)
480     assertEquals(null, cache.getBlock(singleBlocks[9].cacheKey, true, false, true));
481   }

485   public void testScanResistance() throws Exception {
486 
487     long maxSize = 100000;
488     long blockSize = calculateBlockSize(maxSize, 10);
489 
490     LruBlockCache cache = new LruBlockCache(maxSize, blockSize, false,
491         (int)Math.ceil(1.2*maxSize/blockSize),
492         LruBlockCache.DEFAULT_LOAD_FACTOR,
493         LruBlockCache.DEFAULT_CONCURRENCY_LEVEL,
494         0.66f, // min
495         0.99f, // acceptable
496         0.33f, // single
497         0.33f, // multi
498         0.34f, // memory
499         1.2f,  // limit
500         false,
501         16 * 1024 * 1024);
502 
503     CachedItem [] singleBlocks = generateFixedBlocks(20, blockSize, "single");
504     CachedItem [] multiBlocks = generateFixedBlocks(5, blockSize, "multi");
505 
506     // Add 5 multi blocks
507     for (CachedItem block : multiBlocks) {
508       cache.cacheBlock(block.cacheKey, block);
509       cache.getBlock(block.cacheKey, true, false, true);
510     }
511 
512     // Add 5 single blocks
513     for(int i=0;i<5;i++) {
514       cache.cacheBlock(singleBlocks[i].cacheKey, singleBlocks[i]);
515     }
516 
517     // An eviction ran
518     assertEquals(1, cache.getStats().getEvictionCount());
519 
520     // To drop down to 2/3 capacity, we'll need to evict 4 blocks
521     assertEquals(4, cache.getStats().getEvictedCount());
522 
523     // Should have been taken off equally from single and multi
524     assertEquals(null, cache.getBlock(singleBlocks[0].cacheKey, true, false, true));
525     assertEquals(null, cache.getBlock(singleBlocks[1].cacheKey, true, false, true));
526     assertEquals(null, cache.getBlock(multiBlocks[0].cacheKey, true, false, true));
527     assertEquals(null, cache.getBlock(multiBlocks[1].cacheKey, true, false, true));
528 
529     // Let's keep "scanning" by adding single blocks.  From here on we only
530     // expect evictions from the single bucket.
531 
532     // Every time we reach 10 total blocks (every 4 inserts) we get 4 single
533     // blocks evicted.  Inserting 13 blocks should yield 3 more evictions and
534     // 12 more evicted.
535 
536     for(int i=5;i<18;i++) {
537       cache.cacheBlock(singleBlocks[i].cacheKey, singleBlocks[i]);
538     }
539 
540     // 4 total evictions, 16 total evicted
541     assertEquals(4, cache.getStats().getEvictionCount());
542     assertEquals(16, cache.getStats().getEvictedCount());
543 
544     // Should now have 7 total blocks
545     assertEquals(7, cache.getBlockCount());
546 
547   }

763   private long calculateBlockSize(long maxSize, int numBlocks) {
764     long roughBlockSize = maxSize / numBlocks;
765     int numEntries = (int)Math.ceil((1.2)*maxSize/roughBlockSize);
766     long totalOverhead = LruBlockCache.CACHE_FIXED_OVERHEAD +
767         ClassSize.CONCURRENT_HASHMAP +
768         (numEntries * ClassSize.CONCURRENT_HASHMAP_ENTRY) +
769         (LruBlockCache.DEFAULT_CONCURRENCY_LEVEL * ClassSize.CONCURRENT_HASHMAP_SEGMENT);
770     long negateBlockSize = (long)(totalOverhead/numEntries);
771     negateBlockSize += LruCachedBlock.PER_BLOCK_OVERHEAD;
772     return ClassSize.align((long)Math.floor((roughBlockSize - negateBlockSize)*0.99f));
773   }

775   private long calculateBlockSizeDefault(long maxSize, int numBlocks) {
776     long roughBlockSize = maxSize / numBlocks;
777     int numEntries = (int)Math.ceil((1.2)*maxSize/roughBlockSize);
778     long totalOverhead = LruBlockCache.CACHE_FIXED_OVERHEAD +
779         ClassSize.CONCURRENT_HASHMAP +
780         (numEntries * ClassSize.CONCURRENT_HASHMAP_ENTRY) +
781         (LruBlockCache.DEFAULT_CONCURRENCY_LEVEL * ClassSize.CONCURRENT_HASHMAP_SEGMENT);
782     long negateBlockSize = totalOverhead / numEntries;
783     negateBlockSize += LruCachedBlock.PER_BLOCK_OVERHEAD;
784     return ClassSize.align((long)Math.floor((roughBlockSize - negateBlockSize)*
785         LruBlockCache.DEFAULT_ACCEPTABLE_FACTOR));
