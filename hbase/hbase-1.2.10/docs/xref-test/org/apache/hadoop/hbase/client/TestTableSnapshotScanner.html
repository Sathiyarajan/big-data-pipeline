105     util.loadTable(table, FAMILIES, value);

113   public void testNoDuplicateResultsWhenSplitting() throws Exception {
114     setupCluster();
115     TableName tableName = TableName.valueOf("testNoDuplicateResultsWhenSplitting");
116     String snapshotName = "testSnapshotBug";
117     try {
118       if (UTIL.getHBaseAdmin().tableExists(tableName)) {
119         UTIL.deleteTable(tableName);
120       }
121 
122       UTIL.createTable(tableName, FAMILIES);
123       Admin admin = UTIL.getHBaseAdmin();
124 
125       // put some stuff in the table
126       Table table = UTIL.getConnection().getTable(tableName);
127       UTIL.loadTable(table, FAMILIES);
128 
129       // split to 2 regions
130       admin.split(tableName, Bytes.toBytes("eee"));
131       TestTableSnapshotInputFormat.blockUntilSplitFinished(UTIL, tableName, 2);
132 
133       Path rootDir = FSUtils.getRootDir(UTIL.getConfiguration());
134       FileSystem fs = rootDir.getFileSystem(UTIL.getConfiguration());
135 
136       SnapshotTestingUtils.createSnapshotAndValidate(admin, tableName,
137         Arrays.asList(FAMILIES), null, snapshotName, rootDir, fs, true);
138 
139       // load different values
140       byte[] value = Bytes.toBytes("after_snapshot_value");
141       UTIL.loadTable(table, FAMILIES, value);
142 
143       // cause flush to create new files in the region
144       admin.flush(tableName);
145       table.close();
146 
147       Path restoreDir = UTIL.getDataTestDirOnTestFS(snapshotName);
148       Scan scan = new Scan().setStartRow(bbb).setStopRow(yyy); // limit the scan
149 
150       TableSnapshotScanner scanner =
151           new TableSnapshotScanner(UTIL.getConfiguration(), restoreDir, snapshotName, scan);
152 
153       verifyScanner(scanner, bbb, yyy);
154       scanner.close();
155     } finally {
156       UTIL.getHBaseAdmin().deleteSnapshot(snapshotName);
157       UTIL.deleteTable(tableName);
158       tearDownCluster();
159     }

177   private void testScanner(HBaseTestingUtility util, String snapshotName, int numRegions,
178       boolean shutdownCluster) throws Exception {
179     setupCluster();
180     TableName tableName = TableName.valueOf("testScanner");
181     try {
182       createTableAndSnapshot(util, tableName, snapshotName, numRegions);
183 
184       if (shutdownCluster) {
185         util.shutdownMiniHBaseCluster();
186       }
187

189       Scan scan = new Scan(bbb, yyy); // limit the scan

195       scanner.close();
196     } finally {
197       if (!shutdownCluster) {
198         util.getHBaseAdmin().deleteSnapshot(snapshotName);
199         util.deleteTable(tableName);
200         tearDownCluster();
201       }
202     }

205   private void verifyScanner(ResultScanner scanner, byte[] startRow, byte[] stopRow)
206       throws IOException, InterruptedException {
207 
208     HBaseTestingUtility.SeenRowTracker rowTracker =
209         new HBaseTestingUtility.SeenRowTracker(startRow, stopRow);
210 
211     while (true) {
212       Result result = scanner.next();
213       if (result == null) {
214         break;
215       }
216       verifyRow(result);
217       rowTracker.addRow(result.getRow());
218     }
219 
220     // validate all rows are seen
221     rowTracker.validate();
222   }
223 
224   private static void verifyRow(Result result) throws IOException {
225     byte[] row = result.getRow();
226     CellScanner scanner = result.cellScanner();
227     while (scanner.advance()) {
228       Cell cell = scanner.current();
229 
230       //assert that all Cells in the Result have the same key
231      Assert.assertEquals(0, Bytes.compareTo(row, 0, row.length,
232          cell.getRowArray(), cell.getRowOffset(), cell.getRowLength()));
233     }
234 
235     for (int j = 0; j < FAMILIES.length; j++) {
236       byte[] actual = result.getValue(FAMILIES[j], FAMILIES[j]);
237       Assert.assertArrayEquals("Row in snapshot does not match, expected:" + Bytes.toString(row)
238           + " ,actual:" + Bytes.toString(actual), row, actual);
239     }
240   }
