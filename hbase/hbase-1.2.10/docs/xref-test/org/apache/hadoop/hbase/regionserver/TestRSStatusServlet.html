56   * Tests for the region server status page and its template.

58  @Category(SmallTests.class)
59  public class TestRSStatusServlet {
60    private static final Log LOG = LogFactory.getLog(TestRSStatusServlet.class);
61    private HRegionServer rs;
62    private RSRpcServices rpcServices;
63    private RpcServerInterface rpcServer;
64  
65    static final int FAKE_IPC_PORT = 1585;
66    static final int FAKE_WEB_PORT = 1586;
67  
68    private final ServerName fakeServerName =
69        ServerName.valueOf("localhost", FAKE_IPC_PORT, 11111);
70    private final GetServerInfoResponse fakeResponse =
71      ResponseConverter.buildGetServerInfoResponse(fakeServerName, FAKE_WEB_PORT);
72  
73    private final ServerName fakeMasterAddress =
74        ServerName.valueOf("localhost", 60010, 1212121212);
75  
76    @Before
77    public void setupBasicMocks() throws IOException, ServiceException {
78      rs = Mockito.mock(HRegionServer.class);
79      rpcServices = Mockito.mock(RSRpcServices.class);
80      rpcServer = Mockito.mock(RpcServerInterface.class);
81      Mockito.doReturn(HBaseConfiguration.create())
82        .when(rs).getConfiguration();
83      Mockito.doReturn(rpcServices).when(rs).getRSRpcServices();
84      Mockito.doReturn(rpcServer).when(rs).getRpcServer();
85      Mockito.doReturn(fakeResponse).when(rpcServices).getServerInfo(
86        (RpcController)Mockito.any(), (GetServerInfoRequest)Mockito.any());
87      // Fake ZKW
88      ZooKeeperWatcher zkw = Mockito.mock(ZooKeeperWatcher.class);
89      Mockito.doReturn("fakequorum").when(zkw).getQuorum();
90      Mockito.doReturn(zkw).when(rs).getZooKeeper();
91  
92      // Fake CacheConfig
93      LOG.warn("The " + HConstants.HFILE_BLOCK_CACHE_SIZE_KEY + " is set to 0");
94      CacheConfig cacheConf = Mockito.mock(CacheConfig.class);
95      Mockito.doReturn(null).when(cacheConf).getBlockCache();
96      Mockito.doReturn(cacheConf).when(rs).getCacheConfig();
97      
98      // Fake MasterAddressTracker
99      MasterAddressTracker mat = Mockito.mock(MasterAddressTracker.class);
100     Mockito.doReturn(fakeMasterAddress).when(mat).getMasterAddress();
101     Mockito.doReturn(mat).when(rs).getMasterAddressTracker();
102 
103     MetricsRegionServer rms = Mockito.mock(MetricsRegionServer.class);
104     Mockito.doReturn(new MetricsRegionServerWrapperStub()).when(rms).getRegionServerWrapper();
105     Mockito.doReturn(rms).when(rs).getRegionServerMetrics();
106 
107     MetricsHBaseServer ms = Mockito.mock(MetricsHBaseServer.class);
108     Mockito.doReturn(new MetricsHBaseServerWrapperStub()).when(ms).getHBaseServerWrapper();
109     Mockito.doReturn(ms).when(rpcServer).getMetrics();
110   }
111   
112   @Test
113   public void testBasic() throws IOException, ServiceException {
114     new RSStatusTmpl().render(new StringWriter(), rs);
115   }

117   @Test
118   public void testWithRegions() throws IOException, ServiceException {
119     HTableDescriptor htd = new HTableDescriptor(TableName.valueOf("mytable"));
120     List<HRegionInfo> regions = Lists.newArrayList(
121         new HRegionInfo(htd.getTableName(), Bytes.toBytes("a"), Bytes.toBytes("d")),
122         new HRegionInfo(htd.getTableName(), Bytes.toBytes("d"), Bytes.toBytes("z"))
123         );
124     Mockito.doReturn(ResponseConverter.buildGetOnlineRegionResponse(
125       regions)).when(rpcServices).getOnlineRegion((RpcController)Mockito.any(),
126         (GetOnlineRegionRequest)Mockito.any());
127     
128     new RSStatusTmpl().render(new StringWriter(), rs);
129   }
