103    * Make sure that there is only one snapshot returned from the master and its
104    * name and table match the passed in parameters.

106   public static List<SnapshotDescription> assertExistsMatchingSnapshot(
107       Admin admin, String snapshotName, TableName tableName)
108       throws IOException {
109     // list the snapshot
110     List<SnapshotDescription> snapshots = admin.listSnapshots();
111 
112     List<SnapshotDescription> returnedSnapshots = new ArrayList<SnapshotDescription>();
113     for (SnapshotDescription sd : snapshots) {
114       if (snapshotName.equals(sd.getName()) &&
115           tableName.equals(TableName.valueOf(sd.getTable()))) {
116         returnedSnapshots.add(sd);
117       }
118     }
119 
120     Assert.assertTrue("No matching snapshots found.", returnedSnapshots.size()>0);
121     return returnedSnapshots;
122   }

195   /**
196    * Confirm that the snapshot contains references to all the files that should
197    * be in the snapshot. This method also perform some redundant check like
198    * the existence of the snapshotinfo or the regioninfo which are done always
199    * by the MasterSnapshotVerifier, at the end of the snapshot operation.
200    */
201   public static void confirmSnapshotValid(
202       SnapshotDescription snapshotDescriptor, TableName tableName,
203       List<byte[]> nonEmptyTestFamilies, List<byte[]> emptyTestFamilies,
204       Path rootDir, Admin admin, FileSystem fs) throws IOException {
205     final Configuration conf = admin.getConfiguration();
206 
207     // check snapshot dir
208     Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(
209         snapshotDescriptor, rootDir);
210     assertTrue(fs.exists(snapshotDir));
211 
212     SnapshotDescription desc = SnapshotDescriptionUtils.readSnapshotInfo(fs, snapshotDir);
213 
214     // Extract regions and families with store files
215     final Set<byte[]> snapshotFamilies = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
216 
217     SnapshotManifest manifest = SnapshotManifest.open(conf, fs, snapshotDir, desc);
218     Map<String, SnapshotRegionManifest> regionManifests = manifest.getRegionManifestsMap();
219     for (SnapshotRegionManifest regionManifest: regionManifests.values()) {
220       SnapshotReferenceUtil.visitRegionStoreFiles(regionManifest,
221           new SnapshotReferenceUtil.StoreFileVisitor() {
222         @Override
223         public void storeFile(final HRegionInfo regionInfo, final String family,
224               final SnapshotRegionManifest.StoreFile storeFile) throws IOException {
225           snapshotFamilies.add(Bytes.toBytes(family));
226         }
227       });
228     }
229 
230     // Verify that there are store files in the specified families
231     if (nonEmptyTestFamilies != null) {
232       for (final byte[] familyName: nonEmptyTestFamilies) {
233         assertTrue(snapshotFamilies.contains(familyName));
234       }
235     }
236 
237     // Verify that there are no store files in the specified families
238     if (emptyTestFamilies != null) {
239       for (final byte[] familyName: emptyTestFamilies) {
240         assertFalse(snapshotFamilies.contains(familyName));
241       }
242     }
243 
244     // check the region snapshot for all the regions
245     List<HRegionInfo> regions = admin.getTableRegions(tableName);
246     // remove the non-default regions
247     RegionReplicaUtil.removeNonDefaultRegions(regions);
248 
249     // if create snapshot when table splitting, parent region will be included to the snapshot
250     // region manifest. we should exclude the parent regions.
251     int regionCountExclusiveSplitParent = 0;
252     for (SnapshotRegionManifest snapshotRegionManifest : regionManifests.values()) {
253       HRegionInfo hri = HRegionInfo.convert(snapshotRegionManifest.getRegionInfo());
254       if (hri.isOffline() && (hri.isSplit() || hri.isSplitParent())) {
255         continue;
256       }
257       regionCountExclusiveSplitParent++;
258     }
259     assertEquals(regions.size(), regionCountExclusiveSplitParent);
260 
261     // Verify Regions (redundant check, see MasterSnapshotVerifier)
262     for (HRegionInfo info : regions) {
263       String regionName = info.getEncodedName();
264       assertTrue(regionManifests.containsKey(regionName));
265     }
266   }

268   /**
269    * Helper method for testing async snapshot operations. Just waits for the
270    * given snapshot to complete on the server by repeatedly checking the master.
271    *
272    * @param master: the master running the snapshot
273    * @param snapshot: the snapshot to check
274    * @param sleep: amount to sleep between checks to see if the snapshot is done
275    * @throws ServiceException if the snapshot fails
276    */
277   public static void waitForSnapshotToComplete(HMaster master,
278       SnapshotDescription snapshot, long sleep) throws ServiceException {
279     final IsSnapshotDoneRequest request = IsSnapshotDoneRequest.newBuilder()
280         .setSnapshot(snapshot).build();
281     IsSnapshotDoneResponse done = IsSnapshotDoneResponse.newBuilder()
282         .buildPartial();
283     while (!done.getDone()) {
284       done = master.getMasterRpcServices().isSnapshotDone(null, request);
285       try {
286         Thread.sleep(sleep);
287       } catch (InterruptedException e) {
288         throw new ServiceException(e);
289       }
290     }
291   }

293   /*
294    * Take snapshot with maximum of numTries attempts, ignoring CorruptedSnapshotException
295    * except for the last CorruptedSnapshotException
296    */
297   public static void snapshot(Admin admin,
298       final String snapshotName, final String tableName,
299       SnapshotDescription.Type type, int numTries) throws IOException {
300     int tries = 0;
301     CorruptedSnapshotException lastEx = null;
302     while (tries++ < numTries) {
303       try {
304         admin.snapshot(snapshotName, TableName.valueOf(tableName), type);
305         return;
306       } catch (CorruptedSnapshotException cse) {
307         LOG.warn("Got CorruptedSnapshotException", cse);
308         lastEx = cse;
309       }
310     }
311     throw lastEx;
312   }

327    * Expect the snapshot to throw an error when checking if the snapshot is
328    * complete
329    *
330    * @param master master to check
331    * @param snapshot the {@link SnapshotDescription} request to pass to the master
332    * @param clazz expected exception from the master
333    */
334   public static void expectSnapshotDoneException(HMaster master,
335       IsSnapshotDoneRequest snapshot,
336       Class<? extends HBaseSnapshotException> clazz) {
337     try {
338       master.getMasterRpcServices().isSnapshotDone(null, snapshot);
339       Assert.fail("didn't fail to lookup a snapshot");
340     } catch (ServiceException se) {
341       try {
342         throw ProtobufUtil.getRemoteException(se);
343       } catch (HBaseSnapshotException e) {
344         assertEquals("Threw wrong snapshot exception!", clazz, e.getClass());
345       } catch (Throwable t) {
346         Assert.fail("Threw an unexpected exception:" + t);
347       }
348     }

392   public static void createSnapshotAndValidate(Admin admin,
393       TableName tableName, List<byte[]> nonEmptyFamilyNames, List<byte[]> emptyFamilyNames,

413       emptyFamilyNames, rootDir, admin, fs);

417    * Corrupt the specified snapshot by deleting some files.
418    *
419    * @param util {@link HBaseTestingUtility}
420    * @param snapshotName name of the snapshot to corrupt
421    * @return array of the corrupted HFiles
422    * @throws IOException on unexecpted error reading the FS
423    */
424   public static ArrayList corruptSnapshot(final HBaseTestingUtility util, final String snapshotName)
425       throws IOException {
426     final MasterFileSystem mfs = util.getHBaseCluster().getMaster().getMasterFileSystem();
427     final FileSystem fs = mfs.getFileSystem();
428 
429     Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName,
430                                                                         mfs.getRootDir());
431     SnapshotDescription snapshotDesc = SnapshotDescriptionUtils.readSnapshotInfo(fs, snapshotDir);
432     final TableName table = TableName.valueOf(snapshotDesc.getTable());
433 
434     final ArrayList corruptedFiles = new ArrayList();
435     final Configuration conf = util.getConfiguration();
436     SnapshotReferenceUtil.visitTableStoreFiles(conf, fs, snapshotDir, snapshotDesc,
437         new SnapshotReferenceUtil.StoreFileVisitor() {
438       @Override
439       public void storeFile(final HRegionInfo regionInfo, final String family,
440             final SnapshotRegionManifest.StoreFile storeFile) throws IOException {
441         String region = regionInfo.getEncodedName();
442         String hfile = storeFile.getName();
443         HFileLink link = HFileLink.build(conf, table, region, family, hfile);
444         if (corruptedFiles.size() % 2 == 0) {
445           fs.delete(link.getAvailablePath(fs), true);
446           corruptedFiles.add(hfile);
447         }
448       }
449     });
450 
451     assertTrue(corruptedFiles.size() > 0);
452     return corruptedFiles;
453   }

531       private Path[] addRegion(final SnapshotDescription desc) throws IOException {
532         if (this.snapshotted == tableRegions.length) {
533           throw new UnsupportedOperationException("No more regions in the table");
534         }

537         ForeignExceptionDispatcher monitor = new ForeignExceptionDispatcher(desc.getName());
538         SnapshotManifest manifest = SnapshotManifest.create(conf, fs, snapshotDir, desc, monitor);
539         manifest.addRegion(regionData.tableDir, regionData.hri);
540         return regionData.files;

543       private void corruptFile(Path p) throws IOException {
544         String manifestName = p.getName();
545 
546         // Rename the original region-manifest file
547         Path newP = new Path(p.getParent(), manifestName + "1");
548         fs.rename(p, newP);
549 
550         // Create a new region-manifest file
551         FSDataOutputStream out = fs.create(p);
552 
553         //Copy the first 25 bytes of the original region-manifest into the new one,
554         //make it a corrupted region-manifest file.
555         FSDataInputStream input = fs.open(newP);
556         byte[] buffer = new byte[25];
557         int len = input.read(0, buffer, 0, 25);
558         if (len > 1) {
559           out.write(buffer, 0, len - 1);
560         }
561         out.close();
562 
563         // Delete the original region-manifest
564         fs.delete(newP);
565       }

620         SnapshotManifest manifest = SnapshotManifest.create(conf, fs, snapshotDir, desc, monitor);

623         SnapshotDescriptionUtils.completeSnapshot(desc, rootDir, snapshotDir, fs);
624         snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(desc, rootDir);

689     private RegionData[] createTable(final HTableDescriptor htd, final int nregions)
690         throws IOException {
691       Path tableDir = FSUtils.getTableDir(rootDir, htd.getTableName());
692       new FSTableDescriptors(conf).createTableDescriptorForTableDirectory(tableDir, htd, false);
693 
694       assertTrue(nregions % 2 == 0);
695       RegionData[] regions = new RegionData[nregions];
696       for (int i = 0; i < regions.length; i += 2) {
697         byte[] startKey = Bytes.toBytes(0 + i * 2);
698         byte[] endKey = Bytes.toBytes(1 + i * 2);
699 
700         // First region, simple with one plain hfile.
701         HRegionInfo hri = new HRegionInfo(htd.getTableName(), startKey, endKey);
702         HRegionFileSystem rfs = HRegionFileSystem.createRegionOnFileSystem(conf, fs, tableDir, hri);
703         regions[i] = new RegionData(tableDir, hri, 3);
704         for (int j = 0; j < regions[i].files.length; ++j) {
705           Path storeFile = createStoreFile(rfs.createTempName());
706           regions[i].files[j] = rfs.commitStoreFile(TEST_FAMILY, storeFile);
707         }
708 
709         // Second region, used to test the split case.
710         // This region contains a reference to the hfile in the first region.
711         startKey = Bytes.toBytes(2 + i * 2);
712         endKey = Bytes.toBytes(3 + i * 2);
713         hri = new HRegionInfo(htd.getTableName());
714         rfs = HRegionFileSystem.createRegionOnFileSystem(conf, fs, tableDir, hri);
715         regions[i+1] = new RegionData(tableDir, hri, regions[i].files.length);
716         for (int j = 0; j < regions[i].files.length; ++j) {
717           String refName = regions[i].files[j].getName() + '.' + regions[i].hri.getEncodedName();
718           Path refFile = createStoreFile(new Path(rootDir, refName));
719           regions[i+1].files[j] = rfs.commitStoreFile(TEST_FAMILY, refFile);
720         }
721       }
722       return regions;
723     }

786   public static void loadData(final HBaseTestingUtility util, final BufferedMutator mutator, int rows,
787       byte[]... families) throws IOException, InterruptedException {
788     // Ensure one row per region
789     assertTrue(rows >= KEYS.length);
790     for (byte k0: KEYS) {
791       byte[] k = new byte[] { k0 };
792       byte[] value = Bytes.add(Bytes.toBytes(System.currentTimeMillis()), k);
793       byte[] key = Bytes.add(k, Bytes.toBytes(MD5Hash.getMD5AsHex(value)));
794       final byte[][] families1 = families;
795       final byte[] key1 = key;
796       final byte[] value1 = value;
797       mutator.mutate(createPut(families1, key1, value1));
798       rows--;
799     }
800 
801     // Add other extra rows. more rows, more files
802     while (rows-- > 0) {
803       byte[] value = Bytes.add(Bytes.toBytes(System.currentTimeMillis()), Bytes.toBytes(rows));
804       byte[] key = Bytes.toBytes(MD5Hash.getMD5AsHex(value));
805       final byte[][] families1 = families;
806       final byte[] key1 = key;
807       final byte[] value1 = value;
808       mutator.mutate(createPut(families1, key1, value1));
809     }
810     mutator.flush();
811 
812     waitForTableToBeOnline(util, mutator.getName());
813   }

819     for (byte[] family: families) {
