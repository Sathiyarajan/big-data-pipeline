130     @Override
131     protected void handleReportForDutyResponse(
132         final RegionServerStartupResponse c) throws IOException {
133       super.handleReportForDutyResponse(c);
134       // Run this thread to shutdown our filesystem on way out.
135       this.shutdownThread = new SingleFileSystemShutdownThread(getFileSystem());
136     }

138     @Override
139     public void run() {
140       try {
141         this.user.runAs(new PrivilegedAction<Object>(){
142           public Object run() {
143             runRegionServer();
144             return null;
145           }
146         });
147       } catch (Throwable t) {
148         LOG.error("Exception in run", t);
149       } finally {
150         // Run this on the way out.
151         if (this.shutdownThread != null) {
152           this.shutdownThread.start();
153           Threads.shutdown(this.shutdownThread, 30000);
154         }
155       }
156     }
157 
158     private void runRegionServer() {
159       super.run();
160     }

164       super.kill();
165     }

168       this.user.runAs(new PrivilegedAction<Object>() {
169         public Object run() {
170           abortRegionServer(reason, cause);
171           return null;
172         }
173       });
174     }

178     }

181   /**
182    * Alternate shutdown hook.
183    * Just shuts down the passed fs, not all as default filesystem hook does.
184    */
185   static class SingleFileSystemShutdownThread extends Thread {
186     private final FileSystem fs;
187     SingleFileSystemShutdownThread(final FileSystem fs) {
188       super("Shutdown of " + fs);
189       this.fs = fs;
190     }
191     @Override
192     public void run() {
193       try {
194         LOG.info("Hook closing fs=" + this.fs);
195         this.fs.close();
196       } catch (NullPointerException npe) {
197         LOG.debug("Need to fix these: " + npe.toString());
198       } catch (IOException e) {
199         LOG.warn("Running hook", e);
200       }
201     }
202   }

204   private void init(final int nMasterNodes, final int nRegionNodes,
205                  Class<? extends HMaster> masterClass,
206                  Class<? extends MiniHBaseCluster.MiniHBaseClusterRegionServer> regionserverClass)
207   throws IOException, InterruptedException {
208     try {
209       if (masterClass == null){
210         masterClass =  HMaster.class;
211       }
212       if (regionserverClass == null){
213         regionserverClass = MiniHBaseCluster.MiniHBaseClusterRegionServer.class;
214       }
215 
216       // start up a LocalHBaseCluster
217       hbaseCluster = new LocalHBaseCluster(conf, nMasterNodes, 0,
218           masterClass, regionserverClass);
219 
220       // manually add the regionservers as other users
221       for (int i=0; i<nRegionNodes; i++) {
222         Configuration rsConf = HBaseConfiguration.create(conf);
223         User user = HBaseTestingUtility.getDifferentUser(rsConf,
224             ".hfs."+index++);
225         hbaseCluster.addRegionServer(rsConf, i, user);
226       }
227 
228       hbaseCluster.startup();
229     } catch (IOException e) {
230       shutdown();
231       throw e;
232     } catch (Throwable t) {
233       LOG.error("Error starting cluster", t);
234       shutdown();
235       throw new IOException("Shutting down", t);
236     }
237   }

519   /**
520    * Blocks until there is an active master and that master has completed
521    * initialization.
522    *
523    * @return true if an active master becomes available.  false if there are no
524    *         masters left.
525    * @throws InterruptedException
526    */
527   public boolean waitForActiveAndReadyMaster(long timeout) throws IOException {
528     List<JVMClusterUtil.MasterThread> mts;
529     long start = System.currentTimeMillis();
530     while (!(mts = getMasterThreads()).isEmpty()
531         && (System.currentTimeMillis() - start) < timeout) {
532       for (JVMClusterUtil.MasterThread mt : mts) {
533         if (mt.getMaster().isActiveMaster() && mt.getMaster().isInitialized()) {
534           return true;
535         }
536       }
537 
538       Threads.sleep(100);
539     }
540     return false;
541   }

668   public List<HRegion> getRegions(TableName tableName) {
669     List<HRegion> ret = new ArrayList<HRegion>();
670     for (JVMClusterUtil.RegionServerThread rst : getRegionServerThreads()) {
671       HRegionServer hrs = rst.getRegionServer();
672       for (Region region : hrs.getOnlineRegionsLocalContext()) {
673         if (region.getTableDesc().getTableName().equals(tableName)) {
674           ret.add((HRegion)region);
675         }
676       }
677     }
678     return ret;

689   /**
690    * Get the location of the specified region
691    * @param regionName Name of the region in bytes
692    * @return Index into List of {@link MiniHBaseCluster#getRegionServerThreads()}
693    * of HRS carrying hbase:meta. Returns -1 if none found.
694    */
695   public int getServerWith(byte[] regionName) {
696     int index = -1;
697     int count = 0;
698     for (JVMClusterUtil.RegionServerThread rst: getRegionServerThreads()) {
699       HRegionServer hrs = rst.getRegionServer();
700       if (!hrs.isStopped()) {
701         Region region = hrs.getOnlineRegion(regionName);
702         if (region != null) {
703           index = count;
704           break;
705         }
706       }
707       count++;
708     }
709     return index;
710   }

712   @Override
713   public ServerName getServerHoldingRegion(final TableName tn, byte[] regionName)
714   throws IOException {
715     // Assume there is only one master thread which is the active master.
716     // If there are multiple master threads, the backup master threads
717     // should hold some regions. Please refer to #countServedRegions
718     // to see how we find out all regions.
719     HMaster master = getMaster();
720     Region region = master.getOnlineRegion(regionName);
721     if (region != null) {
722       return master.getServerName();
723     }
724     int index = getServerWith(regionName);
725     if (index < 0) {
726       return null;
727     }
728     return getRegionServer(index).getServerName();
729   }
