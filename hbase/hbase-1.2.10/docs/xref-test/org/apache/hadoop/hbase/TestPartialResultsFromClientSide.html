55  /**
56   * These tests are focused on testing how partial results appear to a client. Partial results are
57   * {@link Result}s that contain only a portion of a row's complete list of cells. Partial results
58   * are formed when the server breaches its maximum result size when trying to service a client's RPC
59   * request. It is the responsibility of the scanner on the client side to recognize when partial
60   * results have been returned and to take action to form the complete results.
61   * <p>
62   * Unless the flag {@link Scan#setAllowPartialResults(boolean)} has been set to true, the caller of
63   * {@link ResultScanner#next()} should never see partial results.
64   */

251   public void testOrderingOfCellsInPartialResults(final Scan basePartialScan) throws Exception {
252     // Scan that retrieves results in pieces (partials). By setting allowPartialResults to be true
253     // the results will NOT be reconstructed and instead the caller will see the partial results
254     // returned by the server
255     Scan partialScan = new Scan(basePartialScan);
256     partialScan.setAllowPartialResults(true);
257     ResultScanner partialScanner = TABLE.getScanner(partialScan);
258 
259     // Scan that retrieves all table results in single RPC request
260     Scan oneShotScan = new Scan(basePartialScan);
261     oneShotScan.setMaxResultSize(Long.MAX_VALUE);
262     oneShotScan.setCaching(ROWS.length);
263     ResultScanner oneShotScanner = TABLE.getScanner(oneShotScan);
264 
265     Result oneShotResult = oneShotScanner.next();
266     Result partialResult = null;
267     int iterationCount = 0;
268 
269     while (oneShotResult != null && oneShotResult.rawCells() != null) {
270       List<Cell> aggregatePartialCells = new ArrayList<Cell>();
271       do {
272         partialResult = partialScanner.next();
273         assertTrue("Partial Result is null. iteration: " + iterationCount, partialResult != null);
274         assertTrue("Partial cells are null. iteration: " + iterationCount,
275             partialResult.rawCells() != null);
276 
277         for (Cell c : partialResult.rawCells()) {
278           aggregatePartialCells.add(c);
279         }
280       } while (partialResult.isPartial());
281 
282       assertTrue("Number of cells differs. iteration: " + iterationCount,
283           oneShotResult.rawCells().length == aggregatePartialCells.size());
284       final Cell[] oneShotCells = oneShotResult.rawCells();
285       for (int cell = 0; cell < oneShotCells.length; cell++) {
286         Cell oneShotCell = oneShotCells[cell];
287         Cell partialCell = aggregatePartialCells.get(cell);
288 
289         assertTrue("One shot cell was null", oneShotCell != null);
290         assertTrue("Partial cell was null", partialCell != null);
291         assertTrue("Cell differs. oneShotCell:" + oneShotCell + " partialCell:" + partialCell,
292             oneShotCell.equals(partialCell));
293       }
294 
295       oneShotResult = oneShotScanner.next();
296       iterationCount++;
297     }
298 
299     assertTrue(partialScanner.next() == null);
300 
301     partialScanner.close();
302     oneShotScanner.close();
303   }

325   public void testExpectedNumberOfCellsPerPartialResult(Scan baseScan, int expectedNumberOfCells)
326       throws Exception {
327 
328     if (LOG.isInfoEnabled()) LOG.info("groupSize:" + expectedNumberOfCells);
329 
330     // Use the cellHeapSize to set maxResultSize such that we know how many cells to expect back
331     // from the call. The returned results should NOT exceed expectedNumberOfCells but may be less
332     // than it in cases where expectedNumberOfCells is not an exact multiple of the number of
333     // columns in the table.
334     Scan scan = new Scan(baseScan);
335     scan.setAllowPartialResults(true);
336     scan.setMaxResultSize(getResultSizeForNumberOfCells(expectedNumberOfCells));
337 
338     ResultScanner scanner = TABLE.getScanner(scan);
339     Result result = null;
340     byte[] prevRow = null;
341     while ((result = scanner.next()) != null) {
342       assertTrue(result.rawCells() != null);
343 
344       // Cases when cell count won't equal expectedNumberOfCells:
345       // 1. Returned result is the final result needed to form the complete result for that row
346       // 2. It is the first result we have seen for that row and thus may have been fetched as
347       // the last group of cells that fit inside the maxResultSize
348       assertTrue(
349           "Result's cell count differed from expected number. result: " + result,
350           result.rawCells().length == expectedNumberOfCells || !result.isPartial()
351               || !Bytes.equals(prevRow, result.getRow()));
352       prevRow = result.getRow();
353     }
354 
355     scanner.close();
356   }

358   /**
359    * @return The approximate heap size of a cell in the test table. All cells should have
360    *         approximately the same heap size, so the value is cached to avoid repeating the
361    *         calculation
362    * @throws Exception
363    */
364   private long getCellHeapSize() throws Exception {
365     if (CELL_HEAP_SIZE == -1) {
366       // Do a partial scan that will return a single result with a single cell
367       Scan scan = new Scan();
368       scan.setMaxResultSize(1);
369       scan.setAllowPartialResults(true);
370       ResultScanner scanner = TABLE.getScanner(scan);
371 
372       Result result = scanner.next();
373 
374       assertTrue(result != null);
375       assertTrue(result.rawCells() != null);
376       assertTrue(result.rawCells().length == 1);
377 
378       CELL_HEAP_SIZE = CellUtil.estimatedHeapSizeOf(result.rawCells()[0]);
379       if (LOG.isInfoEnabled()) LOG.info("Cell heap size: " + CELL_HEAP_SIZE);
380       scanner.close();
381     }
382 
383     return CELL_HEAP_SIZE;
384   }

540    * maxResultSize limit is reached before the caching limit, it is likely that partial results will

546     for (int caching = 1; caching <= NUM_ROWS; caching++) {

548         testPartialResultsAndCaching(maxResultRows, caching);

553   /**
554    * @param resultSizeRowLimit The row limit that will be enforced through maxResultSize
555    * @param cachingRowLimit The row limit that will be enforced through caching
556    * @throws Exception
557    */
558   public void testPartialResultsAndCaching(int resultSizeRowLimit, int cachingRowLimit)
559       throws Exception {
560     Scan scan = new Scan();
561     scan.setAllowPartialResults(true);
562 
563     // The number of cells specified in the call to getResultSizeForNumberOfCells is offset to
564     // ensure that the result size we specify is not an exact multiple of the number of cells
565     // in a row. This ensures that partial results will be returned when the result size limit
566     // is reached before the caching limit.
567     int cellOffset = NUM_COLS / 3;
568     long maxResultSize = getResultSizeForNumberOfCells(resultSizeRowLimit * NUM_COLS + cellOffset);
569     scan.setMaxResultSize(maxResultSize);
570     scan.setCaching(cachingRowLimit);
571 
572     ResultScanner scanner = TABLE.getScanner(scan);
573     ClientScanner clientScanner = (ClientScanner) scanner;
574     Result r = null;
575 
576     // Approximate the number of rows we expect will fit into the specified max rsult size. If this
577     // approximation is less than caching, then we expect that the max result size limit will be
578     // hit before the caching limit and thus partial results may be seen
579     boolean expectToSeePartialResults = resultSizeRowLimit < cachingRowLimit;
580     while ((r = clientScanner.next()) != null) {
581       assertTrue(!r.isPartial() || expectToSeePartialResults);
582     }
583 
584     scanner.close();
585   }

618   /**
619    * Make puts to put the input value into each combination of row, family, and qualifier
620    * @param rows
621    * @param families
622    * @param qualifiers
623    * @param value
624    * @return
625    * @throws IOException
626    */
627   static ArrayList<Put> createPuts(byte[][] rows, byte[][] families, byte[][] qualifiers,
628       byte[] value) throws IOException {
629     Put put;
630     ArrayList<Put> puts = new ArrayList<>();
631 
632     for (int row = 0; row < rows.length; row++) {
633       put = new Put(rows[row]);
634       for (int fam = 0; fam < families.length; fam++) {
635         for (int qual = 0; qual < qualifiers.length; qual++) {
636           KeyValue kv = new KeyValue(rows[row], families[fam], qualifiers[qual], qual, value);
637           put.add(kv);
638         }
639       }
640       puts.add(put);
641     }
642 
643     return puts;
644   }

786   /**
787    * Exhausts the scanner by calling next repetitively. Once completely exhausted, close scanner and
788    * return total cell count
789    * @param scanner
790    * @return
791    * @throws Exception
792    */
793   private int countCellsFromScanner(ResultScanner scanner) throws Exception {
794     Result result = null;
795     int numCells = 0;
796     while ((result = scanner.next()) != null) {
797       numCells += result.rawCells().length;
798     }
799 
800     scanner.close();
801     return numCells;
802   }
