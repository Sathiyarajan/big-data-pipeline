217   private void concurrentExec(
218       final Runnable task, final int numThreads) throws Throwable {
219     startSignal = new CountDownLatch(numThreads);
220     doneSignal = new CountDownLatch(numThreads);
221     for (int i = 0; i < numThreads; ++i) {
222       new Thread(new Runnable() {
223         @Override
224         public void run() {
225           try {
226             startSignal.countDown();
227             startSignal.await();
228             task.run();
229           } catch (Throwable e) {
230             failures.incrementAndGet();
231             e.printStackTrace();
232           }
233           doneSignal.countDown();
234         }
235       }).start();
236     }
237     doneSignal.await();
238   }

349         List<Cell> kvs = new ArrayList<Cell>();
350         Scan scan = new Scan(row, row);
351         scan.addColumn(FAM, COUNTER);

354           Bytes.toInt(CellUtil.cloneValue(kvs.iterator().next()));

360         counter += 1;
361         expectedCounter += 1;
362

364         Put p = new Put(row);
365         KeyValue kv =
366             new KeyValue(row, FAM, COUNTER, now, Bytes.toBytes(counter));
367         p.add(kv);
368         mutations.add(p);
369         walEdit.add(kv);

371         // We can also inject some meta data to the walEdit
372         KeyValue metaKv = new KeyValue(
373             row, WALEdit.METAFAMILY,
374             Bytes.toBytes("I just increment counter"),
375             Bytes.toBytes(counter));

429       @Override
430       public void process(long now, HRegion region,
431           List<Mutation> mutations, WALEdit walEdit) throws IOException {
432         List<Cell> kvs = new ArrayList<Cell>();
433         { // First scan to get friends of the person
434           Scan scan = new Scan(row, row);
435           scan.addColumn(FAM, person);
436           doScan(region, scan, kvs);
437         }
438 
439         // Second scan to get friends of friends
440         Scan scan = new Scan(row, row);
441         for (Cell kv : kvs) {
442           byte[] friends = CellUtil.cloneValue(kv);
443           for (byte f : friends) {
444             scan.addColumn(FAM, new byte[]{f});
445           }
446         }
447         doScan(region, scan, kvs);
448 
449         // Collect result
450         result.clear();
451         for (Cell kv : kvs) {
452           for (byte b : CellUtil.cloneValue(kv)) {
453             result.add((char)b + "");
454           }
455         }
456       }

461             FriendsOfFriendsProcessorRequest.newBuilder();
462         builder.setPerson(ByteStringer.wrap(person));
463         builder.setRow(ByteStringer.wrap(row));
464         builder.addAllResult(result);
465         FriendsOfFriendsProcessorRequest f = builder.build();
466         return f;

479     public static class RowSwapProcessor extends
480         BaseRowProcessor<RowSwapProcessorRequest, RowSwapProcessorResponse> {
481       byte[] row1 = new byte[0];
482       byte[] row2 = new byte[0];
483 
484       /**
485        * Empty constructor for Writable
486        */
487       RowSwapProcessor() {
488       }
489 
490       RowSwapProcessor(byte[] row1, byte[] row2) {
491         this.row1 = row1;
492         this.row2 = row2;
493       }
494 
495       @Override
496       public Collection<byte[]> getRowsToLock() {
497         List<byte[]> rows = new ArrayList<byte[]>();
498         rows.add(row1);
499         rows.add(row2);
500         return rows;
501       }
502 
503       @Override
504       public boolean readOnly() {
505         return false;
506       }
507 
508       @Override
509       public RowSwapProcessorResponse getResult() {
510         return RowSwapProcessorResponse.getDefaultInstance();
511       }
512 
513       @Override
514       public void process(long now, HRegion region,
515           List<Mutation> mutations, WALEdit walEdit) throws IOException {
516 
517         // Override the time to avoid race-condition in the unit test caused by
518         // inacurate timer on some machines
519         now = myTimer.getAndIncrement();
520 
521         // Scan both rows
522         List<Cell> kvs1 = new ArrayList<Cell>();
523         List<Cell> kvs2 = new ArrayList<Cell>();
524         doScan(region, new Scan(row1, row1), kvs1);
525         doScan(region, new Scan(row2, row2), kvs2);
526 
527         // Assert swapped
528         if (swapped) {
529           assertEquals(rowSize, kvs2.size());
530           assertEquals(row2Size, kvs1.size());
531         } else {
532           assertEquals(rowSize, kvs1.size());
533           assertEquals(row2Size, kvs2.size());
534         }
535         swapped = !swapped;
536 
537         // Add and delete keyvalues
538         List<List<Cell>> kvs = new ArrayList<List<Cell>>();
539         kvs.add(kvs1);
540         kvs.add(kvs2);
541         byte[][] rows = new byte[][]{row1, row2};
542         for (int i = 0; i < kvs.size(); ++i) {
543           for (Cell kv : kvs.get(i)) {
544             // Delete from the current row and add to the other row
545             Delete d = new Delete(rows[i]);
546             KeyValue kvDelete =
547                 new KeyValue(rows[i], CellUtil.cloneFamily(kv), CellUtil.cloneQualifier(kv), 
548                     kv.getTimestamp(), KeyValue.Type.Delete);
549             d.addDeleteMarker(kvDelete);
550             Put p = new Put(rows[1 - i]);
551             KeyValue kvAdd =
552                 new KeyValue(rows[1 - i], CellUtil.cloneFamily(kv), CellUtil.cloneQualifier(kv),
553                     now, CellUtil.cloneValue(kv));
554             p.add(kvAdd);
555             mutations.add(d);
556             walEdit.add(kvDelete);
557             mutations.add(p);
558             walEdit.add(kvAdd);
559           }
560         }
561       }

641     public static void doScan(
642         HRegion region, Scan scan, List<Cell> result) throws IOException {
643       InternalScanner scanner = null;
644       try {
645         scan.setIsolationLevel(IsolationLevel.READ_UNCOMMITTED);
646         scanner = region.getScanner(scan);
647         result.clear();
648         scanner.next(result);
649       } finally {
650         if (scanner != null) scanner.close();
651       }
652     }

655   static String stringifyKvs(Collection<Cell> kvs) {
656     StringBuilder out = new StringBuilder();
657     out.append("[");
658     if (kvs != null) {
659       for (Cell kv : kvs) {
660         byte[] col = CellUtil.cloneQualifier(kv);
661         byte[] val = CellUtil.cloneValue(kv);
662         if (Bytes.equals(col, COUNTER)) {
663           out.append(Bytes.toStringBinary(col) + ":" +
664                      Bytes.toInt(val) + " ");
665         } else {
666           out.append(Bytes.toStringBinary(col) + ":" +
667                      Bytes.toStringBinary(val) + " ");
668         }
669       }
670     }
671     out.append("]");
672     return out.toString();
673   }
