89    @Test
90    public void testLogCleaning() throws Exception{
91      Configuration conf = TEST_UTIL.getConfiguration();
92      // set TTL
93      long ttl = 10000;
94      conf.setLong("hbase.master.logcleaner.ttl", ttl);
95      conf.setBoolean(HConstants.REPLICATION_ENABLE_KEY, HConstants.REPLICATION_ENABLE_DEFAULT);
96      Replication.decorateMasterConfiguration(conf);
97      Server server = new DummyServer();
98      ReplicationQueues repQueues =
99          ReplicationFactory.getReplicationQueues(server.getZooKeeper(), conf, server);
100     repQueues.init(server.getServerName().toString());
101     final Path oldLogDir = new Path(TEST_UTIL.getDataTestDir(),
102         HConstants.HREGION_OLDLOGDIR_NAME);
103     String fakeMachineName =
104       URLEncoder.encode(server.getServerName().toString(), "UTF8");
105 
106     final FileSystem fs = FileSystem.get(conf);
107 
108     // Create 2 invalid files, 1 "recent" file, 1 very new file and 30 old files
109     long now = System.currentTimeMillis();
110     fs.delete(oldLogDir, true);
111     fs.mkdirs(oldLogDir);
112     // Case 1: 2 invalid files, which would be deleted directly
113     fs.createNewFile(new Path(oldLogDir, "a"));
114     fs.createNewFile(new Path(oldLogDir, fakeMachineName + "." + "a"));
115     // Case 2: 1 "recent" file, not even deletable for the first log cleaner
116     // (TimeToLiveLogCleaner), so we are not going down the chain
117     System.out.println("Now is: " + now);
118     for (int i = 1; i < 31; i++) {
119       // Case 3: old files which would be deletable for the first log cleaner
120       // (TimeToLiveLogCleaner), and also for the second (ReplicationLogCleaner)
121       Path fileName = new Path(oldLogDir, fakeMachineName + "." + (now - i) );
122       fs.createNewFile(fileName);
123       // Case 4: put 3 old log files in ZK indicating that they are scheduled
124       // for replication so these files would pass the first log cleaner
125       // (TimeToLiveLogCleaner) but would be rejected by the second
126       // (ReplicationLogCleaner)
127       if (i % (30/3) == 1) {
128         repQueues.addLog(fakeMachineName, fileName.getName());
129         System.out.println("Replication log file: " + fileName);
130       }
131     }
132 
133     // sleep for sometime to get newer modifcation time
134     Thread.sleep(ttl);
135     fs.createNewFile(new Path(oldLogDir, fakeMachineName + "." + now));
136 
137     // Case 2: 1 newer file, not even deletable for the first log cleaner
138     // (TimeToLiveLogCleaner), so we are not going down the chain
139     fs.createNewFile(new Path(oldLogDir, fakeMachineName + "." + (now + 10000) ));
140 
141     for (FileStatus stat : fs.listStatus(oldLogDir)) {
142       System.out.println(stat.getPath().toString());
143     }
144 
145     assertEquals(34, fs.listStatus(oldLogDir).length);
146 
147     LogCleaner cleaner  = new LogCleaner(1000, server, conf, fs, oldLogDir);
148 
149     cleaner.chore();
150 
151     // We end up with the current log file, a newer one and the 3 old log
152     // files which are scheduled for replication
153     TEST_UTIL.waitFor(1000, new Waiter.Predicate<Exception>() {
154       @Override
155       public boolean evaluate() throws Exception {
156         return 5 == fs.listStatus(oldLogDir).length;
157       }
158     });
159 
160     for (FileStatus file : fs.listStatus(oldLogDir)) {
161       System.out.println("Kept log files: " + file.getPath().getName());
162     }
163   }

183   @Test(timeout=10000)
184   public void testZooKeeperAbortDuringGetListOfReplicators() throws Exception {
185     Configuration conf = TEST_UTIL.getConfiguration();
186 
187     ReplicationLogCleaner cleaner = new ReplicationLogCleaner();
188 
189     List<FileStatus> dummyFiles = Lists.newArrayList(
190         new FileStatus(100, false, 3, 100, System.currentTimeMillis(), new Path("log1")),
191         new FileStatus(100, false, 3, 100, System.currentTimeMillis(), new Path("log2"))
192     );
193 
194     FaultyZooKeeperWatcher faultyZK =
195         new FaultyZooKeeperWatcher(conf, "testZooKeeperAbort-faulty", null);
196     final AtomicBoolean getListOfReplicatorsFailed = new AtomicBoolean(false);
197 
198     try {
199       faultyZK.init();
200       ReplicationQueuesClient replicationQueuesClient = spy(ReplicationFactory.getReplicationQueuesClient(
201         faultyZK, conf, new ReplicationLogCleaner.WarnOnlyAbortable()));
202       doAnswer(new Answer<Object>() {
203         @Override
204         public Object answer(InvocationOnMock invocation) throws Throwable {
205           try {
206             return invocation.callRealMethod();
207           } catch (KeeperException.ConnectionLossException e) {
208             getListOfReplicatorsFailed.set(true);
209             throw e;
210           }
211         }
212       }).when(replicationQueuesClient).getListOfReplicators();
213       replicationQueuesClient.init();
214 
215       cleaner.setConf(conf, faultyZK, replicationQueuesClient);
216       // should keep all files due to a ConnectionLossException getting the queues znodes
217       Iterable<FileStatus> toDelete = cleaner.getDeletableFiles(dummyFiles);
218 
219       assertTrue(getListOfReplicatorsFailed.get());
220       assertFalse(toDelete.iterator().hasNext());
221       assertFalse(cleaner.isStopped());
222     } finally {
223       faultyZK.close();
224     }
225   }

243       cleaner.setConf(conf, zkw);
244       Iterable<FileStatus> filesToDelete = cleaner.getDeletableFiles(dummyFiles);
245       Iterator<FileStatus> iter = filesToDelete.iterator();
246       assertTrue(iter.hasNext());
247       assertEquals(new Path("log1"), iter.next().getPath());
248       assertTrue(iter.hasNext());
249       assertEquals(new Path("log2"), iter.next().getPath());
250       assertFalse(iter.hasNext());
251     } finally {
252       zkw.close();
