167   private List<byte[]> generateRows() {
168     // generate prefixes
169     List<byte[]> prefixes = new ArrayList<byte[]>();
170     prefixes.add(new byte[0]);
171     for (int i = 1; i < numberOfRowPrefixes; ++i) {
172       int prefixLength = averagePrefixLength;
173       prefixLength += randomizer.nextInt(2 * prefixLengthVariance + 1) -
174           prefixLengthVariance;
175       byte[] newPrefix = new byte[prefixLength];
176       randomizer.nextBytes(newPrefix);
177       byte[] newPrefixWithCommon = newPrefix;
178       prefixes.add(newPrefixWithCommon);
179     }
180 
181     // generate rest of the row
182     List<byte[]> rows = new ArrayList<byte[]>();
183     for (int i = 0; i < numberOfRows; ++i) {
184       int suffixLength = averageSuffixLength;
185       suffixLength += randomizer.nextInt(2 * suffixLengthVariance + 1) -
186           suffixLengthVariance;
187       int randomPrefix = randomizer.nextInt(prefixes.size());
188       byte[] row = new byte[prefixes.get(randomPrefix).length +
189                             suffixLength];
190       byte[] rowWithCommonPrefix = Bytes.concat(commonPrefix, row);
191       rows.add(rowWithCommonPrefix);
192     }
193 
194     return rows;
195   }

205   /**
206    * Generate test data useful to test encoders.
207    * @param howMany How many Key values should be generated.
208    * @return sorted list of key values
209    */
210   public List<KeyValue> generateTestKeyValues(int howMany, boolean useTags) {
211     List<KeyValue> result = new ArrayList<KeyValue>();
212 
213     List<byte[]> rows = generateRows();
214     Map<Integer, List<byte[]>> rowsToQualifier = new HashMap<Integer, List<byte[]>>();
215 
216     if(family==null){
217       family = new byte[columnFamilyLength];
218       randomizer.nextBytes(family);
219     }
220 
221     long baseTimestamp = Math.abs(randomizer.nextInt()) / baseTimestampDivide;
222 
223     byte[] value = new byte[valueLength];
224 
225     for (int i = 0; i < howMany; ++i) {
226       long timestamp = baseTimestamp;
227       if(timestampDiffSize > 0){
228         timestamp += randomizer.nextInt(timestampDiffSize);
229       }
230       Integer rowId = randomizer.nextInt(rows.size());
231       byte[] row = rows.get(rowId);
232 
233       // generate qualifier, sometimes it is same, sometimes similar,
234       // occasionally completely different
235       byte[] qualifier;
236       float qualifierChance = randomizer.nextFloat();
237       if (!rowsToQualifier.containsKey(rowId)
238           || qualifierChance > chanceForSameQualifier + chanceForSimilarQualifier) {
239         int qualifierLength = averageQualifierLength;
240         qualifierLength += randomizer.nextInt(2 * qualifierLengthVariance + 1)
241             - qualifierLengthVariance;
242         qualifier = new byte[qualifierLength];
243         randomizer.nextBytes(qualifier);
244 
245         // add it to map
246         if (!rowsToQualifier.containsKey(rowId)) {
247           rowsToQualifier.put(rowId, new ArrayList<byte[]>());
248         }
249         rowsToQualifier.get(rowId).add(qualifier);
250       } else if (qualifierChance > chanceForSameQualifier) {
251         // similar qualifier
252         List<byte[]> previousQualifiers = rowsToQualifier.get(rowId);
253         byte[] originalQualifier = previousQualifiers.get(randomizer.nextInt(previousQualifiers
254             .size()));
255 
256         qualifier = new byte[originalQualifier.length];
257         int commonPrefix = randomizer.nextInt(qualifier.length);
258         System.arraycopy(originalQualifier, 0, qualifier, 0, commonPrefix);
259         for (int j = commonPrefix; j < qualifier.length; ++j) {
260           qualifier[j] = (byte) (randomizer.nextInt() & 0xff);
261         }
262 
263         rowsToQualifier.get(rowId).add(qualifier);
264       } else {
265         // same qualifier
266         List<byte[]> previousQualifiers = rowsToQualifier.get(rowId);
267         qualifier = previousQualifiers.get(randomizer.nextInt(previousQualifiers.size()));
268       }
269 
270       if (randomizer.nextFloat() < chanceForZeroValue) {
271         for (int j = 0; j < value.length; ++j) {
272           value[j] = (byte) 0;
273         }
274       } else {
275         randomizer.nextBytes(value);
276       }
277 
278       if (useTags) {
279         result.add(new KeyValue(row, family, qualifier, timestamp, value, new Tag[] { new Tag(
280             (byte) 1, "value1") }));
281       } else {
282         result.add(new KeyValue(row, family, qualifier, timestamp, value));
283       }
284     }
285 
286     Collections.sort(result, KeyValue.COMPARATOR);
287 
288     return result;
289   }

291   /**
292    * Convert list of KeyValues to byte buffer.
293    * @param keyValues list of KeyValues to be converted.
294    * @return buffer with content from key values
295    */
296   public static ByteBuffer convertKvToByteBuffer(List<KeyValue> keyValues,
297       boolean includesMemstoreTS) {
298     int totalSize = 0;
299     for (KeyValue kv : keyValues) {
300       totalSize += kv.getLength();
301       if (includesMemstoreTS) {
302         totalSize += WritableUtils.getVIntSize(kv.getMvccVersion());
303       }
304     }
305 
306     ByteBuffer result = ByteBuffer.allocate(totalSize);
307     for (KeyValue kv : keyValues) {
308       result.put(kv.getBuffer(), kv.getOffset(), kv.getLength());
309       if (includesMemstoreTS) {
310         ByteBufferUtils.writeVLong(result, kv.getMvccVersion());
311       }
312     }
313     return result;
314   }
