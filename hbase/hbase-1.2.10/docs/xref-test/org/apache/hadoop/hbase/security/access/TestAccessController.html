263   private static void setUpTableAndUserPermissions() throws Exception {
264     HTableDescriptor htd = new HTableDescriptor(TEST_TABLE);
265     HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAMILY);
266     hcd.setMaxVersions(100);
267     htd.addFamily(hcd);
268     htd.setOwner(USER_OWNER);
269     createTable(TEST_UTIL, htd, new byte[][] { Bytes.toBytes("s") });
270 
271     Region region = TEST_UTIL.getHBaseCluster().getRegions(TEST_TABLE).get(0);
272     RegionCoprocessorHost rcpHost = region.getCoprocessorHost();
273     RCP_ENV = rcpHost.createEnvironment(AccessController.class, ACCESS_CONTROLLER,
274       Coprocessor.PRIORITY_HIGHEST, 1, conf);
275 
276     // Set up initial grants
277 
278     grantGlobal(TEST_UTIL, USER_ADMIN.getShortName(),
279       Permission.Action.ADMIN,
280       Permission.Action.CREATE,
281       Permission.Action.READ,
282       Permission.Action.WRITE);
283 
284     grantOnTable(TEST_UTIL, USER_RW.getShortName(),
285       TEST_TABLE, TEST_FAMILY, null,
286       Permission.Action.READ,
287       Permission.Action.WRITE);
288 
289     // USER_CREATE is USER_RW plus CREATE permissions
290     grantOnTable(TEST_UTIL, USER_CREATE.getShortName(),
291       TEST_TABLE, null, null,
292       Permission.Action.CREATE,
293       Permission.Action.READ,
294       Permission.Action.WRITE);
295 
296     grantOnTable(TEST_UTIL, USER_RO.getShortName(),
297       TEST_TABLE, TEST_FAMILY, null,
298       Permission.Action.READ);
299 
300     grantOnTable(TEST_UTIL, USER_ADMIN_CF.getShortName(),
301       TEST_TABLE, TEST_FAMILY,
302       null, Permission.Action.ADMIN, Permission.Action.CREATE);
303 
304     grantGlobal(TEST_UTIL, toGroupEntry(GROUP_ADMIN), Permission.Action.ADMIN);
305     grantGlobal(TEST_UTIL, toGroupEntry(GROUP_CREATE), Permission.Action.CREATE);
306     grantGlobal(TEST_UTIL, toGroupEntry(GROUP_READ), Permission.Action.READ);
307     grantGlobal(TEST_UTIL, toGroupEntry(GROUP_WRITE), Permission.Action.WRITE);
308 
309     assertEquals(5, AccessControlLists.getTablePermissions(conf, TEST_TABLE).size());
310     try {
311       assertEquals(5, AccessControlClient.getUserPermissions(systemUserConnection,
312           TEST_TABLE.toString()).size());
313     } catch (Throwable e) {
314       LOG.error("error during call of AccessControlClient.getUserPermissions. ", e);
315     }
316   }

332   }

1066   public class BulkLoadHelper {
1067     private final FileSystem fs;
1068     private final Path loadPath;
1069     private final Configuration conf;
1070 
1071     public BulkLoadHelper(Path loadPath) throws IOException {
1072       fs = TEST_UTIL.getTestFileSystem();
1073       conf = TEST_UTIL.getConfiguration();
1074       loadPath = loadPath.makeQualified(fs);
1075       this.loadPath = loadPath;
1076     }

1078     private void createHFile(Path path,
1079         byte[] family, byte[] qualifier,
1080         byte[] startKey, byte[] endKey, int numRows) throws IOException {
1081 
1082       HFile.Writer writer = null;
1083       long now = System.currentTimeMillis();
1084       try {
1085         HFileContext context = new HFileContextBuilder().build();
1086         writer = HFile.getWriterFactory(conf, new CacheConfig(conf))
1087             .withPath(fs, path)
1088             .withFileContext(context)
1089             .create();
1090         // subtract 2 since numRows doesn't include boundary keys
1091         for (byte[] key : Bytes.iterateOnSplits(startKey, endKey, true, numRows-2)) {
1092           KeyValue kv = new KeyValue(key, family, qualifier, now, key);
1093           writer.append(kv);
1094         }
1095       } finally {
1096         if(writer != null)
1097           writer.close();
1098       }
1099     }

1101     private void bulkLoadHFile(
1102         TableName tableName,
1103         byte[] family,
1104         byte[] qualifier,
1105         byte[][][] hfileRanges,
1106         int numRowsPerRange) throws Exception {
1107 
1108       Path familyDir = new Path(loadPath, Bytes.toString(family));
1109       fs.mkdirs(familyDir);
1110       int hfileIdx = 0;
1111       for (byte[][] range : hfileRanges) {
1112         byte[] from = range[0];
1113         byte[] to = range[1];
1114         createHFile(new Path(familyDir, "hfile_"+(hfileIdx++)),
1115             family, qualifier, from, to, numRowsPerRange);
1116       }
1117       //set global read so RegionServer can move it
1118       setPermission(loadPath, FsPermission.valueOf("-rwxrwxrwx"));
1119 
1120       try (Connection conn = ConnectionFactory.createConnection(conf);
1121            HTable table = (HTable)conn.getTable(tableName)) {
1122         TEST_UTIL.waitUntilAllRegionsAssigned(tableName);
1123         LoadIncrementalHFiles loader = new LoadIncrementalHFiles(conf);
1124         loader.doBulkLoad(loadPath, table);
1125       }
1126     }
