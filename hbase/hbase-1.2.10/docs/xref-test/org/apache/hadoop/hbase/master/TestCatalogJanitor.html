517   @Test
518   public void testCleanParent() throws IOException, InterruptedException {
519     HBaseTestingUtility htu = new HBaseTestingUtility();
520     setRootDirAndCleanIt(htu, "testCleanParent");
521     Server server = new MockServer(htu);
522     try {
523       MasterServices services = new MockMasterServices(server);
524       CatalogJanitor janitor = new CatalogJanitor(server, services);
525       // Create regions.
526       HTableDescriptor htd = new HTableDescriptor(TableName.valueOf("table"));
527       htd.addFamily(new HColumnDescriptor("f"));
528       HRegionInfo parent =
529         new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
530             Bytes.toBytes("eee"));
531       HRegionInfo splita =
532         new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
533             Bytes.toBytes("ccc"));
534       HRegionInfo splitb =
535         new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
536             Bytes.toBytes("eee"));
537       // Test that when both daughter regions are in place, that we do not
538       // remove the parent.
539       Result r = createResult(parent, splita, splitb);
540       // Add a reference under splitA directory so we don't clear out the parent.
541       Path rootdir = services.getMasterFileSystem().getRootDir();
542       Path tabledir =
543         FSUtils.getTableDir(rootdir, htd.getTableName());
544       Path storedir = HStore.getStoreHomedir(tabledir, splita,
545           htd.getColumnFamilies()[0].getName());
546       Reference ref = Reference.createTopReference(Bytes.toBytes("ccc"));
547       long now = System.currentTimeMillis();
548       // Reference name has this format: StoreFile#REF_NAME_PARSER
549       Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
550       FileSystem fs = services.getMasterFileSystem().getFileSystem();
551       Path path = ref.write(fs, p);
552       assertTrue(fs.exists(path));
553       assertFalse(janitor.cleanParent(parent, r));
554       // Remove the reference file and try again.
555       assertTrue(fs.delete(p, true));
556       assertTrue(janitor.cleanParent(parent, r));
557     } finally {
558       server.stop("shutdown");
559     }
560   }

586   /**
587    * Make sure parent with specified end key gets cleaned up even if daughter is cleaned up before it.
588    *
589    * @param rootDir the test case name, used as the HBase testing utility root
590    * @param lastEndKey the end key of the split parent
591    * @throws IOException
592    * @throws InterruptedException
593    */
594   private void parentWithSpecifiedEndKeyCleanedEvenIfDaughterGoneFirst(
595   final String rootDir, final byte[] lastEndKey)
596   throws IOException, InterruptedException {
597     HBaseTestingUtility htu = new HBaseTestingUtility();
598     setRootDirAndCleanIt(htu, rootDir);
599     Server server = new MockServer(htu);
600     MasterServices services = new MockMasterServices(server);
601     CatalogJanitor janitor = new CatalogJanitor(server, services);
602     final HTableDescriptor htd = createHTableDescriptor();
603 
604     // Create regions: aaa->{lastEndKey}, aaa->ccc, aaa->bbb, bbb->ccc, etc.
605 
606     // Parent
607     HRegionInfo parent = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
608       lastEndKey);
609     // Sleep a second else the encoded name on these regions comes out
610     // same for all with same start key and made in same second.
611     Thread.sleep(1001);
612 
613     // Daughter a
614     HRegionInfo splita = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
615       Bytes.toBytes("ccc"));
616     Thread.sleep(1001);
617     // Make daughters of daughter a; splitaa and splitab.
618     HRegionInfo splitaa = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
619       Bytes.toBytes("bbb"));
620     HRegionInfo splitab = new HRegionInfo(htd.getTableName(), Bytes.toBytes("bbb"),
621       Bytes.toBytes("ccc"));
622 
623     // Daughter b
624     HRegionInfo splitb = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
625       lastEndKey);
626     Thread.sleep(1001);
627     // Make Daughters of daughterb; splitba and splitbb.
628     HRegionInfo splitba = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
629       Bytes.toBytes("ddd"));
630     HRegionInfo splitbb = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ddd"),
631     lastEndKey);
632 
633     // First test that our Comparator works right up in CatalogJanitor.
634     // Just fo kicks.
635     SortedMap<HRegionInfo, Result> regions =
636       new TreeMap<HRegionInfo, Result>(new CatalogJanitor.SplitParentFirstComparator());
637     // Now make sure that this regions map sorts as we expect it to.
638     regions.put(parent, createResult(parent, splita, splitb));
639     regions.put(splitb, createResult(splitb, splitba, splitbb));
640     regions.put(splita, createResult(splita, splitaa, splitab));
641     // Assert its properly sorted.
642     int index = 0;
643     for (Map.Entry<HRegionInfo, Result> e: regions.entrySet()) {
644       if (index == 0) {
645         assertTrue(e.getKey().getEncodedName().equals(parent.getEncodedName()));
646       } else if (index == 1) {
647         assertTrue(e.getKey().getEncodedName().equals(splita.getEncodedName()));
648       } else if (index == 2) {
649         assertTrue(e.getKey().getEncodedName().equals(splitb.getEncodedName()));
650       }
651       index++;
652     }
653 
654     // Now play around with the cleanParent function.  Create a ref from splita
655     // up to the parent.
656     Path splitaRef =
657       createReferences(services, htd, parent, splita, Bytes.toBytes("ccc"), false);
658     // Make sure actual super parent sticks around because splita has a ref.
659     assertFalse(janitor.cleanParent(parent, regions.get(parent)));
660 
661     //splitba, and split bb, do not have dirs in fs.  That means that if
662     // we test splitb, it should get cleaned up.
663     assertTrue(janitor.cleanParent(splitb, regions.get(splitb)));
664 
665     // Now remove ref from splita to parent... so parent can be let go and so
666     // the daughter splita can be split (can't split if still references).
667     // BUT make the timing such that the daughter gets cleaned up before we
668     // can get a chance to let go of the parent.
669     FileSystem fs = FileSystem.get(htu.getConfiguration());
670     assertTrue(fs.delete(splitaRef, true));
671     // Create the refs from daughters of splita.
672     Path splitaaRef =
673       createReferences(services, htd, splita, splitaa, Bytes.toBytes("bbb"), false);
674     Path splitabRef =
675       createReferences(services, htd, splita, splitab, Bytes.toBytes("bbb"), true);
676 
677     // Test splita.  It should stick around because references from splitab, etc.
678     assertFalse(janitor.cleanParent(splita, regions.get(splita)));
679 
680     // Now clean up parent daughter first.  Remove references from its daughters.
681     assertTrue(fs.delete(splitaaRef, true));
682     assertTrue(fs.delete(splitabRef, true));
683     assertTrue(janitor.cleanParent(splita, regions.get(splita)));
684 
685     // Super parent should get cleaned up now both splita and splitb are gone.
686     assertTrue(janitor.cleanParent(parent, regions.get(parent)));
687 
688     services.stop("test finished");
689     janitor.cancel(true);
690   }

697   @Test

700     setRootDirAndCleanIt(htu, "testScanDoesNotCleanRegionsWithExistingParents");
701     Server server = new MockServer(htu);

1078     MultiResponse.Builder builder = MultiResponse.newBuilder();
1079     RegionActionResult.Builder regionActionResultBuilder =
1080         RegionActionResult.newBuilder();
1081     ResultOrException.Builder roeBuilder = ResultOrException.newBuilder();
