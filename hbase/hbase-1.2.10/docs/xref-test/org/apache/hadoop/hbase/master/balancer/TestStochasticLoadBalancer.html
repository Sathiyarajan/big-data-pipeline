62    @Test
63    public void testKeepRegionLoad() throws Exception {
64  
65      ServerName sn = ServerName.valueOf("test:8080", 100);
66      int numClusterStatusToAdd = 20000;
67      for (int i = 0; i < numClusterStatusToAdd; i++) {
68        ServerLoad sl = mock(ServerLoad.class);
69  
70        RegionLoad rl = mock(RegionLoad.class);
71        when(rl.getStores()).thenReturn(i);
72  
73        Map<byte[], RegionLoad> regionLoadMap =
74            new TreeMap<byte[], RegionLoad>(Bytes.BYTES_COMPARATOR);
75        regionLoadMap.put(Bytes.toBytes(REGION_KEY), rl);
76        when(sl.getRegionsLoad()).thenReturn(regionLoadMap);
77  
78        ClusterStatus clusterStatus = mock(ClusterStatus.class);
79        when(clusterStatus.getServers()).thenReturn(Arrays.asList(sn));
80        when(clusterStatus.getLoad(sn)).thenReturn(sl);
81  
82        loadBalancer.setClusterStatus(clusterStatus);
83      }
84      assertTrue(loadBalancer.loads.get(REGION_KEY) != null);
85      assertTrue(loadBalancer.loads.get(REGION_KEY).size() == 15);
86  
87      Queue<RegionLoad> loads = loadBalancer.loads.get(REGION_KEY);
88      int i = 0;
89      while(loads.size() > 0) {
90        RegionLoad rl = loads.remove();
91        assertEquals(i + (numClusterStatusToAdd - 15), rl.getStores());
92        i ++;
93      }
94    }

151   @Test
152   public void testCostAfterUndoAction() {
153     final int runs = 10;
154     loadBalancer.setConf(conf);
155     for (int[] mockCluster : clusterStateMocks) {
156       BaseLoadBalancer.Cluster cluster = mockCluster(mockCluster);
157       loadBalancer.initCosts(cluster);
158       for (int i = 0; i != runs; ++i) {
159         final double expectedCost = loadBalancer.computeCost(cluster, Double.MAX_VALUE);
160         Cluster.Action action = loadBalancer.nextAction(cluster);
161         cluster.doAction(action);
162         loadBalancer.updateCostsWithAction(cluster, action);
163         Cluster.Action undoAction = action.undoAction();
164         cluster.doAction(undoAction);
165         loadBalancer.updateCostsWithAction(cluster, undoAction);
166         final double actualCost = loadBalancer.computeCost(cluster, Double.MAX_VALUE);
167         assertEquals(expectedCost, actualCost, 0);
168       }
169     }

260   public void testReplicaCostForReplicas() {
261     Configuration conf = HBaseConfiguration.create();
262     StochasticLoadBalancer.CostFunction
263         costFunction = new StochasticLoadBalancer.RegionReplicaHostCostFunction(conf);
264 
265     int [] servers = new int[] {3,3,3,3,3};
266     TreeMap<ServerName, List<HRegionInfo>> clusterState = mockClusterServers(servers);
267 
268     BaseLoadBalancer.Cluster cluster;
269 
270     cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
271     costFunction.init(cluster);
272     double costWithoutReplicas = costFunction.cost();
273     assertEquals(0, costWithoutReplicas, 0);
274 
275     // replicate the region from first server to the last server
276     HRegionInfo replica1 = RegionReplicaUtil.getRegionInfoForReplica(
277       clusterState.firstEntry().getValue().get(0),1);
278     clusterState.lastEntry().getValue().add(replica1);
279 
280     cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
281     costFunction.init(cluster);
282     double costWith1ReplicaDifferentServer = costFunction.cost();
283 
284     assertEquals(0, costWith1ReplicaDifferentServer, 0);
285 
286     // add a third replica to the last server
287     HRegionInfo replica2 = RegionReplicaUtil.getRegionInfoForReplica(replica1, 2);
288     clusterState.lastEntry().getValue().add(replica2);
289 
290     cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
291     costFunction.init(cluster);
292     double costWith1ReplicaSameServer = costFunction.cost();
293 
294     assertTrue(costWith1ReplicaDifferentServer < costWith1ReplicaSameServer);
295 
296     // test with replication = 4 for following:
297 
298     HRegionInfo replica3;
299     Iterator<Entry<ServerName, List<HRegionInfo>>> it;
300     Entry<ServerName, List<HRegionInfo>> entry;
301 
302     clusterState = mockClusterServers(servers);
303     it = clusterState.entrySet().iterator();
304     entry = it.next(); //first server
305     HRegionInfo hri = entry.getValue().get(0);
306     replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
307     replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
308     replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
309     entry.getValue().add(replica1);
310     entry.getValue().add(replica2);
311     it.next().getValue().add(replica3); //2nd server
312 
313     cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
314     costFunction.init(cluster);
315     double costWith3ReplicasSameServer = costFunction.cost();
316 
317     clusterState = mockClusterServers(servers);
318     hri = clusterState.firstEntry().getValue().get(0);
319     replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
320     replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
321     replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
322 
323     clusterState.firstEntry().getValue().add(replica1);
324     clusterState.lastEntry().getValue().add(replica2);
325     clusterState.lastEntry().getValue().add(replica3);
326 
327     cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
328     costFunction.init(cluster);
329     double costWith2ReplicasOnTwoServers = costFunction.cost();
330 
331     assertTrue(costWith2ReplicasOnTwoServers < costWith3ReplicasSameServer);
332   }

334   @Test
335   public void testNeedsBalanceForColocatedReplicas() {
336     // check for the case where there are two hosts and with one rack, and where
337     // both the replicas are hosted on the same server
338     List<HRegionInfo> regions = randomRegions(1);
339     ServerName s1 = ServerName.valueOf("host1", 1000, 11111);
340     ServerName s2 = ServerName.valueOf("host11", 1000, 11111);
341     Map<ServerName, List<HRegionInfo>> map = new HashMap<ServerName, List<HRegionInfo>>();
342     map.put(s1, regions);
343     regions.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
344     // until the step above s1 holds two replicas of a region
345     regions = randomRegions(1);
346     map.put(s2, regions);
347     assertTrue(loadBalancer.needsBalance(new Cluster(map, null, null, null)));
348     // check for the case where there are two hosts on the same rack and there are two racks
349     // and both the replicas are on the same rack
350     map.clear();
351     regions = randomRegions(1);
352     List<HRegionInfo> regionsOnS2 = new ArrayList<HRegionInfo>(1);
353     regionsOnS2.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
354     map.put(s1, regions);
355     map.put(s2, regionsOnS2);
356     // add another server so that the cluster has some host on another rack
357     map.put(ServerName.valueOf("host2", 1000, 11111), randomRegions(1));
358     assertTrue(loadBalancer.needsBalance(new Cluster(map, null, null,
359         new ForTestRackManagerOne())));
360   }
