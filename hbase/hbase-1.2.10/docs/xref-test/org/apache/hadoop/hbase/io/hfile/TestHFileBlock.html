117   static int writeTestKeyValues(HFileBlock.Writer hbw, int seed, boolean includesMemstoreTS,
118       boolean useTag) throws IOException {
119     List<KeyValue> keyValues = new ArrayList<KeyValue>();
120     Random randomizer = new Random(42l + seed); // just any fixed number
121 
122     // generate keyValues
123     for (int i = 0; i < NUM_KEYVALUES; ++i) {
124       byte[] row;
125       long timestamp;
126       byte[] family;
127       byte[] qualifier;
128       byte[] value;
129 
130       // generate it or repeat, it should compress well
131       if (0 < i && randomizer.nextFloat() < CHANCE_TO_REPEAT) {
132         row = keyValues.get(randomizer.nextInt(keyValues.size())).getRow();
133       } else {
134         row = new byte[FIELD_LENGTH];
135         randomizer.nextBytes(row);
136       }
137       if (0 == i) {
138         family = new byte[FIELD_LENGTH];
139         randomizer.nextBytes(family);
140       } else {
141         family = keyValues.get(0).getFamily();
142       }
143       if (0 < i && randomizer.nextFloat() < CHANCE_TO_REPEAT) {
144         qualifier = keyValues.get(
145             randomizer.nextInt(keyValues.size())).getQualifier();
146       } else {
147         qualifier = new byte[FIELD_LENGTH];
148         randomizer.nextBytes(qualifier);
149       }
150       if (0 < i && randomizer.nextFloat() < CHANCE_TO_REPEAT) {
151         value = keyValues.get(randomizer.nextInt(keyValues.size())).getValue();
152       } else {
153         value = new byte[FIELD_LENGTH];
154         randomizer.nextBytes(value);
155       }
156       if (0 < i && randomizer.nextFloat() < CHANCE_TO_REPEAT) {
157         timestamp = keyValues.get(
158             randomizer.nextInt(keyValues.size())).getTimestamp();
159       } else {
160         timestamp = randomizer.nextLong();
161       }
162       if (!useTag) {
163         keyValues.add(new KeyValue(row, family, qualifier, timestamp, value));
164       } else {
165         keyValues.add(new KeyValue(row, family, qualifier, timestamp, value, new Tag[] { new Tag(
166             (byte) 1, Bytes.toBytes("myTagVal")) }));
167       }
168     }
169 
170     // sort it and write to stream
171     int totalSize = 0;
172     Collections.sort(keyValues, KeyValue.COMPARATOR);
173 
174     for (KeyValue kv : keyValues) {
175       totalSize += kv.getLength();
176       if (includesMemstoreTS) {
177         long memstoreTS = randomizer.nextLong();
178         kv.setSequenceId(memstoreTS);
179         totalSize += WritableUtils.getVIntSize(memstoreTS);
180       }
181       hbw.write(kv);
182     }
183     return totalSize;
184   }

186   public byte[] createTestV1Block(Compression.Algorithm algo)
187       throws IOException {
188     Compressor compressor = algo.getCompressor();
189     ByteArrayOutputStream baos = new ByteArrayOutputStream();
190     OutputStream os = algo.createCompressionStream(baos, compressor, 0);
191     DataOutputStream dos = new DataOutputStream(os);
192     BlockType.META.write(dos); // Let's make this a meta block.
193     writeTestBlockContents(dos);
194     dos.flush();
195     algo.returnCompressor(compressor);
196     return baos.toByteArray();
197   }

199   static HFileBlock.Writer createTestV2Block(Compression.Algorithm algo,
200       boolean includesMemstoreTS, boolean includesTag) throws IOException {
201     final BlockType blockType = BlockType.DATA;
202     HFileContext meta = new HFileContextBuilder()
203                         .withCompression(algo)
204                         .withIncludesMvcc(includesMemstoreTS)
205                         .withIncludesTags(includesTag)
206                         .withBytesPerCheckSum(HFile.DEFAULT_BYTES_PER_CHECKSUM)
207                         .build();
208     HFileBlock.Writer hbw = new HFileBlock.Writer(null, meta);
209     DataOutputStream dos = hbw.startWriting(blockType);
210     writeTestBlockContents(dos);
211     dos.flush();
212     hbw.ensureBlockReady();
213     assertEquals(1000 * 4, hbw.getUncompressedSizeWithoutHeader());
214     hbw.release();
215     return hbw;
216   }

218   public String createTestBlockStr(Compression.Algorithm algo,
219       int correctLength, boolean useTag) throws IOException {
220     HFileBlock.Writer hbw = createTestV2Block(algo, includesMemstoreTS, useTag);
221     byte[] testV2Block = hbw.getHeaderAndDataForTest();
222     int osOffset = HConstants.HFILEBLOCK_HEADER_SIZE + 9;
223     if (testV2Block.length == correctLength) {
224       // Force-set the "OS" field of the gzip header to 3 (Unix) to avoid
225       // variations across operating systems.
226       // See http://www.gzip.org/zlib/rfc-gzip.html for gzip format.
227       // We only make this change when the compressed block length matches.
228       // Otherwise, there are obviously other inconsistencies.
229       testV2Block[osOffset] = 3;
230     }
231     return Bytes.toStringBinary(testV2Block);
232   }

246   @Test
247   public void testGzipCompression() throws IOException {
248     final String correctTestBlockStr =
249         "DATABLK*\\x00\\x00\\x00>\\x00\\x00\\x0F\\xA0\\xFF\\xFF\\xFF\\xFF"
250             + "\\xFF\\xFF\\xFF\\xFF"
251             + "\\x0" + ChecksumType.getDefaultChecksumType().getCode()
252             + "\\x00\\x00@\\x00\\x00\\x00\\x00["
253             // gzip-compressed block: http://www.gzip.org/zlib/rfc-gzip.html
254             + "\\x1F\\x8B"  // gzip magic signature
255             + "\\x08"  // Compression method: 8 = "deflate"
256             + "\\x00"  // Flags
257             + "\\x00\\x00\\x00\\x00"  // mtime
258             + "\\x00"  // XFL (extra flags)
259             // OS (0 = FAT filesystems, 3 = Unix). However, this field
260             // sometimes gets set to 0 on Linux and Mac, so we reset it to 3.
261             // This appears to be a difference caused by the availability
262             // (and use) of the native GZ codec.
263             + "\\x03"
264             + "\\xED\\xC3\\xC1\\x11\\x00 \\x08\\xC00DD\\xDD\\x7Fa"
265             + "\\xD6\\xE8\\xA3\\xB9K\\x84`\\x96Q\\xD3\\xA8\\xDB\\xA8e\\xD4c"
266             + "\\xD46\\xEA5\\xEA3\\xEA7\\xE7\\x00LI\\x5Cs\\xA0\\x0F\\x00\\x00"
267             + "\\x00\\x00\\x00\\x00"; //  4 byte checksum (ignored)
268     final int correctGzipBlockLength = 95;
269     final String testBlockStr = createTestBlockStr(GZ, correctGzipBlockLength, false);
270     // We ignore the block checksum because createTestBlockStr can change the
271     // gzip header after the block is produced
272     assertEquals(correctTestBlockStr.substring(0, correctGzipBlockLength - 4),
273       testBlockStr.substring(0, correctGzipBlockLength - 4));
274   }

281   protected void testReaderV2Internals() throws IOException {
282     if(includesTag) {
283       TEST_UTIL.getConfiguration().setInt("hfile.format.version", 3);
284     }
285     for (Compression.Algorithm algo : COMPRESSION_ALGORITHMS) {
286       for (boolean pread : new boolean[] { false, true }) {
287           LOG.info("testReaderV2: Compression algorithm: " + algo +
288                    ", pread=" + pread);
289         Path path = new Path(TEST_UTIL.getDataTestDir(), "blocks_v2_"
290             + algo);
291         FSDataOutputStream os = fs.create(path);
292         HFileContext meta = new HFileContextBuilder()
293                            .withCompression(algo)
294                            .withIncludesMvcc(includesMemstoreTS)
295                            .withIncludesTags(includesTag)
296                            .withBytesPerCheckSum(HFile.DEFAULT_BYTES_PER_CHECKSUM)
297                            .build();
298         HFileBlock.Writer hbw = new HFileBlock.Writer(null,
299            meta);
300         long totalSize = 0;
301         for (int blockId = 0; blockId < 2; ++blockId) {
302           DataOutputStream dos = hbw.startWriting(BlockType.DATA);
303           for (int i = 0; i < 1234; ++i)
304             dos.writeInt(i);
305           hbw.writeHeaderAndData(os);
306           totalSize += hbw.getOnDiskSizeWithHeader();
307         }
308         os.close();
309 
310         FSDataInputStream is = fs.open(path);
311         meta = new HFileContextBuilder()
312         .withHBaseCheckSum(true)
313         .withIncludesMvcc(includesMemstoreTS)
314         .withIncludesTags(includesTag)
315         .withCompression(algo).build();
316         HFileBlock.FSReader hbr = new HFileBlock.FSReaderImpl(is, totalSize, meta);
317         HFileBlock b = hbr.readBlockData(0, -1, -1, pread);
318         is.close();
319         assertEquals(0, HFile.getChecksumFailuresCount());
320 
321         b.sanityCheck();
322         assertEquals(4936, b.getUncompressedSizeWithoutHeader());
323         assertEquals(algo == GZ ? 2173 : 4936,
324                      b.getOnDiskSizeWithoutHeader() - b.totalChecksumBytes());
325         HFileBlock expected = b;
326 
327         if (algo == GZ) {
328           is = fs.open(path);
329           hbr = new HFileBlock.FSReaderImpl(is, totalSize, meta);
330           b = hbr.readBlockData(0, 2173 + HConstants.HFILEBLOCK_HEADER_SIZE +
331                                 b.totalChecksumBytes(), -1, pread);
332           assertEquals(expected, b);
333           int wrongCompressedSize = 2172;
334           try {
335             b = hbr.readBlockData(0, wrongCompressedSize
336                 + HConstants.HFILEBLOCK_HEADER_SIZE, -1, pread);
337             fail("Exception expected");
338           } catch (IOException ex) {
339             String expectedPrefix = "On-disk size without header provided is "
340                 + wrongCompressedSize + ", but block header contains "
341                 + b.getOnDiskSizeWithoutHeader() + ".";
342             assertTrue("Invalid exception message: '" + ex.getMessage()
343                 + "'.\nMessage is expected to start with: '" + expectedPrefix
344                 + "'", ex.getMessage().startsWith(expectedPrefix));
345           }
346           is.close();
347         }
348       }
349     }

361   private void testInternals() throws IOException {
362     final int numBlocks = 5;
363     if(includesTag) {
364       TEST_UTIL.getConfiguration().setInt("hfile.format.version", 3);
365     }
366     for (Compression.Algorithm algo : COMPRESSION_ALGORITHMS) {
367       for (boolean pread : new boolean[] { false, true }) {
368         for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
369           LOG.info("testDataBlockEncoding: Compression algorithm=" + algo + ", pread=" + pread +
370             ", dataBlockEncoder=" + encoding);
371           Path path = new Path(TEST_UTIL.getDataTestDir(), "blocks_v2_"
372               + algo + "_" + encoding.toString());
373           FSDataOutputStream os = fs.create(path);
374           HFileDataBlockEncoder dataBlockEncoder = (encoding != DataBlockEncoding.NONE) ?
375               new HFileDataBlockEncoderImpl(encoding) : NoOpDataBlockEncoder.INSTANCE;
376           HFileContext meta = new HFileContextBuilder()
377                               .withCompression(algo)
378                               .withIncludesMvcc(includesMemstoreTS)
379                               .withIncludesTags(includesTag)
380                               .withBytesPerCheckSum(HFile.DEFAULT_BYTES_PER_CHECKSUM)
381                               .build();
382           HFileBlock.Writer hbw = new HFileBlock.Writer(dataBlockEncoder, meta);
383           long totalSize = 0;
384           final List<Integer> encodedSizes = new ArrayList<Integer>();
385           final List<ByteBuffer> encodedBlocks = new ArrayList<ByteBuffer>();
386           for (int blockId = 0; blockId < numBlocks; ++blockId) {
387             hbw.startWriting(BlockType.DATA);
388             writeTestKeyValues(hbw, blockId, includesMemstoreTS, includesTag);
389             hbw.writeHeaderAndData(os);
390             int headerLen = HConstants.HFILEBLOCK_HEADER_SIZE;
391             byte[] encodedResultWithHeader = hbw.getUncompressedBufferWithHeader().array();
392             final int encodedSize = encodedResultWithHeader.length - headerLen;
393             if (encoding != DataBlockEncoding.NONE) {
394               // We need to account for the two-byte encoding algorithm ID that
395               // comes after the 24-byte block header but before encoded KVs.
396               headerLen += DataBlockEncoding.ID_SIZE;
397             }
398             byte[] encodedDataSection =
399                 new byte[encodedResultWithHeader.length - headerLen];
400             System.arraycopy(encodedResultWithHeader, headerLen,
401                 encodedDataSection, 0, encodedDataSection.length);
402             final ByteBuffer encodedBuf =
403                 ByteBuffer.wrap(encodedDataSection);
404             encodedSizes.add(encodedSize);
405             encodedBlocks.add(encodedBuf);
406             totalSize += hbw.getOnDiskSizeWithHeader();
407           }
408           os.close();
409 
410           FSDataInputStream is = fs.open(path);
411           meta = new HFileContextBuilder()
412                 .withHBaseCheckSum(true)
413                 .withCompression(algo)
414                 .withIncludesMvcc(includesMemstoreTS)
415                 .withIncludesTags(includesTag)
416                 .build();
417           HFileBlock.FSReaderImpl hbr = new HFileBlock.FSReaderImpl(is, totalSize, meta);
418           hbr.setDataBlockEncoder(dataBlockEncoder);
419           hbr.setIncludesMemstoreTS(includesMemstoreTS);
420           HFileBlock blockFromHFile, blockUnpacked;
421           int pos = 0;
422           for (int blockId = 0; blockId < numBlocks; ++blockId) {
423             blockFromHFile = hbr.readBlockData(pos, -1, -1, pread);
424             assertEquals(0, HFile.getChecksumFailuresCount());
425             blockFromHFile.sanityCheck();
426             pos += blockFromHFile.getOnDiskSizeWithHeader();
427             assertEquals((int) encodedSizes.get(blockId),
428               blockFromHFile.getUncompressedSizeWithoutHeader());
429             assertEquals(meta.isCompressedOrEncrypted(), !blockFromHFile.isUnpacked());
430             long packedHeapsize = blockFromHFile.heapSize();
431             blockUnpacked = blockFromHFile.unpack(meta, hbr);
432             assertTrue(blockUnpacked.isUnpacked());
433             if (meta.isCompressedOrEncrypted()) {
434               LOG.info("packedHeapsize=" + packedHeapsize + ", unpackedHeadsize=" + blockUnpacked
435                 .heapSize());
436               assertFalse(packedHeapsize == blockUnpacked.heapSize());
437               assertTrue("Packed heapSize should be < unpacked heapSize",
438                 packedHeapsize < blockUnpacked.heapSize());
439             }
440             ByteBuffer actualBuffer = blockUnpacked.getBufferWithoutHeader();
441             if (encoding != DataBlockEncoding.NONE) {
442               // We expect a two-byte big-endian encoding id.
443               assertEquals(
444                 "Unexpected first byte with " + buildMessageDetails(algo, encoding, pread),
445                 Long.toHexString(0), Long.toHexString(actualBuffer.get(0)));
446               assertEquals(
447                 "Unexpected second byte with " + buildMessageDetails(algo, encoding, pread),
448                 Long.toHexString(encoding.getId()), Long.toHexString(actualBuffer.get(1)));
449               actualBuffer.position(2);
450               actualBuffer = actualBuffer.slice();
451             }
452 
453             ByteBuffer expectedBuffer = encodedBlocks.get(blockId);
454             expectedBuffer.rewind();
455 
456             // test if content matches, produce nice message
457             assertBuffersEqual(expectedBuffer, actualBuffer, algo, encoding, pread);
458 
459             // test serialized blocks
460             for (boolean reuseBuffer : new boolean[] { false, true }) {
461               ByteBuffer serialized = ByteBuffer.allocate(blockFromHFile.getSerializedLength());
462               blockFromHFile.serialize(serialized);
463               HFileBlock deserialized =
464                 (HFileBlock) blockFromHFile.getDeserializer().deserialize(serialized, reuseBuffer);
465               assertEquals(
466                 "Serialization did not preserve block state. reuseBuffer=" + reuseBuffer,
467                 blockFromHFile, deserialized);
468               // intentional reference comparison
469               if (blockFromHFile != blockUnpacked) {
470                 assertEquals("Deserializaed block cannot be unpacked correctly.",
471                   blockUnpacked, deserialized.unpack(meta, hbr));
472               }
473             }
474           }
475           is.close();
476         }
477       }
478     }
479   }

481   static String buildMessageDetails(Algorithm compression, DataBlockEncoding encoding,
482       boolean pread) {
483     return String.format("compression %s, encoding %s, pread %s", compression, encoding, pread);

486   static void assertBuffersEqual(ByteBuffer expectedBuffer,
487       ByteBuffer actualBuffer, Compression.Algorithm compression,
488       DataBlockEncoding encoding, boolean pread) {
489     if (!actualBuffer.equals(expectedBuffer)) {
490       int prefix = 0;
491       int minLimit = Math.min(expectedBuffer.limit(), actualBuffer.limit());
492       while (prefix < minLimit &&
493           expectedBuffer.get(prefix) == actualBuffer.get(prefix)) {
494         prefix++;
495       }
496 
497       fail(String.format(
498           "Content mismatch for %s, commonPrefix %d, expected %s, got %s",
499           buildMessageDetails(compression, encoding, pread), prefix,
500           nextBytesToStr(expectedBuffer, prefix),
501           nextBytesToStr(actualBuffer, prefix)));
502     }
503   }

505   /**
506    * Convert a few next bytes in the given buffer at the given position to
507    * string. Used for error messages.
508    */
509   private static String nextBytesToStr(ByteBuffer buf, int pos) {
510     int maxBytes = buf.limit() - pos;
511     int numBytes = Math.min(16, maxBytes);
512     return Bytes.toStringBinary(buf.array(), buf.arrayOffset() + pos,
513         numBytes) + (numBytes < maxBytes ? "..." : "");
514   }

521   protected void testPreviousOffsetInternals() throws IOException {
522     // TODO: parameterize these nested loops.
523     for (Compression.Algorithm algo : COMPRESSION_ALGORITHMS) {
524       for (boolean pread : BOOLEAN_VALUES) {
525         for (boolean cacheOnWrite : BOOLEAN_VALUES) {
526           Random rand = defaultRandom();
527           LOG.info("testPreviousOffset: Compression algorithm=" + algo + ", pread=" + pread +
528             ", cacheOnWrite=" + cacheOnWrite);
529           Path path = new Path(TEST_UTIL.getDataTestDir(), "prev_offset");
530           List<Long> expectedOffsets = new ArrayList<Long>();
531           List<Long> expectedPrevOffsets = new ArrayList<Long>();
532           List<BlockType> expectedTypes = new ArrayList<BlockType>();
533           List<ByteBuffer> expectedContents = cacheOnWrite
534               ? new ArrayList<ByteBuffer>() : null;
535           long totalSize = writeBlocks(rand, algo, path, expectedOffsets,
536               expectedPrevOffsets, expectedTypes, expectedContents);
537 
538           FSDataInputStream is = fs.open(path);
539           HFileContext meta = new HFileContextBuilder()
540                               .withHBaseCheckSum(true)
541                               .withIncludesMvcc(includesMemstoreTS)
542                               .withIncludesTags(includesTag)
543                               .withCompression(algo).build();
544           HFileBlock.FSReader hbr = new HFileBlock.FSReaderImpl(is, totalSize, meta);
545           long curOffset = 0;
546           for (int i = 0; i < NUM_TEST_BLOCKS; ++i) {
547             if (!pread) {
548               assertEquals(is.getPos(), curOffset + (i == 0 ? 0 :
549                   HConstants.HFILEBLOCK_HEADER_SIZE));
550             }
551 
552             assertEquals(expectedOffsets.get(i).longValue(), curOffset);
553             if (detailedLogging) {
554               LOG.info("Reading block #" + i + " at offset " + curOffset);
555             }
556             HFileBlock b = hbr.readBlockData(curOffset, -1, -1, pread);
557             if (detailedLogging) {
558               LOG.info("Block #" + i + ": " + b);
559             }
560             assertEquals("Invalid block #" + i + "'s type:",
561                 expectedTypes.get(i), b.getBlockType());
562             assertEquals("Invalid previous block offset for block " + i
563                 + " of " + "type " + b.getBlockType() + ":",
564                 (long) expectedPrevOffsets.get(i), b.getPrevBlockOffset());
565             b.sanityCheck();
566             assertEquals(curOffset, b.getOffset());
567 
568             // Now re-load this block knowing the on-disk size. This tests a
569             // different branch in the loader.
570             HFileBlock b2 = hbr.readBlockData(curOffset,
571                 b.getOnDiskSizeWithHeader(), -1, pread);
572             b2.sanityCheck();
573 
574             assertEquals(b.getBlockType(), b2.getBlockType());
575             assertEquals(b.getOnDiskSizeWithoutHeader(),
576                 b2.getOnDiskSizeWithoutHeader());
577             assertEquals(b.getOnDiskSizeWithHeader(),
578                 b2.getOnDiskSizeWithHeader());
579             assertEquals(b.getUncompressedSizeWithoutHeader(),
580                 b2.getUncompressedSizeWithoutHeader());
581             assertEquals(b.getPrevBlockOffset(), b2.getPrevBlockOffset());
582             assertEquals(curOffset, b2.getOffset());
583             assertEquals(b.getBytesPerChecksum(), b2.getBytesPerChecksum());
584             assertEquals(b.getOnDiskDataSizeWithHeader(),

594               b = b.unpack(meta, hbr);

600                   bufRead.arrayOffset(),

603                   bufExpected.limit()) == 0;
604               String wrongBytesMsg = "";

606               if (!bytesAreCorrect) {
607                 // Optimization: only construct an error message in case we
608                 // will need it.
609                 wrongBytesMsg = "Expected bytes in block #" + i + " (algo="
610                     + algo + ", pread=" + pread
611                     + ", cacheOnWrite=" + cacheOnWrite + "):\n";
612                 wrongBytesMsg += Bytes.toStringBinary(bufExpected.array(),
613                   bufExpected.arrayOffset(), Math.min(32 + 10, bufExpected.limit()))
614                     + ", actual:\n"
615                     + Bytes.toStringBinary(bufRead.array(),
616                   bufRead.arrayOffset(), Math.min(32 + 10, bufRead.limit()));

623                            " limit " + bufRead.limit() +
624                            " expected offset " + bufExpected.arrayOffset() +
625                            " limit " + bufExpected.limit());
626                   LOG.warn(wrongBytesMsg);
627                 }
628               }
629               assertTrue(wrongBytesMsg, bytesAreCorrect);
630             }

634           is.close();

644   private class BlockReaderThread implements Callable<Boolean> {
645     private final String clientId;
646     private final HFileBlock.FSReader hbr;
647     private final List<Long> offsets;
648     private final List<BlockType> types;
649     private final long fileSize;
650 
651     public BlockReaderThread(String clientId,
652         HFileBlock.FSReader hbr, List<Long> offsets, List<BlockType> types,
653         long fileSize) {
654       this.clientId = clientId;
655       this.offsets = offsets;
656       this.hbr = hbr;
657       this.types = types;
658       this.fileSize = fileSize;
659     }
660 
661     @Override
662     public Boolean call() throws Exception {
663       Random rand = new Random(clientId.hashCode());
664       long endTime = System.currentTimeMillis() + 10000;
665       int numBlocksRead = 0;
666       int numPositionalRead = 0;
667       int numWithOnDiskSize = 0;
668       while (System.currentTimeMillis() < endTime) {
669         int blockId = rand.nextInt(NUM_TEST_BLOCKS);
670         long offset = offsets.get(blockId);
671         boolean pread = rand.nextBoolean();
672         boolean withOnDiskSize = rand.nextBoolean();
673         long expectedSize =
674           (blockId == NUM_TEST_BLOCKS - 1 ? fileSize
675               : offsets.get(blockId + 1)) - offset;
676 
677         HFileBlock b;
678         try {
679           long onDiskSizeArg = withOnDiskSize ? expectedSize : -1;
680           b = hbr.readBlockData(offset, onDiskSizeArg, -1, pread);
681         } catch (IOException ex) {
682           LOG.error("Error in client " + clientId + " trying to read block at "
683               + offset + ", pread=" + pread + ", withOnDiskSize=" +
684               withOnDiskSize, ex);
685           return false;
686         }
687 
688         assertEquals(types.get(blockId), b.getBlockType());
689         assertEquals(expectedSize, b.getOnDiskSizeWithHeader());
690         assertEquals(offset, b.getOffset());
691 
692         ++numBlocksRead;
693         if (pread)
694           ++numPositionalRead;
695         if (withOnDiskSize)
696           ++numWithOnDiskSize;
697       }
698       LOG.info("Client " + clientId + " successfully read " + numBlocksRead +
699         " blocks (with pread: " + numPositionalRead + ", with onDiskSize " +
700         "specified: " + numWithOnDiskSize + ")");
701 
702       return true;
703     }

754   private long writeBlocks(Random rand, Compression.Algorithm compressAlgo,
755       Path path, List<Long> expectedOffsets, List<Long> expectedPrevOffsets,
756       List<BlockType> expectedTypes, List<ByteBuffer> expectedContents
757   ) throws IOException {
758     boolean cacheOnWrite = expectedContents != null;
759     FSDataOutputStream os = fs.create(path);
760     HFileContext meta = new HFileContextBuilder()
761                         .withHBaseCheckSum(true)
762                         .withIncludesMvcc(includesMemstoreTS)
763                         .withIncludesTags(includesTag)
764                         .withCompression(compressAlgo)
765                         .withBytesPerCheckSum(HFile.DEFAULT_BYTES_PER_CHECKSUM)
766                         .build();
767     HFileBlock.Writer hbw = new HFileBlock.Writer(null, meta);
768     Map<BlockType, Long> prevOffsetByType = new HashMap<BlockType, Long>();
769     long totalSize = 0;
770     for (int i = 0; i < NUM_TEST_BLOCKS; ++i) {
771       long pos = os.getPos();
772       int blockTypeOrdinal = rand.nextInt(BlockType.values().length);
773       if (blockTypeOrdinal == BlockType.ENCODED_DATA.ordinal()) {
774         blockTypeOrdinal = BlockType.DATA.ordinal();
775       }
776       BlockType bt = BlockType.values()[blockTypeOrdinal];
777       DataOutputStream dos = hbw.startWriting(bt);
778       int size = rand.nextInt(500);
779       for (int j = 0; j < size; ++j) {
780         // This might compress well.
781         dos.writeShort(i + 1);
782         dos.writeInt(j + 1);
783       }
784 
785       if (expectedOffsets != null)
786         expectedOffsets.add(os.getPos());
787 
788       if (expectedPrevOffsets != null) {
789         Long prevOffset = prevOffsetByType.get(bt);
790         expectedPrevOffsets.add(prevOffset != null ? prevOffset : -1);
791         prevOffsetByType.put(bt, os.getPos());
792       }
793 
794       expectedTypes.add(bt);
795 
796       hbw.writeHeaderAndData(os);
797       totalSize += hbw.getOnDiskSizeWithHeader();
798 
799       if (cacheOnWrite)
800         expectedContents.add(hbw.getUncompressedBufferWithHeader());
801 
802       if (detailedLogging) {
803         LOG.info("Written block #" + i + " of type " + bt
804             + ", uncompressed size " + hbw.getUncompressedSizeWithoutHeader()
805             + ", packed size " + hbw.getOnDiskSizeWithoutHeader()
806             + " at offset " + pos);
807       }
808     }
809     os.close();
810     LOG.info("Created a temporary file at " + path + ", "
811         + fs.getFileStatus(path).getLen() + " byte, compression=" +
812         compressAlgo);
813     return totalSize;
814   }
