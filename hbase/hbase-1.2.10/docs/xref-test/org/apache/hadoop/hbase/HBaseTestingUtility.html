330   /**
331    * Home our data in a dir under {@link #DEFAULT_BASE_TEST_DIRECTORY}.
332    * Give it a random name so can have many concurrent tests running if
333    * we need to.  It needs to amend the {@link #TEST_DIRECTORY_KEY}
334    * System property, as it's what minidfscluster bases
335    * it data dir on.  Moding a System property is not the way to do concurrent
336    * instances -- another instance could grab the temporary
337    * value unintentionally -- but not anything can do about it at moment;
338    * single instance only is how the minidfscluster works.
339    *
340    * We also create the underlying directory for
341    *  hadoop.log.dir, mapreduce.cluster.local.dir and hadoop.tmp.dir, and set the values
342    *  in the conf, and as a system property for hadoop.tmp.dir
343    *
344    * @return The calculated data test build directory, if newly-created.
345    */
346   @Override
347   protected Path setupDataTestDir() {
348     Path testPath = super.setupDataTestDir();
349     if (null == testPath) {
350       return null;
351     }
352 
353     createSubDirAndSystemProperty(
354       "hadoop.log.dir",
355       testPath, "hadoop-log-dir");
356 
357     // This is defaulted in core-default.xml to /tmp/hadoop-${user.name}, but
358     //  we want our own value to ensure uniqueness on the same machine
359     createSubDirAndSystemProperty(
360       "hadoop.tmp.dir",
361       testPath, "hadoop-tmp-dir");
362 
363     // Read and modified in org.apache.hadoop.mapred.MiniMRCluster
364     createSubDir(
365       "mapreduce.cluster.local.dir",
366       testPath, "mapred-local-dir");
367 
368     return testPath;
369   }

376   private void createSubDirAndSystemProperty(
377     String propertyName, Path parent, String subDirName){
378 
379     String sysValue = System.getProperty(propertyName);
380 
381     if (sysValue != null) {
382       // There is already a value set. So we do nothing but hope
383       //  that there will be no conflicts
384       LOG.info("System.getProperty(\""+propertyName+"\") already set to: "+
385         sysValue + " so I do NOT create it in " + parent);
386       String confValue = conf.get(propertyName);
387       if (confValue != null && !confValue.endsWith(sysValue)){
388        LOG.warn(
389          propertyName + " property value differs in configuration and system: "+
390          "Configuration="+confValue+" while System="+sysValue+
391          " Erasing configuration value by system value."
392        );
393       }
394       conf.set(propertyName, sysValue);
395     } else {
396       // Ok, it's not set, so we create it as a subdirectory
397       createSubDir(propertyName, parent, subDirName);
398       System.setProperty(propertyName, conf.get(propertyName));
399     }
400   }

497     //The file system can be either local, mini dfs, or if the configuration

502     Path newDataTestDir = null;

510       newDataTestDir = new Path(base, randomStr);
511       if (deleteOnExit()) fs.deleteOnExit(newDataTestDir);
512     }
513     return newDataTestDir;
514   }

569   /**
570    * Start a minidfscluster.
571    * Can only create one.
572    * @param servers How many DNs to start.
573    * @param hosts hostnames DNs to run on.
574    * @throws Exception
575    * @see {@link #shutdownMiniDFSCluster()}
576    * @return The mini dfs cluster created.
577    */
578   public MiniDFSCluster startMiniDFSCluster(int servers, final String hosts[])
579   throws Exception {
580     createDirsAndSetProperties();
581     EditLogFileOutputStream.setShouldSkipFsyncForTesting(true);
582 
583     // Error level to skip some warnings specific to the minicluster. See HBASE-4709
584     org.apache.log4j.Logger.getLogger(org.apache.hadoop.metrics2.util.MBeans.class).
585         setLevel(org.apache.log4j.Level.ERROR);
586     org.apache.log4j.Logger.getLogger(org.apache.hadoop.metrics2.impl.MetricsSystemImpl.class).
587         setLevel(org.apache.log4j.Level.ERROR);
588 
589 
590     this.dfsCluster = new MiniDFSCluster(0, this.conf, servers, true, true,
591       true, null, null, hosts, null);
592 
593     // Set this just-started cluster as our filesystem.
594     setFs();
595 
596     // Wait for the cluster to be totally up
597     this.dfsCluster.waitClusterUp();
598 
599     //reset the test directory for test file system
600     dataTestDirOnTestFS = null;

1838   public static void modifyTableSync(Admin admin, HTableDescriptor desc)
1839       throws IOException, InterruptedException {
1840     admin.modifyTable(desc.getTableName(), desc);
1841     Pair<Integer, Integer> status = new Pair<Integer, Integer>() {{
1842       setFirst(0);

2816   /**
2817    * Expire a ZooKeeper session as recommended in ZooKeeper documentation
2818    * http://wiki.apache.org/hadoop/ZooKeeper/FAQ#A4
2819    * There are issues when doing this:
2820    * [1] http://www.mail-archive.com/dev@zookeeper.apache.org/msg01942.html
2821    * [2] https://issues.apache.org/jira/browse/ZOOKEEPER-1105
2822    *
2823    * @param nodeZK - the ZK watcher to expire
2824    * @param checkStatus - true to check if we can create an HTable with the
2825    *                    current configuration.
2826    */
2827   public void expireSession(ZooKeeperWatcher nodeZK, boolean checkStatus)
2828     throws Exception {
2829     Configuration c = new Configuration(this.conf);
2830     String quorumServers = ZKConfig.getZKQuorumServersString(c);
2831     ZooKeeper zk = nodeZK.getRecoverableZooKeeper().getZooKeeper();
2832     byte[] password = zk.getSessionPasswd();
2833     long sessionID = zk.getSessionId();
2834 
2835     // Expiry seems to be asynchronous (see comment from P. Hunt in [1]),
2836     //  so we create a first watcher to be sure that the
2837     //  event was sent. We expect that if our watcher receives the event
2838     //  other watchers on the same machine will get is as well.
2839     // When we ask to close the connection, ZK does not close it before
2840     //  we receive all the events, so don't have to capture the event, just
2841     //  closing the connection should be enough.
2842     ZooKeeper monitor = new ZooKeeper(quorumServers,
2843       1000, new org.apache.zookeeper.Watcher(){
2844       @Override
2845       public void process(WatchedEvent watchedEvent) {
2846         LOG.info("Monitor ZKW received event="+watchedEvent);
2847       }
2848     } , sessionID, password);
2849 
2850     // Making it expire
2851     ZooKeeper newZK = new ZooKeeper(quorumServers,
2852         1000, EmptyWatcher.instance, sessionID, password);
2853 
2854     //ensure that we have connection to the server before closing down, otherwise
2855     //the close session event will be eaten out before we start CONNECTING state
2856     long start = System.currentTimeMillis();
2857     while (newZK.getState() != States.CONNECTED
2858          && System.currentTimeMillis() - start < 1000) {
2859        Thread.sleep(1);
2860     }
2861     newZK.close();
2862     LOG.info("ZK Closed Session 0x" + Long.toHexString(sessionID));
2863 
2864     // Now closing & waiting to be sure that the clients get it.
2865     monitor.close();
2866 
2867     if (checkStatus) {
2868       new HTable(new Configuration(conf), TableName.META_TABLE_NAME).close();
2869     }
2870   }

3013   /*
3014    * Retrieves a splittable region randomly from tableName
3015    *
3016    * @param tableName name of table
3017    * @param maxAttempts maximum number of attempts, unlimited for value of -1
3018    * @return the HRegion chosen, null if none was found within limit of maxAttempts
3019    */
3020   public HRegion getSplittableRegion(TableName tableName, int maxAttempts) {
3021     List<HRegion> regions = getHBaseCluster().getRegions(tableName);
3022     int regCount = regions.size();
3023     Set<Integer> attempted = new HashSet<Integer>();
3024     int idx;
3025     int attempts = 0;
3026     do {
3027       regions = getHBaseCluster().getRegions(tableName);
3028       if (regCount != regions.size()) {
3029         // if there was region movement, clear attempted Set
3030         attempted.clear();
3031       }
3032       regCount = regions.size();
3033       // There are chances that before we get the region for the table from an RS the region may
3034       // be going for CLOSE.  This may be because online schema change is enabled
3035       if (regCount > 0) {
3036         idx = random.nextInt(regCount);
3037         // if we have just tried this region, there is no need to try again
3038         if (attempted.contains(idx))
3039           continue;
3040         try {
3041           regions.get(idx).checkSplit();
3042           return regions.get(idx);
3043         } catch (Exception ex) {
3044           LOG.warn("Caught exception", ex);
3045           attempted.add(idx);
3046         }
3047       }
3048       attempts++;
3049     } while (maxAttempts == -1 || attempts < maxAttempts);
3050     return null;
3051   }
