119   /**
120    * Simple mapper that makes KeyValue output.
121    */
122   static class RandomKVGeneratingMapper extends
123       Mapper<NullWritable, NullWritable, ImmutableBytesWritable, KeyValue> {
124 
125     private int keyLength;
126     private static final int KEYLEN_DEFAULT = 10;
127     private static final String KEYLEN_CONF = "randomkv.key.length";
128 
129     private int valLength;
130     private static final int VALLEN_DEFAULT=10;
131     private static final String VALLEN_CONF="randomkv.val.length";
132     private static final byte [] QUALIFIER = Bytes.toBytes("data");
133 
134     @Override
135     protected void setup(Context context) throws IOException,
136         InterruptedException {
137       super.setup(context);
138 
139       Configuration conf = context.getConfiguration();
140       keyLength = conf.getInt(KEYLEN_CONF, KEYLEN_DEFAULT);
141       valLength = conf.getInt(VALLEN_CONF, VALLEN_DEFAULT);
142     }
143 
144     protected void map(
145         NullWritable n1, NullWritable n2,
146         Mapper<NullWritable, NullWritable,
147                ImmutableBytesWritable,KeyValue>.Context context)
148         throws java.io.IOException ,InterruptedException
149     {
150 
151       byte keyBytes[] = new byte[keyLength];
152       byte valBytes[] = new byte[valLength];
153 
154       int taskId = context.getTaskAttemptID().getTaskID().getId();
155       assert taskId < Byte.MAX_VALUE : "Unit tests dont support > 127 tasks!";
156 
157       Random random = new Random();
158       for (int i = 0; i < ROWSPERSPLIT; i++) {
159 
160         random.nextBytes(keyBytes);
161         // Ensure that unique tasks generate unique keys
162         keyBytes[keyLength - 1] = (byte)(taskId & 0xFF);
163         random.nextBytes(valBytes);
164         ImmutableBytesWritable key = new ImmutableBytesWritable(keyBytes);
165 
166         for (byte[] family : TestHFileOutputFormat.FAMILIES) {
167           KeyValue kv = new KeyValue(keyBytes, family, QUALIFIER, valBytes);
168           context.write(key, kv);
169         }
170       }
171     }
172   }

175     job.setInputFormatClass(NMapInputFormat.class);
176     job.setMapperClass(RandomKVGeneratingMapper.class);
177     job.setMapOutputKeyClass(ImmutableBytesWritable.class);
178     job.setMapOutputValueClass(KeyValue.class);
179   }

181   /**
182    * Test that {@link HFileOutputFormat} RecordWriter amends timestamps if
183    * passed a keyvalue whose timestamp is {@link HConstants#LATEST_TIMESTAMP}.
184    * @see <a href="https://issues.apache.org/jira/browse/HBASE-2615">HBASE-2615</a>
185    */
186   @Test
187   public void test_LATEST_TIMESTAMP_isReplaced()
188   throws Exception {
189     Configuration conf = new Configuration(this.util.getConfiguration());
190     RecordWriter<ImmutableBytesWritable, KeyValue> writer = null;
191     TaskAttemptContext context = null;
192     Path dir =
193       util.getDataTestDir("test_LATEST_TIMESTAMP_isReplaced");
194     try {
195       Job job = new Job(conf);
196       FileOutputFormat.setOutputPath(job, dir);
197       context = createTestTaskAttemptContext(job);
198       HFileOutputFormat hof = new HFileOutputFormat();
199       writer = hof.getRecordWriter(context);
200       final byte [] b = Bytes.toBytes("b");
201 
202       // Test 1.  Pass a KV that has a ts of LATEST_TIMESTAMP.  It should be
203       // changed by call to write.  Check all in kv is same but ts.
204       KeyValue kv = new KeyValue(b, b, b);
205       KeyValue original = kv.clone();
206       writer.write(new ImmutableBytesWritable(), kv);
207       assertFalse(original.equals(kv));
208       assertTrue(Bytes.equals(original.getRow(), kv.getRow()));
209       assertTrue(CellUtil.matchingColumn(original, kv.getFamily(), kv.getQualifier()));
210       assertNotSame(original.getTimestamp(), kv.getTimestamp());
211       assertNotSame(HConstants.LATEST_TIMESTAMP, kv.getTimestamp());
212 
213       // Test 2. Now test passing a kv that has explicit ts.  It should not be
214       // changed by call to record write.
215       kv = new KeyValue(b, b, b, kv.getTimestamp() - 1, b);
216       original = kv.clone();
217       writer.write(new ImmutableBytesWritable(), kv);
218       assertTrue(original.equals(kv));
219     } finally {
220       if (writer != null && context != null) writer.close(context);
221       dir.getFileSystem(conf).delete(dir, true);
222     }
223   }

228     TaskAttemptContext context = hadoop.createTestTaskAttemptContext(job, "attempt_200707121733_0001_m_000000_0");

232   /*
233    * Test that {@link HFileOutputFormat} creates an HFile with TIMERANGE
234    * metadata used by time-restricted scans.
235    */
236   @Test
237   public void test_TIMERANGE() throws Exception {
238     Configuration conf = new Configuration(this.util.getConfiguration());
239     RecordWriter<ImmutableBytesWritable, KeyValue> writer = null;
240     TaskAttemptContext context = null;
241     Path dir =
242       util.getDataTestDir("test_TIMERANGE_present");
243     LOG.info("Timerange dir writing to dir: "+ dir);
244     try {
245       // build a record writer using HFileOutputFormat
246       Job job = new Job(conf);
247       FileOutputFormat.setOutputPath(job, dir);
248       context = createTestTaskAttemptContext(job);
249       HFileOutputFormat hof = new HFileOutputFormat();
250       writer = hof.getRecordWriter(context);
251 
252       // Pass two key values with explicit times stamps
253       final byte [] b = Bytes.toBytes("b");
254 
255       // value 1 with timestamp 2000
256       KeyValue kv = new KeyValue(b, b, b, 2000, b);
257       KeyValue original = kv.clone();
258       writer.write(new ImmutableBytesWritable(), kv);
259       assertEquals(original,kv);
260 
261       // value 2 with timestamp 1000
262       kv = new KeyValue(b, b, b, 1000, b);
263       original = kv.clone();
264       writer.write(new ImmutableBytesWritable(), kv);
265       assertEquals(original, kv);
266 
267       // verify that the file has the proper FileInfo.
268       writer.close(context);
269 
270       // the generated file lives 1 directory down from the attempt directory
271       // and is the only file, e.g.
272       // _attempt__0000_r_000000_0/b/1979617994050536795
273       FileSystem fs = FileSystem.get(conf);
274       Path attemptDirectory = hof.getDefaultWorkFile(context, "").getParent();
275       FileStatus[] sub1 = fs.listStatus(attemptDirectory);
276       FileStatus[] file = fs.listStatus(sub1[0].getPath());
277 
278       // open as HFile Reader and pull out TIMERANGE FileInfo.
279       HFile.Reader rd = HFile.createReader(fs, file[0].getPath(),
280           new CacheConfig(conf), conf);
281       Map<byte[],byte[]> finfo = rd.loadFileInfo();
282       byte[] range = finfo.get("TIMERANGE".getBytes());
283       assertNotNull(range);
284 
285       // unmarshall and check values.
286       TimeRangeTracker timeRangeTracker = new TimeRangeTracker();
287       Writables.copyWritable(range, timeRangeTracker);
288       LOG.info(timeRangeTracker.getMinimumTimestamp() +
289           "...." + timeRangeTracker.getMaximumTimestamp());
290       assertEquals(1000, timeRangeTracker.getMinimumTimestamp());
291       assertEquals(2000, timeRangeTracker.getMaximumTimestamp());
292       rd.close();
293     } finally {
294       if (writer != null && context != null) writer.close(context);
295       dir.getFileSystem(conf).delete(dir, true);
296     }
297   }

387   private void doIncrementalLoadTest(
388       boolean shouldChangeRegions) throws Exception {
389     util = new HBaseTestingUtility();
390     Configuration conf = util.getConfiguration();
391     byte[][] splitKeys = generateRandomSplitKeys(4);
392     HBaseAdmin admin = null;
393     try {
394       util.setJobWithoutMRCluster();
395       util.startMiniCluster();
396       Path testDir = util.getDataTestDirOnTestFS("testLocalMRIncrementalLoad");
397       admin = util.getHBaseAdmin();
398       HTable table = util.createTable(TABLE_NAME, FAMILIES, splitKeys);
399       assertEquals("Should start with empty table",
400           0, util.countRows(table));
401       int numRegions = -1;
402       try(RegionLocator r = table.getRegionLocator()) {
403         numRegions = r.getStartKeys().length;
404       }
405       assertEquals("Should make 5 regions", numRegions, 5);
406 
407       // Generate the bulk load files
408       runIncrementalPELoad(conf, table, testDir);
409       // This doesn't write into the table, just makes files
410       assertEquals("HFOF should not touch actual table",
411           0, util.countRows(table));
412 
413 
414       // Make sure that a directory was created for every CF
415       int dir = 0;
416       for (FileStatus f : testDir.getFileSystem(conf).listStatus(testDir)) {
417         for (byte[] family : FAMILIES) {
418           if (Bytes.toString(family).equals(f.getPath().getName())) {
419             ++dir;
420           }
421         }
422       }
423       assertEquals("Column family not found in FS.", FAMILIES.length, dir);
424 
425       // handle the split case
426       if (shouldChangeRegions) {
427         LOG.info("Changing regions in table");
428         admin.disableTable(table.getTableName());
429         while(util.getMiniHBaseCluster().getMaster().getAssignmentManager().
430             getRegionStates().isRegionsInTransition()) {
431           Threads.sleep(200);
432           LOG.info("Waiting on table to finish disabling");
433         }
434         util.deleteTable(table.getName());
435         byte[][] newSplitKeys = generateRandomSplitKeys(14);
436         table = util.createTable(TABLE_NAME, FAMILIES, newSplitKeys);
437         while (table.getRegionLocations().size() != 15 ||
438             !admin.isTableAvailable(table.getTableName())) {
439           Thread.sleep(200);
440           LOG.info("Waiting for new region assignment to happen");
441         }
442       }
443 
444       // Perform the actual load
445       new LoadIncrementalHFiles(conf).doBulkLoad(testDir, table);
446 
447       // Ensure data shows up
448       int expectedRows = NMapInputFormat.getNumMapTasks(conf) * ROWSPERSPLIT;
449       assertEquals("LoadIncrementalHFiles should put expected data in table",
450           expectedRows, util.countRows(table));
451       Scan scan = new Scan();
452       ResultScanner results = table.getScanner(scan);
453       for (Result res : results) {
454         assertEquals(FAMILIES.length, res.rawCells().length);
455         Cell first = res.rawCells()[0];
456         for (Cell kv : res.rawCells()) {
457           assertTrue(CellUtil.matchingRow(first, kv));
458           assertTrue(Bytes.equals(CellUtil.cloneValue(first), CellUtil.cloneValue(kv)));
459         }
460       }
461       results.close();
462       String tableDigestBefore = util.checksumRows(table);
463 
464       // Cause regions to reopen
465       admin.disableTable(TABLE_NAME);
466       while (!admin.isTableDisabled(TABLE_NAME)) {
467         Thread.sleep(200);
468         LOG.info("Waiting for table to disable");
469       }
470       admin.enableTable(TABLE_NAME);
471       util.waitTableAvailable(TABLE_NAME);
472       assertEquals("Data should remain after reopening of regions",
473           tableDigestBefore, util.checksumRows(table));
474     } finally {
475       if (admin != null) admin.close();
476       util.shutdownMiniCluster();
477     }
478   }

539           .setCompressionType(entry.getValue())
540           .setBlockCacheEnabled(false)
541           .setTimeToLive(0));
542     }
543     Mockito.doReturn(mockTableDescriptor).when(table).getTableDescriptor();
544   }

548    *         testing column family compression. Column family names have special characters
549    */
550   private Map<String, Compression.Algorithm>
551       getMockColumnFamiliesForCompression (int numCfs) {
552     Map<String, Compression.Algorithm> familyToCompression = new HashMap<String, Compression.Algorithm>();
553     // use column family names having special characters

555       familyToCompression.put("Family1!@#!@#&", Compression.Algorithm.LZO);
556     }
557     if (numCfs-- > 0) {
558       familyToCompression.put("Family2=asdads&!AASD", Compression.Algorithm.SNAPPY);
559     }
560     if (numCfs-- > 0) {
561       familyToCompression.put("Family2=asdads&!AASD", Compression.Algorithm.GZ);
562     }
563     if (numCfs-- > 0) {
564       familyToCompression.put("Family3", Compression.Algorithm.NONE);
565     }
566     return familyToCompression;
