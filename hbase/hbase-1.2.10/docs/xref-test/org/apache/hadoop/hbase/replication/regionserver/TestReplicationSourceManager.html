202   @Test
203   public void testLogRoll() throws Exception {
204     long baseline = 1000;
205     long time = baseline;
206     MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
207     KeyValue kv = new KeyValue(r1, f1, r1);
208     WALEdit edit = new WALEdit();
209     edit.add(kv);
210 
211     List<WALActionsListener> listeners = new ArrayList<WALActionsListener>();
212     listeners.add(replication);
213     final WALFactory wals = new WALFactory(utility.getConfiguration(), listeners,
214         URLEncoder.encode("regionserver:60020", "UTF8"));
215     final WAL wal = wals.getWAL(hri.getEncodedNameAsBytes());
216     manager.init();
217     HTableDescriptor htd = new HTableDescriptor();
218     htd.addFamily(new HColumnDescriptor(f1));
219     // Testing normal log rolling every 20
220     for(long i = 1; i < 101; i++) {
221       if(i > 1 && i % 20 == 0) {
222         wal.rollWriter();
223       }
224       LOG.info(i);
225       final long txid = wal.append(htd,
226           hri,
227           new WALKey(hri.getEncodedNameAsBytes(), test, System.currentTimeMillis(), mvcc),
228           edit,
229           true);
230       wal.sync(txid);
231     }
232 
233     // Simulate a rapid insert that's followed
234     // by a report that's still not totally complete (missing last one)
235     LOG.info(baseline + " and " + time);
236     baseline += 101;
237     time = baseline;
238     LOG.info(baseline + " and " + time);
239 
240     for (int i = 0; i < 3; i++) {
241       wal.append(htd, hri,
242           new WALKey(hri.getEncodedNameAsBytes(), test, System.currentTimeMillis(), mvcc),
243           edit,
244           true);
245     }
246     wal.sync();
247 
248     int logNumber = 0;
249     for (Map.Entry<String, SortedSet<String>> entry : manager.getWALs().get(slaveId).entrySet()) {
250       logNumber += entry.getValue().size();
251     }
252     assertEquals(6, logNumber);
253 
254     wal.rollWriter();
255 
256     manager.logPositionAndCleanOldLogs(manager.getSources().get(0).getCurrentPath(),
257         "1", 0, false, false);
258 
259     wal.append(htd, hri,
260         new WALKey(hri.getEncodedNameAsBytes(), test, System.currentTimeMillis(), mvcc),
261         edit,
262         true);
263     wal.sync();
264 
265     assertEquals(1, manager.getWALs().size());
266 
267 
268     // TODO Need a case with only 2 WALs and we only want to delete the first one
269   }

272   public void testClaimQueues() throws Exception {
273     conf.setBoolean(HConstants.ZOOKEEPER_USEMULTI, true);
274     final Server server = new DummyServer("hostname0.example.org");
275     ReplicationQueues rq =
276         ReplicationFactory.getReplicationQueues(server.getZooKeeper(), server.getConfiguration(),
277           server);
278     rq.init(server.getServerName().toString());
279     // populate some znodes in the peer znode
280     files.add("log1");
281     files.add("log2");
282     for (String file : files) {
283       rq.addLog("1", file);
284     }
285     // create 3 DummyServers
286     Server s1 = new DummyServer("dummyserver1.example.org");
287     Server s2 = new DummyServer("dummyserver2.example.org");
288     Server s3 = new DummyServer("dummyserver3.example.org");
289 
290     // create 3 DummyNodeFailoverWorkers
291     DummyNodeFailoverWorker w1 = new DummyNodeFailoverWorker(
292         server.getServerName().getServerName(), s1);
293     DummyNodeFailoverWorker w2 = new DummyNodeFailoverWorker(
294         server.getServerName().getServerName(), s2);
295     DummyNodeFailoverWorker w3 = new DummyNodeFailoverWorker(
296         server.getServerName().getServerName(), s3);
297 
298     latch = new CountDownLatch(3);
299     // start the threads
300     w1.start();
301     w2.start();
302     w3.start();
303     // make sure only one is successful
304     int populatedMap = 0;
305     // wait for result now... till all the workers are done.
306     latch.await();
307     populatedMap += w1.isLogZnodesMapPopulated() + w2.isLogZnodesMapPopulated()
308         + w3.isLogZnodesMapPopulated();
309     assertEquals(1, populatedMap);
310     server.abort("", null);
311   }

324     String file2 = group + ".log2";
325     files.add(file1);
326     files.add(file2);
327     for (String file : files) {
328       rq.addLog("1", file);
329     }
330     Server s1 = new DummyServer("dummyserver1.example.org");
331     ReplicationQueues rq1 =
332         ReplicationFactory.getReplicationQueues(s1.getZooKeeper(), s1.getConfiguration(), s1);

334     ReplicationPeers rp1 =

336     rp1.init();

338         manager.new NodeFailoverWorker(server.getServerName().getServerName(), rq1, rp1, new UUID(
339             new Long(1), new Long(2)));
340     w1.start();
341     w1.join(5000);
342     assertEquals(1, manager.getWalsByIdRecoveredQueues().size());
343     String id = "1-" + server.getServerName().getServerName();
344     assertEquals(files, manager.getWalsByIdRecoveredQueues().get(id).get(group));

347     assertEquals(Sets.newHashSet(file2), manager.getWalsByIdRecoveredQueues().get(id).get(group));

451   static class DummyNodeFailoverWorker extends Thread {
452     private SortedMap<String, SortedSet<String>> logZnodesMap;
453     Server server;
454     private String deadRsZnode;
455     ReplicationQueues rq;
456 
457     public DummyNodeFailoverWorker(String znode, Server s) throws Exception {
458       this.deadRsZnode = znode;
459       this.server = s;
460       this.rq =
461           ReplicationFactory.getReplicationQueues(server.getZooKeeper(), server.getConfiguration(),
462             server);
463       this.rq.init(this.server.getServerName().toString());
464     }
465 
466     @Override
467     public void run() {
468       try {
469         logZnodesMap = rq.claimQueues(deadRsZnode);
470         server.abort("Done with testing", null);
471       } catch (Exception e) {
472         LOG.error("Got exception while running NodeFailoverWorker", e);
473       } finally {
474         latch.countDown();
475       }
476     }
477 
478     /**
479      * @return 1 when the map is not empty.
480      */
481     private int isLogZnodesMapPopulated() {
482       Collection<SortedSet<String>> sets = logZnodesMap.values();
483       if (sets.size() > 1) {
484         throw new RuntimeException("unexpected size of logZnodesMap: " + sets.size());
485       }
486       if (sets.size() == 1) {
487         SortedSet<String> s = sets.iterator().next();
488         for (String file : files) {
489           // at least one file was missing
490           if (!s.contains(file)) {
491             return 0;
492           }
493         }
494         return 1; // we found all the files
495       }
496       return 0;
497     }
498   }
