94    public void testPutSameCell() {
95      byte[] bytes = Bytes.toBytes(getName());
96      KeyValue kv = new KeyValue(bytes, bytes, bytes, bytes);
97      long sizeChangeForFirstCell = this.memstore.add(kv);
98      long sizeChangeForSecondCell = this.memstore.add(kv);
99      // make sure memstore size increase won't double-count MSLAB chunk size
100     assertEquals(DefaultMemStore.heapSizeChange(kv, true), sizeChangeForFirstCell);
101     if (this.memstore.allocator != null) {
102       // make sure memstore size increased when using MSLAB
103       assertEquals(memstore.getCellLength(kv), sizeChangeForSecondCell);
104       // make sure chunk size increased even when writing the same cell, if using MSLAB
105       if (this.memstore.allocator instanceof HeapMemStoreLAB) {
106         assertEquals(2 * memstore.getCellLength(kv),
107           ((HeapMemStoreLAB) this.memstore.allocator).getCurrentChunk().getNextFreeOffset());
108       }
109     } else {
110       // make sure no memstore size change w/o MSLAB
111       assertEquals(0, sizeChangeForSecondCell);
112     }

119   public void testScanAcrossSnapshot() throws IOException {
120     int rowCount = addRows(this.memstore);
121     List<KeyValueScanner> memstorescanners = this.memstore.getScanners(0);
122     Scan scan = new Scan();
123     List<Cell> result = new ArrayList<Cell>();
124     Configuration conf = HBaseConfiguration.create();
125     ScanInfo scanInfo =
126         new ScanInfo(conf, null, 0, 1, HConstants.LATEST_TIMESTAMP, KeepDeletedCells.FALSE, 0,
127             this.memstore.comparator);
128     ScanType scanType = ScanType.USER_SCAN;
129     StoreScanner s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
130     int count = 0;
131     try {
132       while (s.next(result)) {
133         LOG.info(result);
134         count++;
135         // Row count is same as column count.
136         assertEquals(rowCount, result.size());
137         result.clear();
138       }
139     } finally {
140       s.close();
141     }
142     assertEquals(rowCount, count);
143     for (KeyValueScanner scanner : memstorescanners) {
144       scanner.close();
145     }
146 
147     memstorescanners = this.memstore.getScanners(mvcc.getReadPoint());
148     // Now assert can count same number even if a snapshot mid-scan.
149     s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
150     count = 0;
151     try {
152       while (s.next(result)) {
153         LOG.info(result);
154         // Assert the stuff is coming out in right order.
155         assertTrue(CellUtil.matchingRow(result.get(0), Bytes.toBytes(count)));
156         count++;
157         // Row count is same as column count.
158         assertEquals(rowCount, result.size());
159         if (count == 2) {
160           this.memstore.snapshot();
161           LOG.info("Snapshotted");
162         }
163         result.clear();
164       }
165     } finally {
166       s.close();
167     }
168     assertEquals(rowCount, count);
169     for (KeyValueScanner scanner : memstorescanners) {
170       scanner.close();
171     }
172     memstorescanners = this.memstore.getScanners(mvcc.getReadPoint());
173     // Assert that new values are seen in kvset as we scan.
174     long ts = System.currentTimeMillis();
175     s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
176     count = 0;
177     int snapshotIndex = 5;
178     try {
179       while (s.next(result)) {
180         LOG.info(result);
181         // Assert the stuff is coming out in right order.
182         assertTrue(CellUtil.matchingRow(result.get(0), Bytes.toBytes(count)));
183         // Row count is same as column count.
184         assertEquals("count=" + count + ", result=" + result, rowCount, result.size());
185         count++;
186         if (count == snapshotIndex) {
187           MemStoreSnapshot snapshot = this.memstore.snapshot();
188           this.memstore.clearSnapshot(snapshot.getId());
189           // Added more rows into kvset.  But the scanner wont see these rows.
190           addRows(this.memstore, ts);
191           LOG.info("Snapshotted, cleared it and then added values (which wont be seen)");
192         }
193         result.clear();
194       }
195     } finally {
196       s.close();
197     }
198     assertEquals(rowCount, count);
199   }

206   public void testScanAcrossSnapshot2() throws IOException, CloneNotSupportedException {
207     // we are going to the scanning across snapshot with two kvs
208     // kv1 should always be returned before kv2
209     final byte[] one = Bytes.toBytes(1);
210     final byte[] two = Bytes.toBytes(2);
211     final byte[] f = Bytes.toBytes("f");
212     final byte[] q = Bytes.toBytes("q");
213     final byte[] v = Bytes.toBytes(3);
214 
215     final KeyValue kv1 = new KeyValue(one, f, q, v);
216     final KeyValue kv2 = new KeyValue(two, f, q, v);
217 
218     // use case 1: both kvs in kvset
219     this.memstore.add(kv1.clone());
220     this.memstore.add(kv2.clone());
221     verifyScanAcrossSnapshot2(kv1, kv2);
222 
223     // use case 2: both kvs in snapshot
224     this.memstore.snapshot();
225     verifyScanAcrossSnapshot2(kv1, kv2);
226 
227     // use case 3: first in snapshot second in kvset
228     this.memstore = new DefaultMemStore();
229     this.memstore.add(kv1.clone());
230     this.memstore.snapshot();
231     this.memstore.add(kv2.clone());
232     verifyScanAcrossSnapshot2(kv1, kv2);
233   }

239     final KeyValueScanner scanner = memstorescanners.get(0);
240     scanner.seek(KeyValueUtil.createFirstOnRow(HConstants.EMPTY_START_ROW));
241     assertEquals(kv1, scanner.next());

246   private void assertScannerResults(KeyValueScanner scanner, KeyValue[] expected)
247       throws IOException {
248     scanner.seek(KeyValueUtil.createFirstOnRow(new byte[]{}));
249     List<Cell> returned = Lists.newArrayList();
250 
251     while (true) {
252       Cell next = scanner.next();
253       if (next == null) break;
254       returned.add(next);
255     }
256 
257     assertTrue(
258         "Got:\n" + Joiner.on("\n").join(returned) +
259         "\nExpected:\n" + Joiner.on("\n").join(expected),
260         Iterables.elementsEqual(Arrays.asList(expected), returned));
261     assertNull(scanner.peek());
262   }

300    * Regression test for HBASE-2616, HBASE-2670.
301    * When we insert a higher-memstoreTS version of a cell but with
302    * the same timestamp, we still need to provide consistent reads
303    * for the same scanner.
304    */
305   public void testMemstoreEditsVisibilityWithSameKey() throws IOException {
306     final byte[] row = Bytes.toBytes(1);
307     final byte[] f = Bytes.toBytes("family");
308     final byte[] q1 = Bytes.toBytes("q1");
309     final byte[] q2 = Bytes.toBytes("q2");
310     final byte[] v1 = Bytes.toBytes("value1");
311     final byte[] v2 = Bytes.toBytes("value2");
312 
313     // INSERT 1: Write both columns val1
314     MultiVersionConcurrencyControl.WriteEntry w =
315         mvcc.begin();
316 
317     KeyValue kv11 = new KeyValue(row, f, q1, v1);
318     kv11.setSequenceId(w.getWriteNumber());
319     memstore.add(kv11);
320 
321     KeyValue kv12 = new KeyValue(row, f, q2, v1);
322     kv12.setSequenceId(w.getWriteNumber());
323     memstore.add(kv12);
324     mvcc.completeAndWait(w);
325 
326     // BEFORE STARTING INSERT 2, SEE FIRST KVS
327     KeyValueScanner s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
328     assertScannerResults(s, new KeyValue[]{kv11, kv12});
329 
330     // START INSERT 2: Write both columns val2
331     w = mvcc.begin();
332     KeyValue kv21 = new KeyValue(row, f, q1, v2);
333     kv21.setSequenceId(w.getWriteNumber());
334     memstore.add(kv21);
335 
336     KeyValue kv22 = new KeyValue(row, f, q2, v2);
337     kv22.setSequenceId(w.getWriteNumber());
338     memstore.add(kv22);
339 
340     // BEFORE COMPLETING INSERT 2, SEE FIRST KVS
341     s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
342     assertScannerResults(s, new KeyValue[]{kv11, kv12});
343 
344     // COMPLETE INSERT 2
345     mvcc.completeAndWait(w);
346 
347     // NOW SHOULD SEE NEW KVS IN ADDITION TO OLD KVS.
348     // See HBASE-1485 for discussion about what we should do with
349     // the duplicate-TS inserts
350     s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
351     assertScannerResults(s, new KeyValue[]{kv21, kv11, kv22, kv12});
352   }

402   private static class ReadOwnWritesTester extends Thread {
403     static final int NUM_TRIES = 1000;
404 
405     final byte[] row;
406 
407     final byte[] f = Bytes.toBytes("family");
408     final byte[] q1 = Bytes.toBytes("q1");
409 
410     final MultiVersionConcurrencyControl mvcc;
411     final MemStore memstore;
412     final AtomicLong startSeqNum;
413 
414     AtomicReference<Throwable> caughtException;
415 
416 
417     public ReadOwnWritesTester(int id,
418                                MemStore memstore,
419                                MultiVersionConcurrencyControl mvcc,
420                                AtomicReference<Throwable> caughtException,
421                                AtomicLong startSeqNum)
422     {
423       this.mvcc = mvcc;
424       this.memstore = memstore;
425       this.caughtException = caughtException;
426       row = Bytes.toBytes(id);
427       this.startSeqNum = startSeqNum;
428     }
429 
430     public void run() {
431       try {
432         internalRun();
433       } catch (Throwable t) {
434         caughtException.compareAndSet(null, t);
435       }
436     }
437 
438     private void internalRun() throws IOException {
439       for (long i = 0; i < NUM_TRIES && caughtException.get() == null; i++) {
440         MultiVersionConcurrencyControl.WriteEntry w =
441             mvcc.begin();
442 
443         // Insert the sequence value (i)
444         byte[] v = Bytes.toBytes(i);
445 
446         KeyValue kv = new KeyValue(row, f, q1, i, v);
447         kv.setSequenceId(w.getWriteNumber());
448         memstore.add(kv);
449         mvcc.completeAndWait(w);
450 
451         // Assert that we can read back
452         KeyValueScanner s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
453         s.seek(kv);
454 
455         Cell ret = s.next();
456         assertNotNull("Didnt find own write at all", ret);
457         assertEquals("Didnt read own writes",
458                      kv.getTimestamp(), ret.getTimestamp());
459       }
460     }
461   }

463   public void testReadOwnWritesUnderConcurrency() throws Throwable {
464 
465     int NUM_THREADS = 8;
466 
467     ReadOwnWritesTester threads[] = new ReadOwnWritesTester[NUM_THREADS];
468     AtomicReference<Throwable> caught = new AtomicReference<Throwable>();
469 
470     for (int i = 0; i < NUM_THREADS; i++) {
471       threads[i] = new ReadOwnWritesTester(i, memstore, mvcc, caught, this.startSeqNum);
472       threads[i].start();
473     }
474 
475     for (int i = 0; i < NUM_THREADS; i++) {
476       threads[i].join();
477     }
478 
479     if (caught.get() != null) {
480       throw caught.get();
481     }
482   }

525   public void testGetNextRow() throws Exception {
526     addRows(this.memstore);
527     // Add more versions to make it a little more interesting.

534       Cell nr = this.memstore.getNextRow(new KeyValue(Bytes.toBytes(i),

536       if (i + 1 == ROW_COUNT) {
537         assertEquals(nr, null);
538       } else {
539         assertTrue(KeyValue.COMPARATOR.compareRows(nr,
540           new KeyValue(Bytes.toBytes(i + 1), System.currentTimeMillis())) == 0);

542     }
543     //starting from each row, validate results should contain the starting row
544     Configuration conf = HBaseConfiguration.create();
545     for (int startRowId = 0; startRowId < ROW_COUNT; startRowId++) {
546       ScanInfo scanInfo = new ScanInfo(conf, FAMILY, 0, 1, Integer.MAX_VALUE,
547         KeepDeletedCells.FALSE, 0, this.memstore.comparator);
548       ScanType scanType = ScanType.USER_SCAN;

550           Bytes.toBytes(startRowId)), scanInfo, scanType, null,
551           memstore.getScanners(0));
552       List<Cell> results = new ArrayList<Cell>();
553       for (int i = 0; scanner.next(results); i++) {
554         int rowId = startRowId + i;
555         Cell left = results.get(0);
556         byte[] row1 = Bytes.toBytes(rowId);
557         assertTrue(
558             "Row name",
559             KeyValue.COMPARATOR.compareRows(left.getRowArray(), left.getRowOffset(),
560                 (int) left.getRowLength(), row1, 0, row1.length) == 0);
561         assertEquals("Count of columns", QUALIFIER_COUNT, results.size());
562         List<Cell> row = new ArrayList<Cell>();
563         for (Cell kv : results) {
564           row.add(kv);
565         }
566         isExpectedRowWithoutTimestamps(rowId, row);
567         // Clear out set.  Otherwise row results accumulate.
568         results.clear();
569       }
570     }
